## Task 186: Distributed Tracing for Data Pipelines — Observability at Scale

### Business Context
NovaMart's data platform has 80+ pipelines across Airflow, Kafka, Spark, and dbt. When data is late or wrong, tracing the root cause through this chain is painful: "The dashboard is stale → which dbt model? → which Airflow DAG? → which Kafka topic? → which source system?" Priya says: "Implement distributed tracing for data. Like microservice tracing (Jaeger/Zipkin), but for data pipelines."

### Objective
Implement distributed tracing across the data pipeline stack: trace a data record from source system through Kafka, Spark, dbt, and into the dashboard with a single trace ID.

### Technical Requirements
- **Tools**: OpenTelemetry, Python, Airflow, Kafka, dbt, Grafana Tempo (or Jaeger)
- **Skills**: Distributed tracing, OpenTelemetry, correlation IDs, observability

### Step-by-Step Guidance

1. **Design the tracing architecture**:
   ```
   Source DB → [Trace: extract_001] → Kafka → [Trace: extract_001]
                                                    ↓
                                        Spark/Flink [Trace: extract_001]
                                                    ↓
                                          dbt [Trace: extract_001]
                                                    ↓
                                      Dashboard [Trace: extract_001]

   Each component propagates the trace_id so you can see the
   full data journey in one trace view.
   ```

2. **Instrument Airflow with OpenTelemetry**:
   ```python
   # tracing/airflow_tracer.py
   from opentelemetry import trace
   from opentelemetry.sdk.trace import TracerProvider
   from opentelemetry.sdk.trace.export import BatchSpanProcessor
   from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter
   from opentelemetry.sdk.resources import Resource

   # Initialize tracer
   resource = Resource.create({"service.name": "novamart-airflow"})
   provider = TracerProvider(resource=resource)
   exporter = OTLPSpanExporter(endpoint="http://tempo:4317")
   provider.add_span_processor(BatchSpanProcessor(exporter))
   trace.set_tracer_provider(provider)

   tracer = trace.get_tracer("novamart.airflow")

   def traced_task(func):
       """Decorator to add tracing to Airflow task functions."""
       def wrapper(*args, **kwargs):
           context = kwargs.get("context", {})
           dag_id = context.get("dag", {}).dag_id if context.get("dag") else "unknown"
           task_id = context.get("task_instance", {}).task_id if context.get("task_instance") else "unknown"

           with tracer.start_as_current_span(
               f"airflow.{dag_id}.{task_id}",
               attributes={
                   "airflow.dag_id": dag_id,
                   "airflow.task_id": task_id,
                   "airflow.execution_date": str(context.get("execution_date", "")),
               }
           ) as span:
               try:
                   result = func(*args, **kwargs)
                   span.set_attribute("rows_processed", result.get("rows", 0) if isinstance(result, dict) else 0)
                   return result
               except Exception as e:
                   span.set_status(trace.Status(trace.StatusCode.ERROR, str(e)))
                   span.record_exception(e)
                   raise
       return wrapper
   ```

3. **Propagate trace context through Kafka**:
   ```python
   # tracing/kafka_propagator.py
   from opentelemetry import context, trace
   from opentelemetry.context.propagation import inject, extract

   class TracedKafkaProducer:
       def __init__(self, producer):
           self.producer = producer
           self.tracer = trace.get_tracer("novamart.kafka")

       def produce(self, topic: str, value: bytes, key: bytes = None):
           with self.tracer.start_as_current_span(
               f"kafka.produce.{topic}",
               attributes={"messaging.system": "kafka", "messaging.destination": topic}
           ) as span:
               # Inject trace context into Kafka headers
               headers = {}
               inject(headers)
               kafka_headers = [(k, v.encode()) for k, v in headers.items()]

               self.producer.produce(
                   topic=topic, value=value, key=key,
                   headers=kafka_headers
               )

   class TracedKafkaConsumer:
       def __init__(self, consumer):
           self.consumer = consumer
           self.tracer = trace.get_tracer("novamart.kafka")

       def poll(self, timeout=1.0):
           msg = self.consumer.poll(timeout)
           if msg and not msg.error():
               # Extract trace context from Kafka headers
               headers = {h[0]: h[1].decode() for h in (msg.headers() or [])}
               ctx = extract(headers)

               with self.tracer.start_as_current_span(
                   f"kafka.consume.{msg.topic()}",
                   context=ctx,
                   attributes={"messaging.system": "kafka",
                               "messaging.source": msg.topic()}
               ):
                   return msg
           return msg
   ```

4. **Instrument dbt with custom logging**:
   ```python
   # tracing/dbt_tracer.py
   """
   dbt doesn't natively support OTel, so we wrap dbt runs
   with trace context and parse the run results.
   """
   import json
   import subprocess
   from opentelemetry import trace

   tracer = trace.get_tracer("novamart.dbt")

   def traced_dbt_run(project_dir: str, models: list, parent_context=None):
       with tracer.start_as_current_span(
           "dbt.run",
           context=parent_context,
           attributes={"dbt.project": project_dir, "dbt.models": str(models)}
       ) as parent_span:
           result = subprocess.run(
               ["dbt", "run", "--select"] + models + ["--project-dir", project_dir],
               capture_output=True, text=True
           )

           # Parse run_results.json for per-model timing
           results_path = f"{project_dir}/target/run_results.json"
           with open(results_path) as f:
               run_results = json.load(f)

           for model_result in run_results["results"]:
               with tracer.start_as_current_span(
                   f"dbt.model.{model_result['unique_id']}",
                   attributes={
                       "dbt.model": model_result["unique_id"],
                       "dbt.status": model_result["status"],
                       "dbt.execution_time": model_result["execution_time"],
                       "dbt.rows_affected": model_result.get("adapter_response", {}).get("rows_affected", 0),
                   }
               ):
                   pass  # Span captures timing

           parent_span.set_attribute("dbt.total_models", len(run_results["results"]))
           parent_span.set_attribute("dbt.success", result.returncode == 0)
   ```

- Ignoring dbt — dbt is often the longest part of the pipeline but the least instrumented

## Task 188: Chaos Engineering for Data — Proactive Failure Testing

### Business Context
NovaMart's data platform has good monitoring and incident response (Tasks 166, 179), but it's reactive. You find problems after they happen. ### Objective

### Technical Requirements
- **Tools**: Python, Airflow, Kafka, Snowflake, Docker

### Step-by-Step Guidance

1. **Define the chaos experiment framework**:
   ```python
   # chaos/experiment.py
   from dataclasses import dataclass, field
   from datetime import datetime
   from enum import Enum

   class ExperimentStatus(Enum):
       PLANNED = "planned"
       RUNNING = "running"
       COMPLETED = "completed"
       ABORTED = "aborted"

   @dataclass
   class ChaosExperiment:
       id: str
       name: str
       hypothesis: str        # What we expect to happen
       method: str            # How we inject the failure
       blast_radius: str      # What's affected
       rollback_plan: str     # How to undo if things go wrong
       steady_state: dict     # Metrics that define "normal"
       status: ExperimentStatus = ExperimentStatus.PLANNED
       results: dict = field(default_factory=dict)
       executed_at: datetime = None
       executed_by: str = ""

       def validate_steady_state(self, current_metrics: dict) -> bool:
           """Check if system is in steady state before experimenting."""
           for metric, expected in self.steady_state.items():
               actual = current_metrics.get(metric)
               if actual is None:
                   return False
               if isinstance(expected, dict):
                   if actual < expected.get("min", float("-inf")):
                       return False
                   if actual > expected.get("max", float("inf")):
                       return False
           return True
   ```

3. **Build the experiment executor**:
   ```python
   # chaos/executor.py
   import subprocess
   import time

   class ChaosExecutor:
       def __init__(self, monitoring_client):
           self.monitoring = monitoring_client

       def run_experiment(self, experiment: ChaosExperiment) -> dict:
           # Step 1: Verify steady state
           current = self.monitoring.get_metrics()
           if not experiment.validate_steady_state(current):
               return {"status": "aborted", "reason": "System not in steady state"}

           experiment.status = ExperimentStatus.RUNNING
           experiment.executed_at = datetime.utcnow()

           # Step 2: Record baseline
           baseline = self.monitoring.snapshot()

           # Step 3: Inject failure
           try:
               self._inject_failure(experiment.method)
               
               # Step 4: Observe (wait and measure)
               observations = []
               for i in range(12):  # Observe for 2 minutes (every 10s)
                   time.sleep(10)
                   obs = self.monitoring.snapshot()
                   observations.append(obs)

               # Step 5: Rollback
               self._rollback(experiment.rollback_plan)

               # Step 6: Wait for recovery
               time.sleep(60)
               recovery = self.monitoring.snapshot()

               experiment.status = ExperimentStatus.COMPLETED
               experiment.results = {
                   "baseline": baseline,
                   "observations": observations,
                   "recovery": recovery,
                   "recovered": experiment.validate_steady_state(
                       self.monitoring.get_metrics()
                   ),
               }

           except Exception as e:
               # Emergency rollback
               self._rollback(experiment.rollback_plan)
               experiment.status = ExperimentStatus.ABORTED
               experiment.results = {"error": str(e)}

           return experiment.results
   ```

## Task 189: Advanced Schema Management — Schema Evolution at Scale

### Business Context
NovaMart's 80+ pipelines consume data from 12 source systems. When a source system changes its schema (adds a column, renames a field, changes a data type), pipelines break. Last month, the Commerce team added a `loyalty_tier` column to the orders table without telling anyone — 3 downstream pipelines failed. Priya says: "Schema evolution needs to be managed, not accidental. Build a schema registry and compatibility enforcement system."

### Objective
Build a comprehensive schema management system: schema registry, compatibility checks, automated migration, and breaking change detection across the entire pipeline.

### Technical Requirements
- **Tools**: Python, JSON Schema/Avro, dbt, SQL, Git
- **Skills**: Schema evolution, compatibility modes, migration strategies, contract testing

### Step-by-Step Guidance

1. **Define schema compatibility rules**:
   ```python
   # schema/compatibility.py
   from enum import Enum
   from dataclasses import dataclass

   class CompatibilityMode(Enum):
       BACKWARD = "backward"     # New schema can read old data
       FORWARD = "forward"       # Old schema can read new data
       FULL = "full"             # Both backward and forward
       NONE = "none"             # No compatibility guarantee

   @dataclass
   class SchemaChange:
       change_type: str     # "add_column", "remove_column", "rename", "type_change"
       field_name: str
       old_value: str = None
       new_value: str = None
       is_breaking: bool = False

   class SchemaCompatibilityChecker:
       def check(self, old_schema: dict, new_schema: dict,
                 mode: CompatibilityMode) -> list[SchemaChange]:
           changes = []
           old_fields = {f["name"]: f for f in old_schema.get("fields", [])}
           new_fields = {f["name"]: f for f in new_schema.get("fields", [])}

           # Removed columns
           for name in old_fields:
               if name not in new_fields:
                   is_breaking = mode in (
                       CompatibilityMode.FORWARD, CompatibilityMode.FULL
                   )
                   changes.append(SchemaChange(
                       change_type="remove_column", field_name=name,
                       old_value=str(old_fields[name]), is_breaking=is_breaking
                   ))

           # Added columns
           for name in new_fields:
               if name not in old_fields:
                   has_default = "default" in new_fields[name]
                   is_breaking = (
                       mode in (CompatibilityMode.BACKWARD, CompatibilityMode.FULL)
                       and not has_default
                   )
                   changes.append(SchemaChange(
                       change_type="add_column", field_name=name,
                       new_value=str(new_fields[name]), is_breaking=is_breaking
                   ))

           # Type changes
           for name in old_fields:
               if name in new_fields:
                   if old_fields[name].get("type") != new_fields[name].get("type"):
                       changes.append(SchemaChange(
                           change_type="type_change", field_name=name,
                           old_value=old_fields[name]["type"],
                           new_value=new_fields[name]["type"],
                           is_breaking=True  # Type changes are always breaking
                       ))
           return changes
   ```

2. **Build the schema registry**:
   ```python
   # schema/registry.py
   import json
   from datetime import datetime

   class SchemaRegistry:
       def __init__(self, conn):
           self.conn = conn
           self.checker = SchemaCompatibilityChecker()

       def register(self, subject: str, schema: dict,
                    mode: CompatibilityMode = CompatibilityMode.BACKWARD) -> dict:
           """Register a new schema version with compatibility check."""
           latest = self.get_latest(subject)

           if latest:
               changes = self.checker.check(latest["schema"], schema, mode)
               breaking = [c for c in changes if c.is_breaking]
               if breaking:
                   raise SchemaCompatibilityError(
                       f"Breaking changes detected for {subject}: "
                       f"{[str(c) for c in breaking]}"
                   )
               version = latest["version"] + 1
           else:
               changes = []
               version = 1

           record = {
               "subject": subject,
               "version": version,
               "schema": schema,
               "changes": [vars(c) for c in changes],
               "registered_at": datetime.utcnow().isoformat(),
               "compatibility_mode": mode.value,
           }

           self.conn.cursor().execute("""
               INSERT INTO schema_registry.schemas
               (subject, version, schema_json, changes_json, registered_at)
               VALUES (%s, %s, %s, %s, %s)
           """, (subject, version, json.dumps(schema),
                 json.dumps(record["changes"]), record["registered_at"]))

           return record

       def get_latest(self, subject: str) -> dict:
           result = self.conn.cursor().execute("""
               SELECT version, schema_json
               FROM schema_registry.schemas
               WHERE subject = %s
               ORDER BY version DESC LIMIT 1
           """, (subject,)).fetchone()

           if result:
               return {"version": result[0], "schema": json.loads(result[1])}
           return None
   ```

   ```python
                               "This change is backward-incompatible. "
                               "Add migration plan or request compatibility override."
                           )
                       })
       return issues

   if __name__ == "__main__":
       changed = subprocess.check_output(
           ["git", "diff", "--name-only", "origin/main"]
       ).decode().strip().split("\n")
       issues = check_schema_changes(changed)
       if issues:
           print("❌ Breaking schema changes detected:")
           for i in issues:
               print(f"  - {i['model']}: {i['breaking_changes']}")
           sys.exit(1)
       print("✅ No breaking schema changes")
   ```

   ```python
   # schema/migrator.py
   class SchemaMigrator:
       def generate_migration(self, subject: str, changes: list[SchemaChange]) -> str:
           """Generate Snowflake DDL for schema migration."""
           table = f"raw_commerce.{subject}"
           statements = []

           for change in changes:
               if change.change_type == "add_column":
                   statements.append(
                       f"ALTER TABLE {table} ADD COLUMN "
                       f"{change.field_name} {change.new_value};"
                   )
               elif change.change_type == "remove_column":
                   statements.append(
                       f"-- WARNING: Dropping column {change.field_name}\n"
                       f"ALTER TABLE {table} DROP COLUMN {change.field_name};"
                   )
               elif change.change_type == "type_change":
                   statements.append(
                       f"-- WARNING: Type change for {change.field_name}: "
                       f"{change.old_value} → {change.new_value}\n"
                       f"ALTER TABLE {table} ALTER COLUMN "
                       f"{change.field_name} SET DATA TYPE {change.new_value};"
                   )
           return "\n".join(statements)
   ```

   ```python
   # docs/freshness_check.py
   import os
   import re
   from datetime import datetime, timedelta

   def check_doc_freshness(docs_dir: str, max_age_days: int = 180) -> list:
       stale = []
       for root, _, files in os.walk(docs_dir):
           for f in files:
               if f.endswith(".md"):
                   path = os.path.join(root, f)
                   with open(path) as fh:
                       content = fh.read()
                   # Look for last_reviewed date in front matter
                   match = re.search(r"last_reviewed:\s*(\d{4}-\d{2}-\d{2})", content)
                   if match:
                       reviewed = datetime.strptime(match.group(1), "%Y-%m-%d")
                       age = (datetime.now() - reviewed).days
                       if age > max_age_days:
                           stale.append({"file": path, "days_stale": age})
                   else:
                       stale.append({"file": path, "days_stale": "no date"})
       return stale
   ```

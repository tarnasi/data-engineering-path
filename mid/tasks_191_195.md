## Task 191: Feature Store — Bridging Data Engineering and Machine Learning

### Business Context
NovaMart's ML team builds models for product recommendations, fraud detection, and demand forecasting. Their biggest pain point: feature engineering. Each model team builds their own features from raw data, resulting in duplicated work, inconsistent calculations, and training-serving skew (features computed differently at training time vs serving time). Priya says: "Build a feature store. One place to define, compute, store, and serve ML features — shared across all models."

### Objective
Design and implement a feature store that serves both batch (training) and real-time (serving) ML features from the data platform.

### Technical Requirements
- **Tools**: Python, Redis (online store), Snowflake (offline store), Airflow, Feast (or custom)
- **Skills**: Feature stores, ML data patterns, online/offline consistency, point-in-time joins

### Step-by-Step Guidance

1. **Understand the dual-store architecture**:
   ```
   ┌─────────────────────────────────────────────────────────┐
   │                    Feature Store                         │
   │                                                          │
   │  ┌───────────────────┐     ┌───────────────────┐       │
   │  │   Offline Store    │     │   Online Store     │       │
   │  │   (Snowflake)      │     │   (Redis)          │       │
   │  │                    │     │                    │       │
   │  │ Historical features│     │ Latest features    │       │
   │  │ for training       │     │ for serving (< 5ms)│       │
   │  │ Point-in-time joins│     │ Key-value lookup   │       │
   │  └───────────────────┘     └───────────────────┘       │
   │           ▲                          ▲                   │
   │           │ Batch (Airflow)          │ Stream (Flink)    │
   │           │ Daily/hourly             │ Real-time         │
   │  ┌───────┴──────────────────────────┴───────────┐      │
   │  │        Feature Computation Engine              │      │
   │  │   (dbt for batch, Flink for streaming)         │      │
   │  └────────────────────────────────────────────────┘      │
   └─────────────────────────────────────────────────────────┘
   
   Training: ML team queries Offline Store with point-in-time correctness
   Serving:  App queries Online Store for real-time predictions (< 5ms)
   ```

3. **Build the feature computation engine**:
   ```python
   # feature_store/engine.py
   import pandas as pd
   import redis
   import json

   class FeatureEngine:
       def __init__(self, snowflake_conn, redis_client):
           self.sf = snowflake_conn
           self.redis = redis_client

       def compute_batch(self, feature: FeatureDefinition):
           """Compute feature and store in both offline + online stores."""
           # Offline store: write to Snowflake feature table
           self.sf.cursor().execute(f"""
               CREATE OR REPLACE TABLE feature_store.{feature.name} AS
               {feature.computation}
           """)

           # Online store: push latest values to Redis
           df = pd.read_sql(
               f"SELECT * FROM feature_store.{feature.name}", self.sf
           )
           pipe = self.redis.pipeline()
           for _, row in df.iterrows():
               key = f"feature:{feature.entity}:{row[feature.entity + '_id']}:{feature.name}"
               pipe.setex(key, 86400, str(row[feature.name]))  # 24h TTL
           pipe.execute()

           return {"rows": len(df), "feature": feature.name}

       def get_training_data(self, entity_ids: list, features: list[str],
                             timestamp_col: str = None) -> pd.DataFrame:
           """Get historical features for training with point-in-time correctness."""
           feature_tables = [f"feature_store.{f}" for f in features]
           
           # Point-in-time join to prevent data leakage
           query = f"""
               SELECT e.*, {', '.join(f'f{i}.{features[i]}' 
                   for i in range(len(features)))}
               FROM (SELECT * FROM entity_table) e
           """
           for i, ft in enumerate(feature_tables):
               query += f"""
                   LEFT JOIN {ft} f{i}
                       ON e.entity_id = f{i}.entity_id
               """
           return pd.read_sql(query, self.sf)

       def get_online_features(self, entity_id: str, entity_type: str,
                               features: list[str]) -> dict:
           """Get latest features for real-time serving (< 5ms)."""
           pipe = self.redis.pipeline()
           for feature in features:
               pipe.get(f"feature:{entity_type}:{entity_id}:{feature}")
           values = pipe.execute()
           return dict(zip(features, values))
   ```

4. **Build the feature catalog and discovery**:
   ```python
   # feature_store/catalog.py
   class FeatureCatalog:
       def __init__(self, features: list[FeatureDefinition]):
           self.features = {f.name: f for f in features}

       def search(self, query: str) -> list[FeatureDefinition]:
           query = query.lower()
           return [f for f in self.features.values()
                   if query in f.name.lower()
                   or query in f.description.lower()
                   or any(query in t for t in f.tags)]

       def get_by_entity(self, entity: str) -> list[FeatureDefinition]:
           return [f for f in self.features.values() if f.entity == entity]

       def generate_docs(self) -> str:
           doc = "# Feature Catalog\n\n"
           by_entity = {}
           for f in self.features.values():
               by_entity.setdefault(f.entity, []).append(f)

           for entity, features in by_entity.items():
               doc += f"## {entity.title()} Features\n\n"
               doc += "| Feature | Type | Freshness | Owner | Description |\n"
               doc += "|---------|------|-----------|-------|-------------|\n"
               for f in features:
                   doc += (f"| `{f.name}` | {f.dtype} | {f.freshness} | "
                          f"{f.owner} | {f.description} |\n")
               doc += "\n"
           return doc
   ```

## Task 192: Disaster Recovery — Multi-Region Failover

### Business Context
NovaMart's data platform runs entirely in us-east-1. When AWS had a 4-hour outage in that region last year, all dashboards went dark and real-time processing stopped. The board mandated: "Critical data systems must survive a full regional outage." Priya says: "Design and test a DR strategy. This doesn't mean running everything twice — it means knowing what's critical, how fast we need to recover, and proving it works."

### Objective
Design a disaster recovery strategy for the data platform with defined RTO/RPO targets, implement the recovery procedures, and test them.

### Technical Requirements
- **Tools**: AWS (S3 cross-region replication, multi-region), Snowflake (replication), Terraform, Python
- **Skills**: Disaster recovery, RTO/RPO planning, failover procedures, DR testing

### Step-by-Step Guidance

1. **Classify systems by criticality and define RTO/RPO**:
   ```markdown
   ## DR Classification

   | System | Criticality | RPO | RTO | DR Strategy |
   |--------|:-----------:|:---:|:---:|-------------|
   | Executive dashboards | Critical | 1 hour | 2 hours | Active-passive (Snowflake replication) |
   | Real-time streaming | Critical | 0 (no loss) | 30 min | Active-active (multi-region Kafka) |
   | Nightly batch pipelines | High | 24 hours | 4 hours | Cold standby (replay from S3) |
   | Data science notebooks | Medium | 24 hours | 8 hours | Backup restore |
   | Dev/staging environments | Low | 48 hours | 24 hours | Rebuild from IaC |

   RPO = Recovery Point Objective (how much data can we lose?)
   RTO = Recovery Time Objective (how fast must we recover?)
   ```

2. **Implement Snowflake database replication**:

   import boto3

   def setup_s3_replication(source_bucket: str, dest_bucket: str,

       """Enable cross-region replication for data lake."""
       s3 = boto3.client("s3")

               {

   ## DR Failover Runbook

   ### Trigger Criteria
   Initiate failover when:
   - AWS us-east-1 is down for > 30 minutes AND
   - No ETA for recovery from AWS status page AND
   - Approved by Ravi (Engineering Manager) or Priya (Staff DE)

   ### Failover Steps (estimated: 45 minutes)

   #### Phase 1: Assess (5 min)
   - [ ] Confirm us-east-1 is genuinely down (not just our services)
   - [ ] Check AWS Health Dashboard

   #### Phase 2: Database Failover (15 min)

   #### Phase 3: Redirect Consumers (10 min)

   - [ ] API health check returns 200
   - [ ] Spot-check 3 key metrics against known values
   - [ ] Notify stakeholders: "DR failover complete"

   ### Failback Steps (when primary recovers)

   - Wait for full sync (may take hours for large datasets)
   - Reverse the failover steps
   - Validate primary data matches DR
   - Switch back to primary connections

### Step-by-Step Guidance

1. **Design Problem 1: Real-Time Fraud Detection Pipeline**:
   ```markdown
   ## Design: Real-Time Fraud Detection for NovaMart

   ### Requirements
   - Process every order (30K orders/hour peak) in < 2 seconds
   - Flag suspicious orders before payment is captured
   - Combine real-time signals (velocity, location) with historical (customer behavior)
   - False positive rate < 5% (don't block legitimate customers)
   - Must handle Black Friday (10x normal traffic)

   ### High-Level Architecture
   ```mermaid
   graph LR
       A[Order Service] -->|event| B[Kafka: orders.raw]
       B --> C[Flink: Enrichment]
       C --> D[Flink: Fraud Rules Engine]
       D -->|score > threshold| E[Kafka: orders.flagged]
       D -->|score < threshold| F[Kafka: orders.approved]
       E --> G[Review Queue / Manual Review]
       F --> H[Payment Service]
       C -->|lookup| I[Redis: Customer Features]
       C -->|lookup| J[Redis: Velocity Counters]
       D -->|async| K[ML Model Service]
   ```

   ### Key Design Decisions

   | Decision | Options | Choice | Rationale |
   |----------|---------|--------|-----------|
   | Processing | Batch / Stream / Lambda | **Stream** | < 2s requirement eliminates batch |
   | ML model | In-stream / sidecar / API | **Async API** | Model updates independent of pipeline |
   | Feature lookup | Snowflake / Redis / embedded | **Redis** | < 5ms lookup required |
   | Rules engine | Hardcoded / configurable | **YAML-configured** | Fraud analysts can update rules |

   ### Detailed Components

   **Velocity Counter (Redis)**:
   ```python

   # Key: velocity:{customer_id}:{window}
   # Example: velocity:C123:1h → 5 (5 orders in last hour)
   
   def check_velocity(customer_id: str) -> dict:
       return {
           "orders_1h": redis.get(f"velocity:{customer_id}:1h") or 0,
           "orders_24h": redis.get(f"velocity:{customer_id}:24h") or 0,
           "unique_ips_1h": redis.scard(f"ips:{customer_id}:1h"),
       }
   ```

   **Rules Engine**:
   ```yaml

       condition: "velocity.orders_1h > 5"
       score: 30
     - name: new_customer_large_order

       score: 25
     - name: mismatched_country
       condition: "customer.country != order.shipping_country"
       score: 20
     - name: known_fraud_device
       condition: "order.device_fingerprint IN blocked_devices"
       score: 100  # Instant block
   ```

   ### Failure Modes & Mitigations
   | Failure | Impact | Mitigation |
   |---------|--------|------------|
   | Flink down | Orders not scored | Default-allow with async scoring |
   | Redis down | No features | Fallback to rule-only scoring |
   | ML model timeout | No model score | Score with rules only, flag for review |
   | Kafka backpressure | Delayed scoring | Auto-scale Flink, alert on lag > 30s |

   ### Scale Estimates
   - 30K orders/hour = ~8 orders/second (normal)
   - Black Friday: 80 orders/second
   - Redis: 8 lookups/order × 80 orders/sec = 640 reads/sec (trivial)
   - Kafka: 80 messages/sec (single partition sufficient)
   - Flink: 1 TaskManager handles 200 events/sec → 1 TM sufficient, 2 for HA
   ```

   # metrics/platform_metrics.py
   from dataclasses import dataclass
   from datetime import datetime
   import pandas as pd

   @dataclass
   class PlatformMetrics:
       date: datetime
       # Reliability
       sla_compliance_pct: float
       mttd_minutes: float
       mttr_minutes: float
       p1_incidents: int
       # Efficiency
       total_pipelines: int
       monthly_cost: float
       cost_per_pipeline: float
       self_serve_teams: int
       avg_pipeline_deploy_days: float
       toil_pct: float
       # Usage
       dashboard_mau: int
       queries_per_day: int
       data_products_count: int
       # Quality
       quality_score_avg: float
       test_coverage_pct: float

   class MetricsCollector:
       def __init__(self, conn):
           self.conn = conn

       def collect_daily(self) -> PlatformMetrics:
           return PlatformMetrics(
               date=datetime.utcnow(),
               sla_compliance_pct=self._calc_sla_compliance(),
               mttd_minutes=self._calc_mttd(),
               mttr_minutes=self._calc_mttr(),
               p1_incidents=self._count_p1_incidents(),
               total_pipelines=self._count_pipelines(),
               monthly_cost=self._get_monthly_cost(),
               cost_per_pipeline=self._get_monthly_cost() / max(self._count_pipelines(), 1),
               self_serve_teams=self._count_self_serve_teams(),
               avg_pipeline_deploy_days=self._calc_deploy_time(),
               toil_pct=self._calc_toil_pct(),
               dashboard_mau=self._get_dashboard_mau(),
               queries_per_day=self._get_daily_queries(),
               data_products_count=self._count_data_products(),
               quality_score_avg=self._calc_quality_score(),
               test_coverage_pct=self._calc_test_coverage(),
           )

       def _calc_sla_compliance(self) -> float:
           result = pd.read_sql("""
               SELECT
                   COUNT(CASE WHEN met_sla THEN 1 END) * 100.0 / COUNT(*) 
                   AS sla_pct
               FROM platform_metrics.sla_checks
               WHERE check_date >= DATEADD(month, -1, CURRENT_DATE())
           """, self.conn)
           return round(result.iloc[0]["sla_pct"], 2)
   ```

   # metrics/okr_tracker.py
   class OKRTracker:
       def __init__(self, metrics_history: pd.DataFrame):
           self.history = metrics_history

       def progress_report(self) -> str:
           latest = self.history.iloc[-1]
           report = "## Platform OKR Progress (This Month)\n\n"

           okrs = [
               ("SLA Compliance", latest["sla_compliance_pct"], 99.5, "%"),
               ("MTTD", latest["mttd_minutes"], 10, "min", True),  # Lower is better
               ("MTTR", latest["mttr_minutes"], 30, "min", True),
               ("Cost/Pipeline", latest["cost_per_pipeline"], 150, "$", True),
               ("Dashboard MAU", latest["dashboard_mau"], 150, "users"),
               ("Toil %", latest["toil_pct"], 15, "%", True),
           ]

           for okr in okrs:
               name, current, target = okr[0], okr[1], okr[2]
               unit = okr[3]
               lower_better = okr[4] if len(okr) > 4 else False

               if lower_better:
                   progress = max(0, (target / max(current, 0.01)) * 100)
                   on_track = current <= target
               else:
                   progress = min(100, (current / max(target, 0.01)) * 100)
                   on_track = current >= target

               icon = "✅" if on_track else "🔄"
               report += (f"- {icon} **{name}**: {current}{unit} "
                         f"(target: {target}{unit}) — {progress:.0f}%\n")

           return report
   ```

2. **Build the federated governance framework**:
   ```python
   # platform/governance/mesh_governance.py
   from dataclasses import dataclass
   from pathlib import Path
   import yaml

   @dataclass
   class DataProductStandard:
       """Minimum requirements for a data product to be published."""
       required_fields = [
           "domain", "owner", "slack", "sla", "quality", "contract_version"
       ]
       min_test_coverage: float = 0.80
       max_freshness_hours: int = 24
       required_docs: list = None

       def __post_init__(self):
           self.required_docs = [
               "description", "schema", "sla", "consumers"
           ]

   class MeshGovernor:
       def __init__(self, standards: DataProductStandard):
           self.standards = standards

       def validate_manifest(self, manifest_path: str) -> dict:
           with open(manifest_path) as f:
               manifest = yaml.safe_load(f)

           issues = []

           # Check required fields
           for field in self.standards.required_fields:
               if field not in manifest:
                   issues.append(f"Missing required field: {field}")

           # Check each data product
           for product in manifest.get("data_products", []):
               name = product.get("name", "unknown")

               # SLA defined?
               if "sla" not in product:
                   issues.append(f"{name}: Missing SLA definition")

               # Quality coverage meets minimum?
               coverage = product.get("quality", {}).get("coverage", 0)
               if isinstance(coverage, str):
                   coverage = float(coverage.strip("%")) / 100
               if coverage < self.standards.min_test_coverage:
                   issues.append(
                       f"{name}: Test coverage {coverage:.0%} "
                       f"< {self.standards.min_test_coverage:.0%} required"
                   )

               # Documentation complete?
               for doc in self.standards.required_docs:
                   if doc not in product:
                       issues.append(f"{name}: Missing documentation: {doc}")

           return {
               "valid": len(issues) == 0,
               "issues": issues,
               "domain": manifest.get("domain", "unknown")
           }

       def generate_scorecard(self, manifest_path: str) -> dict:
           """Generate a data product health scorecard."""
           with open(manifest_path) as f:
               manifest = yaml.safe_load(f)

           products = manifest.get("data_products", [])
           return {
               "domain": manifest.get("domain"),
               "total_products": len(products),
               "avg_coverage": sum(
                   p.get("quality", {}).get("coverage", 0)
                   for p in products
               ) / max(len(products), 1),
               "products_with_sla": sum(
                   1 for p in products if "sla" in p
               ),
               "total_consumers": sum(
                   len(p.get("consumers", []))
                   for p in products
               ),
           }
   ```

4. **Build the data product registry**:

   ```python
   # platform/registry/product_registry.py
   class DataProductRegistry:
       """Central registry of all published data products."""

       def __init__(self, registry_path: str = "data_products/"):
           self.registry_path = Path(registry_path)

       def discover_all(self) -> list[dict]:
           """Scan all domain manifests and build unified registry."""
           products = []
           for manifest_file in self.registry_path.glob("*/manifest.yaml"):
               with open(manifest_file) as f:
                   manifest = yaml.safe_load(f)
               domain = manifest["domain"]
               for product in manifest.get("data_products", []):
                   product["domain"] = domain
                   product["manifest_path"] = str(manifest_file)
                   products.append(product)
           return products

       def find_consumers(self, product_name: str) -> list[str]:
           """Find all consumers of a data product (for impact analysis)."""
           all_products = self.discover_all()
           for p in all_products:
               if p["name"] == product_name:
                   return p.get("consumers", [])
           return []

       def dependency_graph(self) -> dict:
           """Build a graph of domain interdependencies."""
           all_products = self.discover_all()
           graph = {}
           for p in all_products:
               domain = p["domain"]
               consumers = p.get("consumers", [])
               if domain not in graph:
                   graph[domain] = {"produces": [], "consumes_from": []}
               graph[domain]["produces"].append(p["name"])
           return graph
   ```

2. **Build the erasure pipeline**:
   ```python
   # compliance/erasure_pipeline.py
   import hashlib
   import logging
   from dataclasses import dataclass
   from datetime import datetime
   from typing import Optional

   logger = logging.getLogger(__name__)

   @dataclass
   class ErasureRequest:
       request_id: str
       customer_id: str
       customer_email: str
       requested_at: datetime
       deadline: datetime        # Must complete within 30 days (GDPR)
       status: str = "pending"   # pending → processing → verifying → complete

   @dataclass
   class ErasureResult:
       request_id: str
       system: str
       table: str
       rows_affected: int
       action: str              # "anonymized", "deleted", "tombstoned"
       completed_at: datetime
       verified: bool = False

   class ErasurePipeline:
       def __init__(self, pii_registry_path: str, connections: dict):
           self.registry = self._load_registry(pii_registry_path)
           self.connections = connections
           self.results: list[ErasureResult] = []

       def process(self, request: ErasureRequest) -> list[ErasureResult]:
           logger.info(f"Processing erasure request {request.request_id} "
                       f"for customer {request.customer_id}")

           # Phase 1: Snowflake anonymization
           for location in self.registry["snowflake"]:
               result = self._anonymize_snowflake(
                   request, location
               )
               self.results.append(result)

           # Phase 2: PostgreSQL deletion
           for location in self.registry["postgresql"]:
               result = self._delete_postgres(request, location)
               self.results.append(result)

           # Phase 3: Kafka tombstones
           for location in self.registry["kafka"]:
               result = self._tombstone_kafka(request, location)
               self.results.append(result)

           # Phase 4: External system cleanup
           for system in self.registry["external_systems"]:
               result = self._cleanup_external(request, system)
               self.results.append(result)

           return self.results

       def _anonymize_snowflake(
           self, request: ErasureRequest, location: dict
       ) -> ErasureResult:
           conn = self.connections["snowflake"]
           table = f"{location['schema']}.{location['table']}"
           pii_cols = location["pii_columns"]
           method = location["anonymization"]

           if method == "hash":
               set_clauses = ", ".join(
                   f"{col} = SHA2({col}, 256)" for col in pii_cols
               )
           elif method == "anonymize":
               set_clauses = ", ".join(
                   f"{col} = 'REDACTED_' || {request.request_id}"
                   for col in pii_cols
               )
           else:
               set_clauses = ", ".join(
                   f"{col} = NULL" for col in pii_cols
               )

           sql = f"""
               UPDATE {table}
               SET {set_clauses}, _anonymized_at = CURRENT_TIMESTAMP()
               WHERE customer_id = %(customer_id)s
                 AND _anonymized_at IS NULL
           """
           cursor = conn.cursor()
           cursor.execute(sql, {"customer_id": request.customer_id})
           rows = cursor.rowcount
           conn.commit()

           return ErasureResult(
               request_id=request.request_id,
               system="snowflake", table=table,
               rows_affected=rows, action=method,
               completed_at=datetime.utcnow()
           )

       def verify(self, request: ErasureRequest) -> dict:
           """Verify no PII remains for this customer."""
           verification = {"passed": True, "checks": []}
           conn = self.connections["snowflake"]

           for location in self.registry["snowflake"]:
               table = f"{location['schema']}.{location['table']}"
               for col in location["pii_columns"]:
                   cursor = conn.cursor()
                   cursor.execute(f"""
                       SELECT COUNT(*) FROM {table}
                       WHERE customer_id = %(cid)s
                         AND {col} IS NOT NULL
                         AND {col} NOT LIKE 'REDACTED_%%'
                         AND LENGTH({col}) != 64  -- not SHA256
                   """, {"cid": request.customer_id})
                   count = cursor.fetchone()[0]
                   check = {
                       "table": table, "column": col,
                       "remaining_pii_rows": count,
                       "passed": count == 0
                   }
                   verification["checks"].append(check)
                   if count > 0:
                       verification["passed"] = False

           return verification
   ```

3. **Build the Airflow DAG for automated processing**:
   ```python
   # dags/gdpr_erasure.py
   from airflow.decorators import dag, task
   from datetime import datetime, timedelta

   @dag(
       schedule="@hourly",
       start_date=datetime(2025, 1, 1),
       catchup=False,
       tags=["compliance", "gdpr"],
       default_args={"retries": 2, "retry_delay": timedelta(minutes=5)}
   )
   def gdpr_erasure_pipeline():

       @task
       def fetch_pending_requests() -> list[dict]:
           """Fetch unprocessed erasure requests from the compliance queue."""
           conn = get_snowflake_conn()
           requests = pd.read_sql("""
               SELECT request_id, customer_id, customer_email,
                      requested_at, deadline
               FROM compliance.erasure_requests
               WHERE status = 'pending'
               ORDER BY requested_at
               LIMIT 20
           """, conn)
           return requests.to_dict(orient="records")

       @task
       def process_erasure(request_data: dict):
           pipeline = ErasurePipeline(
               "compliance/pii_registry.yaml", get_connections()
           )
           request = ErasureRequest(**request_data)
           results = pipeline.process(request)

           # Log to audit table
           log_audit_trail(request, results)

           # Verify
           verification = pipeline.verify(request)
           if not verification["passed"]:
               raise ValueError(
                   f"Verification failed for {request.request_id}: "
                   f"{verification}"
               )

           # Mark complete
           update_request_status(request.request_id, "complete")
           return {"request_id": request.request_id, "verified": True}

       requests = fetch_pending_requests()
       process_erasure.expand(request_data=requests)

   gdpr_erasure_pipeline()
   ```

   ```python
   """
   POC: Migrate the Commerce Orders pipeline from Airflow to Dagster.
   Original: dags/commerce_orders.py (Airflow)
   """
   from dagster import (
       asset, AssetExecutionContext, MaterializeResult,
       AutoMaterializePolicy, FreshnessPolicy,
       define_asset_job, ScheduleDefinition
   )
   import pandas as pd

   @asset(
       description="Raw orders from PostgreSQL CDC",
       group_name="commerce",
       auto_materialize_policy=AutoMaterializePolicy.eager(),
       freshness_policy=FreshnessPolicy(maximum_lag_minutes=120),
   )
   def raw_orders(context: AssetExecutionContext) -> MaterializeResult:
       """Ingest raw orders from source database."""
       conn = context.resources.source_db
       df = pd.read_sql(
           "SELECT * FROM orders WHERE updated_at > %(watermark)s",
           conn, params={"watermark": get_watermark("raw_orders")}
       )
       write_to_snowflake(df, "raw_commerce.orders")
       return MaterializeResult(
           metadata={
               "row_count": len(df),
               "max_updated_at": str(df["updated_at"].max())
           }
       )

   @asset(
       description="Cleaned and validated orders",
       group_name="commerce",
       deps=[raw_orders],
   )
   def staging_orders(context: AssetExecutionContext) -> MaterializeResult:
       """dbt run staging_orders equivalent."""
       result = context.resources.dbt.run(
           select="staging_commerce.stg_orders"
       )
       return MaterializeResult(
           metadata={"dbt_run_result": result.to_dict()}
       )

   @asset(
       description="Gold-layer fact table for orders",
       group_name="commerce",
       deps=[staging_orders],
       freshness_policy=FreshnessPolicy(maximum_lag_minutes=120),
   )
   def fact_orders(context: AssetExecutionContext) -> MaterializeResult:
       """dbt run fact_orders equivalent."""
       result = context.resources.dbt.run(
           select="gold_commerce.fact_orders"
       )
       return MaterializeResult(
           metadata={"dbt_run_result": result.to_dict()}
       )

   # Define the job + schedule
   commerce_orders_job = define_asset_job(
       name="commerce_orders_job",
       selection=[raw_orders, staging_orders, fact_orders],
   )

   commerce_orders_schedule = ScheduleDefinition(
       job=commerce_orders_job,
       cron_schedule="0 */2 * * *",
   )
   ```

1. **Build the cost visibility layer**:
   ```sql
   -- Cost attribution view: who's spending what
   CREATE OR REPLACE VIEW analytics.warehouse_cost_by_team AS
   WITH warehouse_usage AS (
       SELECT
           warehouse_name,
           start_time::date AS usage_date,
           SUM(credits_used) AS credits
       FROM snowflake.account_usage.warehouse_metering_history
       WHERE start_time >= DATEADD(month, -3, CURRENT_DATE())
       GROUP BY 1, 2
   ),
   warehouse_team_map AS (
       -- Map warehouses to teams via naming convention
       SELECT warehouse_name,
           CASE
               WHEN warehouse_name LIKE '%COMMERCE%' THEN 'commerce'
               WHEN warehouse_name LIKE '%MARKETING%' THEN 'marketing'
               WHEN warehouse_name LIKE '%FINANCE%' THEN 'finance'
               WHEN warehouse_name LIKE '%ML%' THEN 'ml'
               WHEN warehouse_name LIKE '%PLATFORM%' THEN 'platform'
               ELSE 'unattributed'
           END AS team
       FROM (SELECT DISTINCT warehouse_name FROM warehouse_usage)
   )
   SELECT
       u.usage_date,
       m.team,
       u.warehouse_name,
       u.credits,
       u.credits * 3.50 AS estimated_cost_usd  -- $3.50/credit (Standard)
   FROM warehouse_usage u
   JOIN warehouse_team_map m ON u.warehouse_name = m.warehouse_name;
   ```

2. **Build automated cost anomaly detection**:
   ```python
   # finops/cost_monitor.py
   import pandas as pd
   from datetime import datetime, timedelta

   class CostMonitor:
       def __init__(self, conn, alert_threshold_pct: float = 20.0):
           self.conn = conn
           self.threshold = alert_threshold_pct

       def daily_check(self) -> list[dict]:
           """Compare today's costs to 7-day moving average."""
           query = """
           WITH daily_costs AS (
               SELECT
                   team,
                   usage_date,
                   SUM(estimated_cost_usd) AS daily_cost
               FROM analytics.warehouse_cost_by_team
               WHERE usage_date >= DATEADD(day, -14, CURRENT_DATE())
               GROUP BY 1, 2
           ),
           with_baseline AS (
               SELECT
                   team,
                   usage_date,
                   daily_cost,
                   AVG(daily_cost) OVER (
                       PARTITION BY team
                       ORDER BY usage_date
                       ROWS BETWEEN 7 PRECEDING AND 1 PRECEDING
                   ) AS baseline_7d
               FROM daily_costs
           )
           SELECT
               team,
               usage_date,
               daily_cost,
               baseline_7d,
               ROUND((daily_cost - baseline_7d) / NULLIF(baseline_7d, 0) * 100, 1) AS pct_change
           FROM with_baseline
           WHERE usage_date = CURRENT_DATE() - 1
             AND ABS(pct_change) > %(threshold)s
           ORDER BY ABS(pct_change) DESC
           """
           df = pd.read_sql(query, self.conn, params={"threshold": self.threshold})
           return df.to_dict(orient="records")

       def find_expensive_queries(self, min_credits: float = 1.0) -> pd.DataFrame:
           """Find queries that consumed the most credits."""
           return pd.read_sql("""
               SELECT
                   query_id,
                   user_name,
                   warehouse_name,
                   execution_status,
                   total_elapsed_time / 1000 AS duration_sec,
                   credits_used_cloud_services,
                   query_text
               FROM snowflake.account_usage.query_history
               WHERE start_time >= DATEADD(day, -1, CURRENT_DATE())
                 AND credits_used_cloud_services > %(min)s
               ORDER BY credits_used_cloud_services DESC
               LIMIT 20
           """, self.conn, params={"min": min_credits})
   ```

3. **Implement automated optimization rules**:
   ```python
   # finops/optimizer.py
   class AutoOptimizer:
       def __init__(self, conn):
           self.conn = conn

       def find_idle_warehouses(self, idle_minutes: int = 30) -> list[dict]:
           """Find warehouses that auto-suspend is too high."""
           return pd.read_sql("""
               SELECT warehouse_name, auto_suspend / 60 AS auto_suspend_min
               FROM snowflake.account_usage.warehouses
               WHERE auto_suspend > %(max_idle)s
                 AND is_current = 'YES'
           """, self.conn, params={"max_idle": idle_minutes * 60}).to_dict("records")

       def find_unused_tables(self, days_unused: int = 90) -> list[dict]:
           """Tables not queried in N days — candidates for archival."""
           return pd.read_sql("""
               SELECT
                   table_schema,
                   table_name,
                   row_count,
                   bytes / (1024*1024*1024) AS size_gb,
                   last_altered
               FROM snowflake.account_usage.table_storage_metrics
               WHERE table_name NOT IN (
                   SELECT DISTINCT
                       split_part(objects_modified, '.', 2)
                   FROM snowflake.account_usage.access_history
                   WHERE query_start_time >= DATEADD(day, -%(days)s, CURRENT_DATE())
               )
               AND bytes > 100000000  -- > 100MB
               ORDER BY bytes DESC
           """, self.conn, params={"days": days_unused}).to_dict("records")

       def generate_recommendations(self) -> list[dict]:
           recs = []
           idle = self.find_idle_warehouses()
           for wh in idle:
               recs.append({
                   "type": "reduce_auto_suspend",
                   "target": wh["warehouse_name"],
                   "current": f"{wh['auto_suspend_min']}min",
                   "recommended": "5min",
                   "estimated_savings": "$50-200/month"
               })

           unused = self.find_unused_tables()
           for t in unused:
               recs.append({
                   "type": "archive_unused_table",
                   "target": f"{t['table_schema']}.{t['table_name']}",
                   "size_gb": round(t["size_gb"], 2),
                   "last_used": str(t["last_altered"]),
                   "estimated_savings": f"${round(t['size_gb'] * 23, 2)}/month"
               })

           return recs
   ```

4. **Monthly chargeback report**:
   ```sql
   -- Monthly cost report per team
   SELECT
       team,
       DATE_TRUNC('month', usage_date) AS month,
       SUM(estimated_cost_usd) AS total_cost,
       SUM(credits) AS total_credits,
       COUNT(DISTINCT usage_date) AS active_days,
       ROUND(SUM(estimated_cost_usd) / COUNT(DISTINCT usage_date), 2) AS avg_daily_cost
   FROM analytics.warehouse_cost_by_team
   GROUP BY 1, 2
   ORDER BY month DESC, total_cost DESC;
   ```

### Technical Requirements

- **Skills**: System design, estimation, architecture, trade-off analysis, communication

### Step-by-Step Guidance

1. **The prompt** (Priya presents this):
   ```
   "Design the data infrastructure for a ride-sharing company
   similar to Uber/Lyft.

   Requirements:
   - 10M rides/day across 50 cities
   - Real-time driver matching needs live supply/demand data
   - Riders and drivers see surge pricing updated every 30 seconds
   - Analytics team needs daily reports: revenue, rides, wait times
   - ML team needs features for ETA prediction and fraud detection
   - Finance needs exact revenue reconciliation (every cent must match)

   Design the data infrastructure. You have 45 minutes."
   ```

2. **Step 1: Clarify requirements** (5 minutes):
   ```markdown
   ## Clarification Questions I'd Ask

   **Data Volume**:
   - 10M rides/day ÷ 86400 sec = ~116 rides/sec at steady state
   - Peak: probably 3-5x → ~500 rides/sec
   - Each ride generates: request, match, start, location updates, end, payment
   - Location updates every 5 sec for ~15 min avg ride → ~180 events/ride
   - Total events: 10M × 180 = 1.8B events/day ≈ 21K events/sec

   **Latency Requirements**:
   - Surge pricing: 30-second freshness → need streaming aggregation
   - Analytics: daily batch is fine → traditional ETL
   - ML features: real-time (for matching) + batch (for training)
   - Finance: exact — batch with reconciliation checks

   **Retention**:
   - Raw events: 90 days hot, 7 years cold (regulatory)
   - Aggregated: 3 years hot
   ```

3. **Step 2: Back-of-envelope estimation** (5 minutes):
   ```markdown
   ## Sizing Estimates

   **Storage**:
   - Event size: ~500 bytes average (JSON with location, metadata)
   - Daily raw: 1.8B × 500B = 900 GB/day
   - Monthly raw: ~27 TB
   - Columnar (Parquet): ~5:1 compression → ~5.4 TB/month

   **Compute**:
   - Streaming: 21K events/sec → 4-8 Flink task managers
   - Batch: 900 GB/day → medium Spark cluster (20 nodes, 2-hour job)
   - Warehouse: Snowflake XL for analytics, M for ad-hoc

   **Cost Estimate** (monthly):
   - Storage: ~$1,500 (S3) + $5,000 (Snowflake)
   - Streaming compute: ~$3,000 (Flink on EKS)
   - Batch compute: ~$2,000 (EMR, spot instances)
   - Snowflake compute: ~$8,000
   - Total: ~$20K/month
   ```

4. **Step 3: High-level architecture** (15 minutes):

5. **Step 4: Detailed component design** (15 minutes):
   ```markdown
   ## Key Design Decisions

   ### 1. Kafka Topic Design
   - `rides.events` — Partitioned by city_id (50 partitions)
   - `rides.locations` — Partitioned by driver_id (high cardinality, 256 partitions)
   - `rides.payments` — Partitioned by ride_id (needs order guarantee per ride)

   Why city_id for events? Surge pricing is per-city, co-locating
   events by city means Flink can compute surge without shuffling.

   ### 2. Surge Pricing Pipeline (Real-Time)
   - Flink tumbling window: 30 seconds
   - Input: ride requests + driver availability per city
   - Computation: demand/supply ratio → surge multiplier
   - Output: Redis key `surge:{city_id}` with TTL 60s
   - Fallback: if Flink is down, Redis TTL expires → no surge (fail safe)

   ### 3. Batch Pipeline (Lambda Architecture)
   - Bronze: raw JSON from Kafka → S3, partitioned by date/hour/city
   - Silver: Spark dedup, schema enforcement, type casting
   - Gold: Snowflake fact_rides, fact_payments, dim_drivers, dim_cities
   - Finance reconciliation: FULL OUTER JOIN payments vs ride completions

   ### 4. Trade-offs
   | Decision | Alternative | Why This Choice |
   |----------|------------|-----------------|
   | Kafka (not Kinesis) | Kinesis simpler | Need replay, multi-consumer, 7-day retention |
   | Flink (not Spark Streaming) | Spark simpler | True event-time processing, lower latency |
   | Redis (not Snowflake) | Snowflake simpler | Sub-millisecond reads for surge pricing |
   | Lambda (not Kappa) | Kappa is cleaner | Finance needs exact batch reconciliation |
   ```

6. **Step 5: Failure modes and monitoring** (5 minutes):
   ```markdown
   ## What Can Go Wrong

   | Failure | Impact | Mitigation |
   |---------|--------|------------|
   | Flink crashes | No surge updates for 30s+ | Redis TTL → graceful degrade |
   | Kafka broker down | Event loss | 3-broker replication, acks=all |
   | Spark job fails | Stale analytics | Airflow retry + SLA alerts |
   | Redis OOM | No real-time features | Eviction policy + capacity alerts |
   | Late-arriving events | Incorrect daily totals | Reprocessing window (T+2 batch) |

   ## Key SLIs/SLOs
   - Surge freshness: < 30s (SLO: 99.9%)
   - Event processing latency: < 5s (SLO: 99%)
   - Daily analytics freshness: < 6 hours (SLO: 99.5%)
   - Finance reconciliation accuracy: 100% (SLO: 100%)
   ```

### Common Mistakes
- Jumping to architecture without clarifying requirements — you might design for the wrong scale
- Only batch OR only streaming — ride-sharing needs both (Lambda architecture)
- Ignoring failure modes — "what happens when X goes down?" is the most important question
- Not estimating — "we need Kafka" is useless without "21K events/sec, 50 partitions"
- Over-designing — don't design microservices for a problem that needs 3 components

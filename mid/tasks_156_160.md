2. **Build the auto-detecting anomaly engine**:
   ```python
   # anomaly_engine/detector.py
   import numpy as np
   import pandas as pd
   from scipy import stats
   from dataclasses import dataclass
   from typing import Optional

   @dataclass
   class AnomalyResult:
       metric_name: str
       pipeline: str
       timestamp: str
       observed: float
       expected: float
       score: float         # 0.0 (normal) to 1.0 (highly anomalous)
       method: str
       is_anomaly: bool
       direction: str       # "high", "low", or "both"

   class AutoDetector:
       """Zero-config anomaly detection using ensemble methods."""

       def __init__(self, min_history_days: int = 14):
           self.min_history = min_history_days

       def detect(
           self, history: pd.Series, current: float,
           metric_name: str = "", pipeline: str = ""
       ) -> AnomalyResult:
           if len(history) < self.min_history:
               return AnomalyResult(
                   metric_name=metric_name, pipeline=pipeline,
                   timestamp=str(pd.Timestamp.now()),
                   observed=current, expected=history.mean(),
                   score=0.0, method="insufficient_data",
                   is_anomaly=False, direction="both"
               )

           scores = []

           # Method 1: Z-score (handles normal distributions)
           z = abs((current - history.mean()) / max(history.std(), 1e-10))
           scores.append(min(z / 4.0, 1.0))

           # Method 2: IQR-based (handles skewed distributions)
           q1, q3 = history.quantile(0.25), history.quantile(0.75)
           iqr = q3 - q1
           lower, upper = q1 - 2.0 * iqr, q3 + 2.0 * iqr
           if iqr > 0:
               if current < lower:
                   scores.append(min(abs(current - lower) / iqr, 1.0))
               elif current > upper:
                   scores.append(min(abs(current - upper) / iqr, 1.0))
               else:
                   scores.append(0.0)

           # Method 3: Recent trend (detect sudden shifts)
           recent = history.tail(7)
           older = history.iloc[:-7] if len(history) > 7 else history
           if older.std() > 0:
               trend_z = abs(
                   (current - recent.mean()) / max(older.std(), 1e-10)
               )
               scores.append(min(trend_z / 3.0, 1.0))

           # Method 4: Day-of-week seasonality
           if len(history) >= 28:
               dow = pd.Timestamp.now().dayofweek
               same_dow = history[history.index.dayofweek == dow]
               if len(same_dow) >= 3 and same_dow.std() > 0:
                   seasonal_z = abs(
                       (current - same_dow.mean()) / same_dow.std()
                   )
                   scores.append(min(seasonal_z / 3.5, 1.0))

           # Ensemble: weighted average
           final_score = np.mean(scores)
           direction = "high" if current > history.mean() else "low"

           return AnomalyResult(
               metric_name=metric_name, pipeline=pipeline,
               timestamp=str(pd.Timestamp.now()),
               observed=current, expected=history.mean(),
               score=round(final_score, 3),
               method="ensemble",
               is_anomaly=final_score > 0.7,
               direction=direction
           )
   ```

4. **Build the alert router** — prevent alert fatigue:
   ```python
   # anomaly_engine/alert_router.py
   class AlertRouter:
       def __init__(self, redis_client):
           self.redis = redis_client

       def should_alert(self, result: AnomalyResult) -> bool:
           # Rule 1: Only alert on score > 0.7
           if result.score < 0.7:
               return False

           # Rule 2: Dedup — don't re-alert same pipeline+metric in 4h
           dedup_key = f"alert:{result.pipeline}:{result.metric_name}"
           if self.redis.exists(dedup_key):
               return False
           self.redis.setex(dedup_key, 14400, "1")  # 4 hour TTL

           # Rule 3: Suppress if pipeline has known maintenance window
           if self.is_in_maintenance(result.pipeline):
               return False

           return True

       def route(self, result: AnomalyResult) -> str:
           """Route to appropriate channel based on severity."""
           if result.score > 0.9:
               return "pagerduty"      # Wake someone up
           elif result.score > 0.8:
               return "slack_urgent"    # #data-alerts
           else:
               return "slack_info"      # #data-monitoring (FYI)
   ```

5. **Airflow DAG to run across all pipelines**:
   ```python
   @dag(schedule="0 */4 * * *", start_date=datetime(2025, 1, 1))
   def anomaly_scan():
       @task
       def scan_all():
           detector = AutoDetector(min_history_days=14)
           metrics = collect_all_pipeline_metrics(get_snowflake_conn())
           router = AlertRouter(get_redis())
           anomalies = []

           for metric_name, history in metrics.items():
               if len(history) < 14:
                   continue
               current = history.iloc[-1]
               baseline = history.iloc[:-1]
               result = detector.detect(
                   baseline, current, metric_name=metric_name
               )
               if result.is_anomaly and router.should_alert(result):
                   anomalies.append(result)
                   router.route(result)

           log_results(anomalies)
           return {"total_metrics": len(metrics), "anomalies": len(anomalies)}

       scan_all()
   anomaly_scan()
   ```

### Common Mistakes
- Setting thresholds too low — generates 100+ alerts/day, everyone ignores them
- No seasonality handling — every Monday looks anomalous compared to Sunday
- Alerting on metrics with insufficient history — noisy false positives
- No deduplication — same anomaly fires every 4 hours for days

   ```markdown
   ### 1. ❌ Direct Writes to Gold Layer

   `gold.customer_recommendations`.

   **Concern**: This bypasses all data quality checks, SCD tracking, and lineage.
   If the reco model writes bad data, it pollutes dashboards downstream.

   a 4XL warehouse (~$60/hour) and still get p99 > 2 seconds.

   **Suggested alternative**: Pre-compute recommendations into Redis.
   Cache TTL = 1 hour. Cache hit rate should be > 95%.

   **Data to support this**: [Include actual Snowflake query profile showing
   cold vs warm latency at different warehouse sizes]
   ```

## Task 160: Cross-Functional Data Strategy — Aligning Platform with Product

### Business Context
NovaMart's CPO (Chief Product Officer) is frustrated: "I keep hearing 'we don't have the data for that' from every product team. We have a million data pipelines but product teams still can't answer basic questions like: what's the conversion rate for users who saw a recommendation vs those who didn't?" The problem isn't missing data — it's that the data platform wasn't designed with product analytics in mind. Ravi asks you to align the data platform roadmap with Product's needs.

### Objective
Create a cross-functional data strategy that bridges the gap between the data platform team and product teams, ensuring the platform enables product analytics and experimentation.

### Technical Requirements

- **Skills**: Cross-functional communication, product analytics, experimentation platforms, stakeholder management

3. **Build the experiment results pipeline**:
   ```sql
   -- gold.experiment_results — Auto-computed daily
   WITH assignments AS (
       SELECT
           experiment_id,
           variant,
           user_id,
           MIN(assigned_at) AS first_assignment
       FROM staging.experiment_assignments
       GROUP BY 1, 2, 3
   ),
   conversions AS (
       SELECT
           a.experiment_id,
           a.variant,
           COUNT(DISTINCT a.user_id) AS users,
           COUNT(DISTINCT CASE WHEN o.order_id IS NOT NULL
                          THEN a.user_id END) AS converters,
           SUM(COALESCE(o.total_amount, 0)) AS total_revenue
       FROM assignments a
       LEFT JOIN gold.fact_orders o
           ON a.user_id = o.user_id
           AND o.order_date >= a.first_assignment::date
           AND o.order_date < a.first_assignment::date + INTERVAL '14 days'
       GROUP BY 1, 2
   )
   SELECT
       experiment_id,
       variant,
       users,
       converters,
       ROUND(converters::FLOAT / NULLIF(users, 0), 4) AS conversion_rate,
       total_revenue,
       ROUND(total_revenue / NULLIF(users, 0), 2) AS revenue_per_user
   FROM conversions;
   ```

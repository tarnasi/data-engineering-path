## Task 166: SRE for Data — Error Budgets and Toil Reduction

### Objective

2. **Build the error budget tracker**:
   ```python
   # sre/error_budget.py
   from dataclasses import dataclass
   from datetime import datetime, timedelta
   import pandas as pd

   @dataclass
   class ErrorBudgetStatus:
       slo_name: str
       target: float
       actual: float
       budget_total_minutes: float
       budget_consumed_minutes: float
       budget_remaining_pct: float
       status: str  # "healthy", "warning", "exhausted"

   class ErrorBudgetTracker:
       def __init__(self, conn, slo_config: dict):
           self.conn = conn
           self.config = slo_config

       def calculate(self, slo_name: str, window_days: int = 30) -> ErrorBudgetStatus:
           slo = next(s for s in self.config["slos"] if s["name"] == slo_name)
           target = slo["target"] / 100.0
           total_minutes = window_days * 24 * 60
           budget_minutes = total_minutes * (1 - target)

           # Query actual SLI from monitoring data
           actual_good = self._query_good_events(slo_name, window_days)
           actual_total = self._query_total_events(slo_name, window_days)
           actual_rate = actual_good / max(actual_total, 1)

           consumed = total_minutes * (1 - actual_rate)
           remaining_pct = max(0, (budget_minutes - consumed) / budget_minutes * 100)

           if remaining_pct > 50:
               status = "healthy"
           elif remaining_pct > 10:
               status = "warning"
           else:
               status = "exhausted"

           return ErrorBudgetStatus(
               slo_name=slo_name, target=target,
               actual=round(actual_rate, 4),
               budget_total_minutes=round(budget_minutes, 1),
               budget_consumed_minutes=round(consumed, 1),
               budget_remaining_pct=round(remaining_pct, 1),
               status=status
           )

       def _query_good_events(self, slo_name: str, days: int) -> int:
           return pd.read_sql(f"""
               SELECT COUNT(*) as cnt FROM sre.sli_measurements
               WHERE slo_name = '{slo_name}'
                 AND measured_at >= DATEADD(day, -{days}, CURRENT_TIMESTAMP())
                 AND is_good = TRUE
           """, self.conn).iloc[0]["cnt"]

       def _query_total_events(self, slo_name: str, days: int) -> int:
           return pd.read_sql(f"""
               SELECT COUNT(*) as cnt FROM sre.sli_measurements
               WHERE slo_name = '{slo_name}'
                 AND measured_at >= DATEADD(day, -{days}, CURRENT_TIMESTAMP())
           """, self.conn).iloc[0]["cnt"]
   ```

## Task 167: Zero-Trust Data Access — Security Hardening

### Business Context
NovaMart's security team ran a penetration test on the data platform. Findings: 3 service accounts with admin access that haven't been rotated in 6 months; 12 users with access to PII they don't need; and the Airflow metadata DB password is in plaintext in a config file. The CISO says: "Adopt zero-trust for data access. Every query should be authenticated, authorized, and logged. No standing privileges."

### Objective
Implement zero-trust data access: dynamic credentials, just-in-time access, column-level security, and comprehensive audit logging.

### Technical Requirements
- **Tools**: Snowflake (RBAC, masking policies), HashiCorp Vault (or AWS Secrets Manager), Python, SQL
- **Skills**: Zero-trust architecture, dynamic credentials, JIT access, column-level security, audit logging

### Step-by-Step Guidance

1. **Audit current access state**:
   ```sql
   -- Find over-privileged users
   SELECT
       grantee_name,
       role,
       privilege,
       table_schema,
       table_name,
       granted_on
   FROM snowflake.account_usage.grants_to_users
   WHERE deleted_on IS NULL
     AND privilege IN ('OWNERSHIP', 'ALL')
     AND table_schema IN ('RAW_COMMERCE', 'GOLD_COMMERCE')
   ORDER BY grantee_name;

   -- Find unused access (granted but never queried)
   WITH granted_access AS (
       SELECT DISTINCT grantee_name, table_name
       FROM snowflake.account_usage.grants_to_users
       WHERE deleted_on IS NULL
   ),
   actual_usage AS (
       SELECT DISTINCT user_name, 
           split_part(objects_modified[0]:'objectName', '.', 3) AS table_name
       FROM snowflake.account_usage.access_history
       WHERE query_start_time >= DATEADD(month, -3, CURRENT_DATE())
   )
   SELECT g.grantee_name, g.table_name, 'UNUSED - revoke candidate' AS recommendation
   FROM granted_access g
   LEFT JOIN actual_usage u 
       ON g.grantee_name = u.user_name AND g.table_name = u.table_name
   WHERE u.user_name IS NULL;
   ```

2. **Implement dynamic credentials with a secrets manager**:
   ```python
   # security/credential_manager.py
   import boto3
   import json
   from datetime import datetime, timedelta

   class DynamicCredentialManager:
       """Issue short-lived Snowflake credentials via AWS Secrets Manager."""

       def __init__(self, region: str = "us-east-1"):
           self.client = boto3.client("secretsmanager", region_name=region)

       def get_connection(self, service_name: str, ttl_hours: int = 8) -> dict:
           """Get scoped credentials for a specific service."""
           secret = self.client.get_secret_value(
               SecretId=f"novamart/snowflake/{service_name}"
           )
           creds = json.loads(secret["SecretString"])
           return {
               "account": creds["account"],
               "user": creds["user"],
               "password": creds["password"],
               "role": creds["role"],          # Scoped role
               "warehouse": creds["warehouse"],  # Scoped warehouse
               "database": creds["database"],
           }

       def rotate_secret(self, service_name: str):
           """Trigger automatic rotation."""
           self.client.rotate_secret(
               SecretId=f"novamart/snowflake/{service_name}",
               RotationRules={"AutomaticallyAfterDays": 30}
           )
   ```

3. **Implement column-level masking policies**:
   ```sql
   -- Dynamic masking: PII visible only to authorized roles
   CREATE OR REPLACE MASKING POLICY pii_email_mask AS
   (val STRING) RETURNS STRING ->
       CASE
           WHEN CURRENT_ROLE() IN ('PII_ADMIN', 'COMPLIANCE_TEAM')
               THEN val
           WHEN CURRENT_ROLE() IN ('ANALYTICS_TEAM')
               THEN SHA2(val, 256)  -- Hashed for joins, not readable
           ELSE '***MASKED***'
       END;

   ALTER TABLE gold_commerce.dim_customers
       MODIFY COLUMN email SET MASKING POLICY pii_email_mask;
   ALTER TABLE gold_commerce.dim_customers
       MODIFY COLUMN phone SET MASKING POLICY pii_email_mask;

   -- Row-level security: teams only see their region
   CREATE OR REPLACE ROW ACCESS POLICY region_filter AS
   (region_col STRING) RETURNS BOOLEAN ->
       CURRENT_ROLE() = 'GLOBAL_ADMIN'
       OR region_col IN (
           SELECT allowed_region
           FROM security.role_region_mapping
           WHERE role_name = CURRENT_ROLE()
       );

   ALTER TABLE gold_commerce.fact_orders
       ADD ROW ACCESS POLICY region_filter ON (region);
   ```

4. **Build just-in-time access request system**:
   ```python
   # security/jit_access.py
   from dataclasses import dataclass
   from datetime import datetime, timedelta

   @dataclass
   class AccessRequest:
       requester: str
       role_requested: str
       reason: str
       duration_hours: int
       approved_by: str = None
       expires_at: datetime = None

   class JITAccessManager:
       def __init__(self, conn):
           self.conn = conn

       def grant_temporary_access(self, request: AccessRequest):
           """Grant role with automatic expiration."""
           if not request.approved_by:
               raise ValueError("Access must be approved before granting")

           request.expires_at = datetime.utcnow() + timedelta(
               hours=request.duration_hours
           )

           # Grant the role
           self.conn.cursor().execute(
               f"GRANT ROLE {request.role_requested} "
               f"TO USER {request.requester}"
           )

           # Schedule revocation
           self._schedule_revoke(request)

           # Audit log
           self._log_access_grant(request)

       def revoke_expired(self):
           """Revoke all expired temporary grants — run hourly."""
           expired = self.conn.cursor().execute("""
               SELECT requester, role_granted
               FROM security.jit_access_log
               WHERE expires_at < CURRENT_TIMESTAMP()
                 AND revoked_at IS NULL
           """).fetchall()

           for user, role in expired:
               self.conn.cursor().execute(
                   f"REVOKE ROLE {role} FROM USER {user}"
               )
               self.conn.cursor().execute("""
                   UPDATE security.jit_access_log
                   SET revoked_at = CURRENT_TIMESTAMP()
                   WHERE requester = %s AND role_granted = %s
                     AND revoked_at IS NULL
               """, (user, role))
   ```

3. **Build the PII filter transform**:
   ```python
   # kafka_transforms/pii_filter.py
   """
   Kafka SMT (Single Message Transform) that strips PII fields
   before cross-region replication.
   """
   import hashlib
   import json

   class PIIFilter:
       def __init__(self, config: dict):
           self.remove_fields = config.get("fields.remove", "").split(",")
           self.hash_fields = config.get("fields.hash", "").split(",")

       def transform(self, record: dict) -> dict:
           value = json.loads(record["value"])

           # Remove PII fields entirely
           for field in self.remove_fields:
               field = field.strip()
               if field in value:
                   del value[field]

           # Hash identifier fields (preserve join-ability)
           for field in self.hash_fields:
               field = field.strip()
               if field in value and value[field]:
                   value[field] = hashlib.sha256(
                       str(value[field]).encode()
                   ).hexdigest()[:16]

           record["value"] = json.dumps(value)
           return record
   ```

4. **Build a replication lag monitor**:
   ```python
   # monitoring/replication_monitor.py
   from kafka import KafkaAdminClient, KafkaConsumer
   from dataclasses import dataclass

   @dataclass
   class ReplicationStatus:
       source_cluster: str
       target_cluster: str
       topic: str
       source_offset: int
       target_offset: int
       lag: int
       lag_seconds: float

   class ReplicationMonitor:
       def __init__(self, clusters: dict):
           self.clusters = clusters

       def check_lag(self, source: str, target: str, topic: str) -> ReplicationStatus:
           src_consumer = KafkaConsumer(
               bootstrap_servers=self.clusters[source]
           )
           tgt_consumer = KafkaConsumer(
               bootstrap_servers=self.clusters[target]
           )

           src_offsets = src_consumer.end_offsets(
               src_consumer.partitions_for_topic(topic)
           )
           tgt_offsets = tgt_consumer.end_offsets(
               tgt_consumer.partitions_for_topic(f"{source}.{topic}")
           )

           total_src = sum(src_offsets.values())
           total_tgt = sum(tgt_offsets.values())

           return ReplicationStatus(
               source_cluster=source, target_cluster=target,
               topic=topic, source_offset=total_src,
               target_offset=total_tgt,
               lag=total_src - total_tgt,
               lag_seconds=self._estimate_lag_time(total_src - total_tgt)
           )
   ```

   @dataclass
   class GrowthScenario:

       query_multiplier: float

   ```sql
   -- macros/currency/convert_currency.sql
   {% macro convert_currency(
       amount_column,
       from_currency_column,
       to_currency='USD',
       rate_table='ref("dim_exchange_rates")',
       date_column='order_date'
   ) %}
   (
       {{ amount_column }} * COALESCE(
           rates.rate,
           CASE
               WHEN {{ from_currency_column }} = '{{ to_currency }}' THEN 1.0
               ELSE NULL  -- Force failure if no rate found
           END
       )
   )
   {%- endmacro %}

   -- macros/incremental/dedup_merge.sql
   {% macro dedup_merge(
       unique_key,
       updated_at_column='_loaded_at',
       strategy='delete+insert'
   ) %}
   {# Generates a dedup CTE for incremental models #}
   WITH ranked AS (
       SELECT
           *,
           ROW_NUMBER() OVER (
               PARTITION BY {{ unique_key }}
               ORDER BY {{ updated_at_column }} DESC
           ) AS _rn
       FROM {{ this }}
       {% if is_incremental() %}
       WHERE {{ updated_at_column }} >= (
           SELECT MAX({{ updated_at_column }}) FROM {{ this }}
       )
       {% endif %}
   )
   SELECT * EXCLUDE (_rn) FROM ranked WHERE _rn = 1
   {%- endmacro %}

   -- macros/privacy/mask_pii.sql
   {% macro mask_pii(column, method='hash') %}
   {% if method == 'hash' %}
       SHA2({{ column }}::STRING, 256)
   {% elif method == 'redact' %}
       '***REDACTED***'
   {% elif method == 'partial_email' %}
       CONCAT(LEFT({{ column }}, 2), '***@', SPLIT_PART({{ column }}, '@', 2))
   {% else %}
       {{ column }}
   {% endif %}
   {%- endmacro %}

   -- macros/audit/add_audit_columns.sql
   {% macro add_audit_columns() %}
       CURRENT_TIMESTAMP() AS _loaded_at,
       '{{ invocation_id }}' AS _dbt_invocation_id,
       '{{ this.name }}' AS _source_model
   {%- endmacro %}
   ```

4. **Build generic tests**:
   ```sql
   -- tests/generic/test_freshness_hours.sql
   {% test freshness_hours(model, column_name, max_hours=6) %}
   SELECT
       {{ column_name }} AS ts,
       DATEDIFF(hour, {{ column_name }}, CURRENT_TIMESTAMP()) AS hours_stale
   FROM {{ model }}
   WHERE {{ column_name }} = (SELECT MAX({{ column_name }}) FROM {{ model }})
   HAVING hours_stale > {{ max_hours }}
   {% endtest %}

   -- tests/generic/test_row_count_anomaly.sql
   {% test row_count_anomaly(model, min_rows=1, max_deviation_pct=50) %}
   {# Fails if today's row count deviates >X% from 7-day average #}
   WITH daily AS (
       SELECT COUNT(*) AS cnt FROM {{ model }}
   )
   SELECT cnt
   FROM daily
   WHERE cnt < {{ min_rows }}
      OR cnt < (SELECT COUNT(*) * (1 - {{ max_deviation_pct }}/100.0)
                FROM {{ model }})
   {% endtest %}
   ```

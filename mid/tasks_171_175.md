## Task 171: Lakehouse Architecture — Unifying Batch and Streaming

### Business Context
NovaMart runs two parallel systems: a data lake (S3 + Spark) for batch processing and Snowflake for analytics. This "two-system" approach causes problems: data is duplicated, freshness is inconsistent, and the team maintains two pipelines for the same data. Priya says: "The industry is moving toward lakehouse — one system that does both. Evaluate whether Apache Iceberg on Snowflake or Databricks Delta Lake can unify our architecture."

### Objective
Design and prototype a lakehouse architecture that unifies batch and streaming workloads on a single storage layer with ACID transactions.

### Technical Requirements
- **Tools**: Apache Iceberg, Snowflake (Iceberg Tables), Spark, Python
- **Skills**: Lakehouse architecture, table formats, ACID on object storage, time travel

### Step-by-Step Guidance

1. **Map current dual architecture and pain points**:
   ```markdown
   ## Current State vs Lakehouse Target

   | Aspect | Current (Dual) | Lakehouse (Unified) |
   |--------|---------------|---------------------|
   | Storage | S3 (Parquet) + Snowflake (internal) | S3 (Iceberg) — single copy |
   | Batch | Spark reads S3 | Spark reads Iceberg |
   | Analytics | Snowflake queries internal tables | Snowflake queries Iceberg external |
   | Streaming | Flink → S3 → Spark → Snowflake | Flink → Iceberg directly |
   | Freshness | 2-6 hours (batch lag) | Minutes (streaming append) |
   | Cost | 2x storage (duplicated) | 1x storage |
   | Time travel | Snowflake only (90 days) | Iceberg snapshots (configurable) |
   ```

2. **Build the Iceberg prototype**:
   ```python
   # lakehouse/iceberg_setup.py
   from pyspark.sql import SparkSession

   spark = (SparkSession.builder
       .appName("NovaMart-Lakehouse")
       .config("spark.sql.catalog.novamart", "org.apache.iceberg.spark.SparkCatalog")
       .config("spark.sql.catalog.novamart.type", "rest")
       .config("spark.sql.catalog.novamart.uri", "http://iceberg-rest:8181")
       .config("spark.sql.catalog.novamart.warehouse", "s3://novamart-lakehouse/")
       .config("spark.sql.extensions",
               "org.apache.iceberg.spark.extensions.IcebergSparkSessionExtensions")
       .getOrCreate()
   )

   # Create an Iceberg table
   spark.sql("""
       CREATE TABLE IF NOT EXISTS novamart.gold.fact_orders (
           order_id STRING,
           customer_id STRING,
           order_date DATE,
           total_amount DECIMAL(12, 2),
           status STRING,
           region STRING,
           _loaded_at TIMESTAMP
       )
       USING iceberg
       PARTITIONED BY (days(order_date), region)
       TBLPROPERTIES (
           'write.format.default' = 'parquet',
           'write.metadata.compression-codec' = 'gzip',
           'history.expire.max-snapshot-age-ms' = '604800000'
       )
   """)

   # Streaming write to Iceberg (from Kafka via Spark)
   orders_stream = (spark.readStream
       .format("kafka")
       .option("subscribe", "novamart.orders")
       .option("kafka.bootstrap.servers", "localhost:9092")
       .load()
       .selectExpr("CAST(value AS STRING)")
   )

   # Parse and write to Iceberg
   parsed = orders_stream.select(
       from_json(col("value"), order_schema).alias("data")
   ).select("data.*")

   (parsed.writeStream
       .format("iceberg")
       .outputMode("append")
       .option("checkpointLocation", "s3://novamart-lakehouse/checkpoints/orders")
       .toTable("novamart.gold.fact_orders")
   )
   ```

3. **Build the event publisher library**:
   ```python
   # eventbus/publisher.py
   from confluent_kafka import SerializingProducer
   from confluent_kafka.schema_registry import SchemaRegistryClient
   from confluent_kafka.schema_registry.avro import AvroSerializer
   import uuid
   from datetime import datetime

   class EventPublisher:
       def __init__(self, bootstrap_servers: str, schema_registry_url: str):
           self.registry = SchemaRegistryClient({"url": schema_registry_url})
           self._producers = {}

       def publish(self, topic: str, event_type: str, payload: dict,
                   key: str = None):
           if topic not in self._producers:
               schema_str = self.registry.get_latest_version(
                   f"{topic}-value"
               ).schema.schema_str
               serializer = AvroSerializer(self.registry, schema_str)
               self._producers[topic] = SerializingProducer({
                   "bootstrap.servers": self.bootstrap_servers,
                   "value.serializer": serializer,
               })

           # Add standard envelope
           payload["event_id"] = str(uuid.uuid4())
           payload["event_time"] = int(datetime.utcnow().timestamp() * 1000)

           self._producers[topic].produce(
               topic=topic, key=key, value=payload,
               on_delivery=self._delivery_report
           )
           self._producers[topic].flush()

       def _delivery_report(self, err, msg):
           if err:
               raise Exception(f"Event delivery failed: {err}")
   ```

4. **Build a consumer framework for data pipelines**:
   ```python
   # eventbus/consumer.py
   from confluent_kafka import DeserializingConsumer
   from confluent_kafka.schema_registry.avro import AvroDeserializer
   from abc import ABC, abstractmethod

   class EventHandler(ABC):
       @abstractmethod
       def handle(self, event: dict) -> None:
           pass

       @abstractmethod
       def on_error(self, event: dict, error: Exception) -> None:
           pass

   class EventConsumer:
       def __init__(self, topics: list, group_id: str,
                    handler: EventHandler, config: dict):
           deserializer = AvroDeserializer(
               SchemaRegistryClient({"url": config["schema_registry_url"]})
           )
           self.consumer = DeserializingConsumer({
               "bootstrap.servers": config["bootstrap_servers"],
               "group.id": group_id,
               "auto.offset.reset": "earliest",
               "value.deserializer": deserializer,
           })
           self.consumer.subscribe(topics)
           self.handler = handler

       def run(self):
           while True:
               msg = self.consumer.poll(timeout=1.0)
               if msg is None:
                   continue
               if msg.error():
                   self.handler.on_error({}, msg.error())
                   continue
               try:
                   self.handler.handle(msg.value())
               except Exception as e:
                   self.handler.on_error(msg.value(), e)
   ```

## Task 174: Data Encryption — End-to-End Encryption at Rest and in Transit

### Business Context
NovaMart's security audit (Task 167) identified a gap: while Snowflake encrypts data at rest, the data lake on S3 uses default encryption, Kafka traffic is unencrypted between brokers, and temporary files in Airflow workers contain PII in plaintext. The CISO mandates: "End-to-end encryption. Data must be encrypted at rest, in transit, and during processing — everywhere."

### Objective
Implement comprehensive encryption for the data platform: at rest (storage), in transit (network), and application-level encryption for sensitive fields.

### Technical Requirements
- **Tools**: AWS KMS, S3 (SSE-KMS), Kafka (TLS/SSL), Python (cryptography library)
- **Skills**: Encryption at rest/in transit, key management, envelope encryption, field-level encryption

2. **Upgrade S3 to KMS encryption**:

   # encryption/s3_encryption.py
   import boto3

   def enforce_kms_encryption(bucket_name: str, kms_key_id: str):
       """Set S3 bucket to require KMS encryption for all objects."""
       s3 = boto3.client("s3")

       # Set default encryption
       s3.put_bucket_encryption(
           Bucket=bucket_name,
           ServerSideEncryptionConfiguration={
               "Rules": [{
                   "ApplyServerSideEncryptionByDefault": {
                       "SSEAlgorithm": "aws:kms",
                       "KMSMasterKeyID": kms_key_id
                   },
                   "BucketKeyEnabled": True  # Reduces KMS API costs
               }]
           }
       )

       # Deny unencrypted uploads via bucket policy
       policy = {
           "Version": "2012-10-17",
           "Statement": [{
               "Sid": "DenyUnencryptedUploads",
               "Effect": "Deny",
               "Principal": "*",
               "Action": "s3:PutObject",
               "Resource": f"arn:aws:s3:::{bucket_name}/*",
               "Condition": {
                   "StringNotEquals": {
                       "s3:x-amz-server-side-encryption": "aws:kms"
                   }
               }
           }]
       }
       s3.put_bucket_policy(Bucket=bucket_name, Policy=json.dumps(policy))

4. **Implement field-level encryption for PII**:

   # encryption/field_encryption.py
   from cryptography.fernet import Fernet
   import boto3
   import base64

   class FieldEncryptor:
       """Encrypt/decrypt individual fields using envelope encryption."""

       def __init__(self, kms_key_id: str, region: str = "us-east-1"):
           self.kms = boto3.client("kms", region_name=region)
           self.kms_key_id = kms_key_id
           self._data_key = None
           self._fernet = None
           self._rotate_data_key()

       def _rotate_data_key(self):
           response = self.kms.generate_data_key(
               KeyId=self.kms_key_id, KeySpec="AES_256"
           )
           self._data_key = response["Plaintext"]
           self._encrypted_data_key = response["CiphertextBlob"]
           self._fernet = Fernet(
               base64.urlsafe_b64encode(self._data_key[:32])
           )

       def encrypt_field(self, plaintext: str) -> str:
           if plaintext is None:
               return None
           return self._fernet.encrypt(plaintext.encode()).decode()

       def decrypt_field(self, ciphertext: str) -> str:
           if ciphertext is None:
               return None
           return self._fernet.decrypt(ciphertext.encode()).decode()

       def encrypt_dataframe(self, df, columns: list):
           """Encrypt specific columns in a pandas DataFrame."""
           for col in columns:
               if col in df.columns:
                   df[col] = df[col].apply(
                       lambda x: self.encrypt_field(str(x)) if x else None
                   )
           return df
   ```

   # performance/profiler.py
   import pandas as pd
   from datetime import timedelta

   def get_dag_task_durations(dag_id: str, execution_date: str) -> pd.DataFrame:
       """Get task durations from Airflow metadata DB."""
       return pd.read_sql(f"""
           SELECT
               task_id,
               start_date,
               end_date,
               duration AS duration_sec,
               state,
               upstream_task_ids
           FROM task_instance
           WHERE dag_id = '{dag_id}'
             AND execution_date = '{execution_date}'
           ORDER BY start_date
       """, airflow_conn)

   def find_critical_path(tasks: pd.DataFrame) -> list[str]:
       """
       Find the longest path through the DAG (critical path).
       Tasks NOT on the critical path can be delayed without
       affecting total runtime.
       """
       # Build dependency graph
       graph = {}
       for _, row in tasks.iterrows():
           graph[row["task_id"]] = {
               "duration": row["duration_sec"],
               "deps": row["upstream_task_ids"] or [],
               "earliest_start": 0,
               "earliest_finish": 0,
           }

       # Forward pass: earliest start/finish
       for task_id in topological_sort(graph):
           node = graph[task_id]
           if node["deps"]:
               node["earliest_start"] = max(
                   graph[d]["earliest_finish"] for d in node["deps"]
               )
           node["earliest_finish"] = (
               node["earliest_start"] + node["duration"]
           )

       # The critical path is tasks where slack = 0
       total_time = max(n["earliest_finish"] for n in graph.values())
       critical = [
           tid for tid, n in graph.items()
           if n["earliest_finish"] == total_time
              or n["earliest_start"] + n["duration"] == n["earliest_finish"]
       ]
       return critical
   ```

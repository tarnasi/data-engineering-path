## Task 181: Data Quality as a Service — Self-Serve Quality Framework

### Business Context
Domain teams are frustrated. They want data quality checks but don't want to learn dbt tests, Great Expectations, or write SQL assertions. A Marketing analyst says: "I just want to say 'this column should never be null' and 'row count should be within 20% of yesterday' — why do I need to write code?" You need to abstract the complexity behind a simple interface.

### Objective
Build a self-serve data quality framework where domain teams define quality rules in simple YAML, and the platform automatically generates, schedules, and monitors the tests.

### Technical Requirements
- **Tools**: Python, YAML, dbt (generic tests), SQL (Snowflake), Slack
- **Skills**: Framework design, abstraction layers, self-serve platforms, test generation

2. **Build the quality rule engine**:
   ```python
   # quality/rule_engine.py
   from abc import ABC, abstractmethod
   from dataclasses import dataclass

   @dataclass
   class QualityResult:
       rule_name: str
       dataset: str
       passed: bool
       severity: str
       failing_rows: int
       total_rows: int
       details: str

   class QualityRule(ABC):
       @abstractmethod
       def generate_sql(self, dataset: str, config: dict) -> str:
           pass

   class NotNullRule(QualityRule):
       def generate_sql(self, dataset: str, config: dict) -> str:
           return f"""
               SELECT COUNT(*) AS failures
               FROM {dataset}
               WHERE {config['column']} IS NULL
           """

   class RangeRule(QualityRule):
       def generate_sql(self, dataset: str, config: dict) -> str:
           return f"""
               SELECT COUNT(*) AS failures
               FROM {dataset}
               WHERE {config['column']} < {config['min']}
                  OR {config['column']} > {config['max']}
                  OR {config['column']} IS NULL
           """

   class RowCountAnomalyRule(QualityRule):
       def generate_sql(self, dataset: str, config: dict) -> str:
           return f"""
               WITH current_count AS (
                   SELECT COUNT(*) AS cnt FROM {dataset}
               ),
               historical AS (
                   SELECT AVG(row_count) AS avg_count
                   FROM quality.row_count_history
                   WHERE table_name = '{dataset}'
                     AND measured_at >= DATEADD(day, -7, CURRENT_DATE())
               )
               SELECT
                   CASE WHEN ABS(c.cnt - h.avg_count) / NULLIF(h.avg_count, 0) * 100
                       > {config['max_deviation_pct']}
                   THEN c.cnt ELSE 0 END AS failures
               FROM current_count c, historical h
           """

   class FreshnessRule(QualityRule):
       def generate_sql(self, dataset: str, config: dict) -> str:
           return f"""
               SELECT CASE
                   WHEN DATEDIFF(hour,
                       MAX({config['column']}),
                       CURRENT_TIMESTAMP()
                   ) > {config['max_hours']}
                   THEN 1 ELSE 0 END AS failures
               FROM {dataset}
           """

   RULE_REGISTRY = {
       "not_null": NotNullRule(),
       "range": RangeRule(),
       "row_count_anomaly": RowCountAnomalyRule(),
       "freshness": FreshnessRule(),
       "unique": UniqueRule(),
       "accepted_values": AcceptedValuesRule(),
       "foreign_key": ForeignKeyRule(),
   }

   class QualityEngine:
       def __init__(self, conn):
           self.conn = conn

       def evaluate_rules(self, config_path: str) -> list[QualityResult]:
           import yaml
           with open(config_path) as f:
               config = yaml.safe_load(f)

           results = []
           for rule_config in config["rules"]:
               rule = RULE_REGISTRY[rule_config["type"]]
               sql = rule.generate_sql(config["dataset"], rule_config)
               failures = self.conn.cursor().execute(sql).fetchone()[0]
               results.append(QualityResult(
                   rule_name=rule_config["name"],
                   dataset=config["dataset"],
                   passed=failures == 0,
                   severity=rule_config["severity"],
                   failing_rows=failures,
                   total_rows=self._count_rows(config["dataset"]),
                   details=f"{failures} rows failed {rule_config['type']} check"
               ))
           return results
   ```

3. **Auto-generate dbt tests from YAML rules**:
   ```python
   # quality/dbt_generator.py
   import yaml

   def generate_dbt_schema(config_path: str) -> str:
       with open(config_path) as f:
           config = yaml.safe_load(f)

       schema = config["dataset"].split(".")[-1]
       output = {
           "version": 2,
           "models": [{
               "name": schema,
               "columns": []
           }]
       }

       for rule in config["rules"]:
           if rule["type"] == "not_null":
               output["models"][0]["columns"].append({
                   "name": rule["column"],
                   "tests": ["not_null"]
               })
           elif rule["type"] == "unique":
               output["models"][0]["columns"].append({
                   "name": rule["column"],
                   "tests": ["unique"]
               })
           elif rule["type"] == "accepted_values":
               output["models"][0]["columns"].append({
                   "name": rule["column"],
                   "tests": [{"accepted_values": {"values": rule["values"]}}]
               })

       return yaml.dump(output, default_flow_style=False)
   ```

4. **Build the quality dashboard and alerting**:
   ```python
   # quality/reporter.py
   def generate_quality_report(results: list[QualityResult]) -> str:
       critical_failures = [r for r in results if not r.passed and r.severity == "critical"]
       warning_failures = [r for r in results if not r.passed and r.severity == "warning"]
       passed = [r for r in results if r.passed]

       report = f"""
   ## Data Quality Report — {results[0].dataset if results else 'N/A'}

   | Status | Count |
   |--------|-------|
   | ✅ Passed | {len(passed)} |
   | ⚠️ Warnings | {len(warning_failures)} |
   | ❌ Critical Failures | {len(critical_failures)} |
   """
       if critical_failures:
           report += "\n### Critical Failures (blocking)\n"
           for r in critical_failures:
               report += f"- **{r.rule_name}**: {r.details}\n"
       return report
   ```

## Task 183: Multi-Tenant Data Platform — Isolation and Fair Sharing

### Business Context
NovaMart's data platform serves 6 domain teams. Marketing's team just ran a massive ad-hoc query that consumed all available Snowflake compute for 45 minutes, blocking Finance's month-end close reports. The CFO is not pleased. ### Objective

### Technical Requirements
- **Tools**: Snowflake (resource monitors, warehouses), Airflow (pools, priority), Python
- **Skills**: Multi-tenancy, resource isolation, fair scheduling, workload management

1. **Design the tenant isolation model**:

   ```markdown
   | Isolation Layer | Mechanism | Purpose |
   |----------------|-----------|---------|

   | Storage | Separate schemas + RBAC | Data access control |
   ```

3. **Implement Airflow resource pools**:

   ```python
   # dags/config/pools.py
   """Configure Airflow pools for multi-tenant fair scheduling."""
   from airflow.models import Pool
   from airflow.utils.db import create_session

   POOL_CONFIG = {
       "snowflake_compute": {
           "slots": 20,       # Total concurrent Snowflake tasks
           "description": "Global Snowflake compute pool"
       },
       "commerce_slots": {
           "slots": 8,        # Commerce team gets 8 concurrent slots
           "description": "Commerce team pipeline slots"
       },
       "marketing_slots": {
           "slots": 4,        # Marketing gets 4 (proportional to usage)
           "description": "Marketing team pipeline slots"
       },
       "finance_slots": {
           "slots": 6,        # Finance gets 6 (higher during month-end)
           "description": "Finance team pipeline slots"
       },
       "ad_hoc_slots": {
           "slots": 2,        # Ad-hoc queries limited
           "description": "Ad-hoc analysis pool (limited)"
       },
   }

   def setup_pools():
       with create_session() as session:
           for name, config in POOL_CONFIG.items():
               pool = session.query(Pool).filter(Pool.pool == name).first()
               if not pool:
                   pool = Pool(pool=name)
               pool.slots = config["slots"]
               pool.description = config["description"]
               session.merge(pool)
   ```

3. **Add rate limiting and authentication**:
   ```python
   # api/middleware.py
   from fastapi import Request
   from starlette.middleware.base import BaseHTTPMiddleware
   import time

   class RateLimitMiddleware(BaseHTTPMiddleware):
       def __init__(self, app, requests_per_minute: int = 60):
           super().__init__(app)
           self.rpm = requests_per_minute

       async def dispatch(self, request: Request, call_next):
           client_id = request.headers.get("X-API-Key", request.client.host)
           key = f"ratelimit:{client_id}:{int(time.time()) // 60}"

           current = cache.incr(key)
           if current == 1:
               cache.expire(key, 60)

           if current > self.rpm:
               return JSONResponse(
                   status_code=429,
                   content={"detail": "Rate limit exceeded",
                            "retry_after": 60}
               )

           response = await call_next(request)
           response.headers["X-RateLimit-Remaining"] = str(
               max(0, self.rpm - current)
           )
           return response
   ```

   ```python
   # compliance/evidence_collector.py
   from dataclasses import dataclass
   from datetime import datetime
   import pandas as pd

   @dataclass
   class ComplianceEvidence:
       control_id: str
       control_name: str
       evidence_type: str
       collected_at: datetime
       status: str  # "passing", "failing", "needs_review"
       data: pd.DataFrame
       summary: str

   class SOC2EvidenceCollector:
       def __init__(self, snowflake_conn):
           self.conn = snowflake_conn

       def collect_access_controls(self) -> ComplianceEvidence:
           """CC6.1: Logical access is restricted to authorized users."""
           data = pd.read_sql("""
               SELECT
                   u.name AS user_name,
                   u.disabled,
                   u.ext_authn_duo,  -- MFA status
                   u.last_success_login,
                   DATEDIFF(day, u.last_success_login, CURRENT_DATE()) AS days_since_login,
                   LISTAGG(DISTINCT g.role, ', ') AS roles
               FROM snowflake.account_usage.users u
               LEFT JOIN snowflake.account_usage.grants_to_users g
                   ON u.name = g.grantee_name AND g.deleted_on IS NULL
               WHERE u.deleted_on IS NULL
               GROUP BY 1, 2, 3, 4, 5
               ORDER BY days_since_login DESC
           """, self.conn)

           # Flag issues
           inactive = data[data["days_since_login"] > 90]
           no_mfa = data[data["ext_authn_duo"] == "FALSE"]

           status = "passing"
           if len(inactive) > 0 or len(no_mfa) > 0:
               status = "failing"

           return ComplianceEvidence(
               control_id="CC6.1",
               control_name="Logical Access Controls",
               evidence_type="user_access_review",
               collected_at=datetime.utcnow(),
               status=status,
               data=data,
               summary=(
                   f"{len(data)} total users, "
                   f"{len(inactive)} inactive >90 days, "
                   f"{len(no_mfa)} without MFA"
               )
           )

       def collect_change_management(self) -> ComplianceEvidence:
           """CC7.1: Changes are authorized and tracked."""
           data = pd.read_sql("""
               SELECT
                   query_id,
                   user_name,
                   role_name,
                   query_type,
                   query_text,
                   start_time,
                   execution_status
               FROM snowflake.account_usage.query_history
               WHERE query_type IN ('CREATE_TABLE', 'ALTER_TABLE', 
                                     'DROP_TABLE', 'CREATE_VIEW')
                 AND start_time >= DATEADD(month, -1, CURRENT_DATE())
               ORDER BY start_time DESC
           """, self.conn)

           return ComplianceEvidence(
               control_id="CC7.1",
               control_name="Change Management",
               evidence_type="schema_changes",
               collected_at=datetime.utcnow(),
               status="needs_review",
               data=data,
               summary=f"{len(data)} schema changes in last 30 days"
           )

       def collect_all(self) -> list[ComplianceEvidence]:
           return [
               self.collect_access_controls(),
               self.collect_change_management(),
               self.collect_data_retention(),
               self.collect_encryption_status(),
               self.collect_incident_metrics(),
           ]
   ```

3. **Implement data retention enforcement**:
   ```python
   # compliance/retention_policy.py
   RETENTION_POLICIES = {
       "raw_commerce": {"retention_days": 365, "action": "archive_to_glacier"},
       "staging": {"retention_days": 90, "action": "delete"},
       "gold_commerce": {"retention_days": 730, "action": "archive_to_glacier"},
       "temp": {"retention_days": 7, "action": "delete"},
       "audit_logs": {"retention_days": 2555, "action": "retain"},  # 7 years
   }

   class RetentionEnforcer:
       def __init__(self, conn):
           self.conn = conn

       def enforce(self, schema: str, policy: dict) -> dict:
           retention_days = policy["retention_days"]
           action = policy["action"]

           # Find tables with data beyond retention
           expired = pd.read_sql(f"""
               SELECT table_name, 
                      MIN(_loaded_at) AS oldest_record,
                      COUNT(*) AS expired_rows
               FROM {schema}.*
               WHERE _loaded_at < DATEADD(day, -{retention_days}, CURRENT_DATE())
               GROUP BY table_name
           """, self.conn)

           results = {"schema": schema, "action": action, "tables": []}
           for _, row in expired.iterrows():
               if action == "delete":
                   self.conn.cursor().execute(f"""
                       DELETE FROM {schema}.{row['table_name']}
                       WHERE _loaded_at < DATEADD(day, -{retention_days}, CURRENT_DATE())
                   """)
               elif action == "archive_to_glacier":
                   self._archive_to_glacier(schema, row["table_name"], retention_days)
               results["tables"].append({
                   "table": row["table_name"],
                   "rows_affected": row["expired_rows"]
               })
           return results
   ```

   ```python
   # compliance/report_generator.py
   def generate_soc2_report(evidence_list: list[ComplianceEvidence]) -> str:
       report = f"""
   # SOC 2 Compliance Evidence Report
   **Generated**: {datetime.utcnow().strftime('%Y-%m-%d %H:%M UTC')}
   **Period**: Last 30 days

   ## Summary

   | Control | Status | Summary |
   |---------|--------|---------|
   """
       for e in evidence_list:
           icon = {"passing": "✅", "failing": "❌", "needs_review": "⚠️"}[e.status]
           report += f"| {e.control_id}: {e.control_name} | {icon} | {e.summary} |\n"

       report += "\n## Detailed Evidence\n\n"
       for e in evidence_list:
           report += f"### {e.control_id}: {e.control_name}\n"
           report += f"**Collected**: {e.collected_at}\n"
           report += f"**Status**: {e.status}\n"
           report += f"**Summary**: {e.summary}\n\n"

       return report
   ```

1. **Extract lineage from dbt**:
   ```python
   # lineage/dbt_lineage.py
   import json
   from collections import defaultdict

   class DbtLineageExtractor:
       def __init__(self, manifest_path: str):
           with open(manifest_path) as f:
               self.manifest = json.load(f)

       def get_column_lineage(self, model_name: str) -> dict:
           """Get column-level lineage for a dbt model."""
           node_key = f"model.novamart.{model_name}"
           node = self.manifest["nodes"].get(node_key)
           if not node:
               return {}

           return {
               "model": model_name,
               "columns": node.get("columns", {}),
               "depends_on": node["depends_on"]["nodes"],
               "referenced_by": self._find_downstream(node_key),
               "raw_sql": node.get("raw_code", ""),
           }

       def _find_downstream(self, node_key: str) -> list:
           downstream = []
           for key, node in self.manifest["nodes"].items():
               if node_key in node.get("depends_on", {}).get("nodes", []):
                   downstream.append(key)
           # Also check exposures (dashboards)
           for key, exposure in self.manifest.get("exposures", {}).items():
               if node_key in exposure.get("depends_on", {}).get("nodes", []):
                   downstream.append(key)
           return downstream

       def build_full_graph(self) -> dict:
           """Build the complete lineage graph."""
           graph = {"nodes": [], "edges": []}
           for key, node in self.manifest["nodes"].items():
               graph["nodes"].append({
                   "id": key,
                   "name": node.get("name"),
                   "type": node.get("resource_type"),
                   "schema": node.get("schema"),
               })
               for dep in node.get("depends_on", {}).get("nodes", []):
                   graph["edges"].append({"source": dep, "target": key})
           return graph
   ```

3. **Build automated impact analysis**:
   ```python
   # lineage/impact_analyzer.py
   from collections import deque

   class ImpactAnalyzer:
       def __init__(self, graph: dict):
           self.adjacency = self._build_adjacency(graph)

       def _build_adjacency(self, graph: dict) -> dict:
           adj = {}
           for edge in graph["edges"]:
               adj.setdefault(edge["source"], []).append(edge["target"])
           return adj

       def analyze_impact(self, changed_node: str) -> dict:
           """BFS to find all downstream impacts."""
           visited = set()
           queue = deque([(changed_node, 0)])
           impacts = {"tables": [], "dashboards": [], "ml_models": []}

           while queue:
               node, depth = queue.popleft()
               if node in visited:
                   continue
               visited.add(node)

               # Classify the impacted node
               if "exposure" in node:
                   impacts["dashboards"].append({"node": node, "depth": depth})
               elif "model" in node:
                   impacts["tables"].append({"node": node, "depth": depth})

               for downstream in self.adjacency.get(node, []):
                   if downstream not in visited:
                       queue.append((downstream, depth + 1))

           return {
               "changed": changed_node,
               "total_impacted": len(visited) - 1,
               "impacts": impacts,
           }

   # Usage:

   # result = analyzer.analyze_impact("model.novamart.dim_promotions")
   # → {"total_impacted": 14, "dashboards": ["executive_revenue", ...]}
   ```

   ```python
   # runbooks/search.py
   import os
   import yaml
   import re

   class RunbookIndex:
       def __init__(self, runbook_dir: str = "runbooks/"):
           self.runbooks = self._load_all(runbook_dir)

       def _load_all(self, directory: str) -> list:
           runbooks = []
           for f in os.listdir(directory):
               if f.endswith(".md"):
                   with open(os.path.join(directory, f)) as fh:
                       content = fh.read()
                       # Parse YAML front matter
                       match = re.match(r"---\n(.*?)\n---", content, re.DOTALL)
                       if match:
                           meta = yaml.safe_load(match.group(1))
                           meta["file"] = f
                           meta["content"] = content
                           runbooks.append(meta)
           return runbooks

       def find_by_alert(self, alert_name: str) -> list:
           return [r for r in self.runbooks if r.get("alert_name") == alert_name]

       def search(self, query: str) -> list:
           query_lower = query.lower()
           results = []
           for rb in self.runbooks:
               score = 0
               if query_lower in rb.get("title", "").lower():
                   score += 10
               if query_lower in str(rb.get("tags", [])).lower():
                   score += 5
               if query_lower in rb.get("content", "").lower():
                   score += 1
               if score > 0:
                   results.append((rb, score))
           return sorted(results, key=lambda x: -x[1])
   ```

2. **Build the Snowflake cost attribution**:
   ```sql
   -- Compute costs per team (using warehouse tagging)
   WITH warehouse_costs AS (
       SELECT
           warehouse_name,
           SUM(credits_used) AS total_credits,
           SUM(credits_used) * 3.50 AS cost_usd  -- $3.50 per credit
       FROM snowflake.account_usage.warehouse_metering_history
       WHERE start_time >= DATE_TRUNC('month', CURRENT_DATE())
       GROUP BY warehouse_name
   ),
   team_mapping AS (
       SELECT * FROM (VALUES
           ('COMMERCE_WH', 'Commerce'),
           ('MARKETING_WH', 'Marketing'),
           ('FINANCE_WH', 'Finance'),
           ('PLATFORM_WH', 'Platform'),
           ('ANALYTICS_WH', 'Shared')
       ) AS t(warehouse_name, team)
   )
   SELECT
       tm.team,
       wc.warehouse_name,
       wc.total_credits,
       wc.cost_usd
   FROM warehouse_costs wc
   JOIN team_mapping tm ON wc.warehouse_name = tm.warehouse_name
   ORDER BY wc.cost_usd DESC;

   -- Storage costs per team (by schema ownership)
   SELECT
       table_schema,
       CASE
           WHEN table_schema LIKE '%COMMERCE%' THEN 'Commerce'
           WHEN table_schema LIKE '%MARKETING%' THEN 'Marketing'
           WHEN table_schema LIKE '%FINANCE%' THEN 'Finance'
           ELSE 'Shared'
       END AS team,
       SUM(active_bytes) / POWER(1024, 4) AS tb,
       SUM(active_bytes) / POWER(1024, 4) * 23.0 AS cost_usd  -- $23/TB/month
   FROM snowflake.account_usage.table_storage_metrics
   WHERE active_bytes > 0
   GROUP BY 1, 2
   ORDER BY cost_usd DESC;
   ```

   ```python
   from dataclasses import dataclass
   import pandas as pd

   @dataclass
   class TeamCostReport:
       team: str
       month: str
       compute_cost: float
       storage_cost: float
       kafka_cost: float
       s3_cost: float
       shared_infra_cost: float
       total_cost: float
       cost_per_pipeline: float
       month_over_month_change_pct: float

   class CostAttributionEngine:
       def __init__(self, snowflake_conn):
           self.conn = snowflake_conn

       def generate_team_report(self, team: str, month: str) -> TeamCostReport:
           compute = self._get_compute_cost(team, month)
           storage = self._get_storage_cost(team, month)
           kafka = self._get_kafka_cost(team, month)
           s3 = self._get_s3_cost(team, month)
           shared = self._get_shared_infra_split(month)
           total = compute + storage + kafka + s3 + shared
           pipeline_count = self._get_pipeline_count(team)
           prev_total = self._get_previous_month_total(team, month)
           mom_change = ((total - prev_total) / max(prev_total, 1)) * 100

           return TeamCostReport(
               team=team, month=month,
               compute_cost=round(compute, 2),
               storage_cost=round(storage, 2),
               kafka_cost=round(kafka, 2),
               s3_cost=round(s3, 2),
               shared_infra_cost=round(shared, 2),
               total_cost=round(total, 2),
               cost_per_pipeline=round(total / max(pipeline_count, 1), 2),
               month_over_month_change_pct=round(mom_change, 1)
           )

       def generate_monthly_summary(self, month: str) -> pd.DataFrame:
           teams = ["Commerce", "Marketing", "Finance", "Platform", "Logistics"]
           reports = [self.generate_team_report(t, month) for t in teams]
   ```

3. **Build the incident tracking system**:
   ```python
   # incidents/tracker.py
   from dataclasses import dataclass, field
   from datetime import datetime
   from enum import Enum

   class Severity(Enum):
       P1 = "P1_critical"
       P2 = "P2_high"
       P3 = "P3_medium"

   class IncidentStatus(Enum):
       ACTIVE = "active"
       MITIGATED = "mitigated"
       RESOLVED = "resolved"
       POSTMORTEM_PENDING = "postmortem_pending"
       CLOSED = "closed"

   @dataclass
   class Incident:
       id: str
       title: str
       severity: Severity
       status: IncidentStatus
       commander: str
       detected_at: datetime
       resolved_at: datetime = None
       detection_method: str = ""  # "automated" or "human"
       time_to_detect_min: int = 0
       time_to_resolve_min: int = 0
       action_items_total: int = 0
       action_items_closed: int = 0

   class IncidentMetrics:
       def __init__(self, incidents: list[Incident]):
           self.incidents = incidents

       def mttr(self) -> float:
           """Mean Time to Resolve (minutes)."""
           resolved = [i for i in self.incidents if i.resolved_at]
           if not resolved:
               return 0
           return sum(i.time_to_resolve_min for i in resolved) / len(resolved)

       def mttd(self) -> float:
           """Mean Time to Detect (minutes)."""
           return sum(i.time_to_detect_min for i in self.incidents) / max(len(self.incidents), 1)

       def action_item_completion_rate(self) -> float:
           total = sum(i.action_items_total for i in self.incidents)
           closed = sum(i.action_items_closed for i in self.incidents)
           return (closed / max(total, 1)) * 100
   ```

- Incident tracker: `tasks/179_incidents/incidents/tracker.py`

- Action items with no owner/deadline — "we should add a test" never gets done
- Not tracking detection method — if humans find incidents before alerts, your monitoring is broken
- Postmortems without follow-through — the learning is in the action items, not the document

## Task 180: Advanced Airflow — Dynamic DAGs and Plugin Architecture

### Business Context
NovaMart has 80+ Airflow DAGs that are 70% identical: extract from source → transform in Snowflake → test → notify. Each new pipeline requires copy-pasting a DAG file and changing a few parameters. This leads to inconsistency and bugs. Priya says: "Build a DAG factory. Define pipelines in YAML config files, and generate DAGs programmatically. One pattern, many pipelines."

### Objective
Build a DAG factory that generates Airflow DAGs from YAML configuration, reducing boilerplate by 90% and ensuring consistency across all pipelines.

### Technical Requirements
- **Tools**: Airflow 2.x, Python, YAML, Jinja2
- **Skills**: Dynamic DAG generation, Airflow plugins, configuration-driven pipelines, custom operators

2. **Build the DAG factory**:
   ```python
   # dags/factory/dag_factory.py
   import yaml
   import os
   from datetime import timedelta
   from airflow import DAG
   from airflow.operators.python import PythonOperator
   from airflow.providers.dbt.cloud.operators.dbt import DbtCloudRunJobOperator
   from airflow.operators.empty import EmptyOperator
   from pathlib import Path

   def create_dag_from_config(config_path: str) -> DAG:
       with open(config_path) as f:
           config = yaml.safe_load(f)

       pipeline = config["pipeline"]

       default_args = {
           "owner": pipeline["owner"],
           "retries": pipeline.get("retry", {}).get("count", 1),
           "retry_delay": timedelta(
               minutes=pipeline.get("retry", {}).get("delay_minutes", 5)
           ),
           "sla": timedelta(hours=pipeline.get("sla_hours", 6)),
       }

       dag = DAG(
           dag_id=pipeline["name"],
           default_args=default_args,
           schedule_interval=pipeline["schedule"],
           tags=pipeline.get("tags", []),
           catchup=False,
           max_active_runs=1,
       )

       with dag:
           start = EmptyOperator(task_id="start")
           end = EmptyOperator(task_id="end")

           # Extract tasks
           extract_tasks = []
           if "extract" in config:
               for table in config["extract"]["tables"]:
                   task = PythonOperator(
                       task_id=f"extract_{table['name']}",
                       python_callable=extract_table,
                       op_kwargs={
                           "connection_id": config["extract"]["connection_id"],
                           "schema": config["extract"]["schema"],
                           "table": table["name"],
                           "incremental_key": table.get("incremental_key"),
                       },
                   )
                   start >> task
                   extract_tasks.append(task)

           # Transform tasks
           transform_task = PythonOperator(
               task_id="dbt_run",
               python_callable=run_dbt_models,
               op_kwargs={
                   "project": config["transform"]["project"],
                   "models": config["transform"]["models"],
                   "vars": config["transform"].get("vars", {}),
               },
           )
           for et in extract_tasks:
               et >> transform_task

           # Quality tasks
           if "quality" in config:
               quality_task = PythonOperator(
                   task_id="quality_checks",
                   python_callable=run_quality_checks,
                   op_kwargs={"checks": config["quality"]},
               )
               transform_task >> quality_task
               quality_task >> end
           else:
               transform_task >> end

       return dag

   # Auto-discover and register all pipeline configs
   CONFIG_DIR = Path(__file__).parent.parent / "config" / "pipelines"

   for config_file in CONFIG_DIR.glob("*.yaml"):
       dag = create_dag_from_config(str(config_file))
       globals()[dag.dag_id] = dag
   ```

3. **Build custom operators for common patterns**:
   ```python
   # plugins/operators/extract_operator.py
   from airflow.models import BaseOperator
   from airflow.hooks.base import BaseHook

   class IncrementalExtractOperator(BaseOperator):
       """Extract data incrementally from a source database."""

       template_fields = ["sql", "target_path"]

       def __init__(self, connection_id: str, table: str,
                    incremental_key: str, target_path: str, **kwargs):
           super().__init__(**kwargs)
           self.connection_id = connection_id
           self.table = table
           self.incremental_key = incremental_key
           self.target_path = target_path

       def execute(self, context):
           hook = BaseHook.get_hook(self.connection_id)
           last_value = self._get_highwater_mark(context)

           sql = f"""
               SELECT * FROM {self.table}
               WHERE {self.incremental_key} > '{last_value}'
               ORDER BY {self.incremental_key}
           """

           df = hook.get_pandas_df(sql)
           row_count = len(df)

           if row_count > 0:
               df.to_parquet(self.target_path, index=False)
               self._update_highwater_mark(context, df[self.incremental_key].max())

           self.log.info(f"Extracted {row_count} rows from {self.table}")
           return row_count
   ```

4. **Add a pipeline validation command**:
   ```python
   # scripts/validate_pipeline_config.py
   import yaml
   import sys
   from jsonschema import validate

   PIPELINE_SCHEMA = {
       "type": "object",
       "required": ["pipeline", "transform"],
       "properties": {
           "pipeline": {
               "type": "object",
               "required": ["name", "owner", "schedule"],
               "properties": {
                   "name": {"type": "string", "pattern": "^[a-z_]+$"},
                   "schedule": {"type": "string"},
                   "owner": {"type": "string"},
               }
           },
           "extract": {"type": "object"},
           "transform": {"type": "object"},
           "quality": {"type": "object"},
           "notify": {"type": "object"},
       }
   }

   def validate_config(path: str) -> list:
       errors = []
       with open(path) as f:
           config = yaml.safe_load(f)
       try:
           validate(instance=config, schema=PIPELINE_SCHEMA)
       except Exception as e:
           errors.append(str(e))
       return errors

   if __name__ == "__main__":
       for path in sys.argv[1:]:
           errors = validate_config(path)
           if errors:
               print(f"❌ {path}: {errors}")
               sys.exit(1)
           print(f"✅ {path}: valid")
   ```

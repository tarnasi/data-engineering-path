2. **Build the technology evaluation framework**:
   ```python
   # vendor/evaluation_framework.py
   from dataclasses import dataclass

   @dataclass
   class TechnologyEvaluation:
       name: str
       category: str
       scores: dict  # category -> score (1-5)
       total_score: float = 0
       recommendation: str = ""

   class TechEvaluator:
       """Structured framework for evaluating technologies."""

       CRITERIA = {
           "technical_fit": {
               "weight": 0.25,
               "questions": [
                   "Does it solve our specific problem well?",
                   "Does it integrate with our existing stack?",
                   "Can it handle our scale (current and 5x)?",
                   "Is the API/SDK well-designed?",
               ],
           },
           "operational_maturity": {
               "weight": 0.20,
               "questions": [
                   "Is it battle-tested in production at similar scale?",
                   "Quality of monitoring, logging, alerting?",
                   "Disaster recovery and backup capabilities?",
                   "SLA guarantees?",
               ],
           },
           "team_capability": {
               "weight": 0.15,
               "questions": [
                   "Does our team have experience with this tool?",
                   "Learning curve for new team members?",
                   "Community size and support quality?",
                   "Documentation quality?",
               ],
           },
           "total_cost": {
               "weight": 0.20,
               "questions": [
                   "Licensing / subscription cost?",
                   "Implementation cost (engineer-weeks)?",
                   "Ongoing operational cost?",
                   "Migration cost if we switch later?",
               ],
           },
           "strategic_alignment": {
               "weight": 0.20,
               "questions": [
                   "Does it reduce vendor lock-in or increase it?",
                   "Is the vendor financially stable?",
                   "Roadmap alignment with our needs?",
                   "Does it support our multi-cloud strategy?",
               ],
           },
       }

       def evaluate(self, name: str, category: str,
                    scores: dict[str, float]) -> TechnologyEvaluation:
           """Score a technology on all criteria."""
           weighted_total = 0
           for criterion, config in self.CRITERIA.items():
               score = scores.get(criterion, 0)
               weighted_total += score * config["weight"]

           recommendation = self._recommend(weighted_total)

           return TechnologyEvaluation(
               name=name,
               category=category,
               scores=scores,
               total_score=round(weighted_total, 2),
               recommendation=recommendation,
           )

       def compare(self, evaluations: list[TechnologyEvaluation]) -> str:
           """Generate comparison report."""
           report = "# Technology Comparison\n\n"
           report += "| Tool | Tech Fit | Ops Maturity | Team Cap | Cost | Strategy | **Total** | Rec |\n"
           report += "|------|:--------:|:----------:|:--------:|:----:|:--------:|:---------:|:---:|\n"

           for e in sorted(evaluations, key=lambda x: x.total_score,
                          reverse=True):
               s = e.scores
               report += (
                   f"| {e.name} | {s.get('technical_fit', 0):.1f} | "
                   f"{s.get('operational_maturity', 0):.1f} | "
                   f"{s.get('team_capability', 0):.1f} | "
                   f"{s.get('total_cost', 0):.1f} | "
                   f"{s.get('strategic_alignment', 0):.1f} | "
                   f"**{e.total_score:.2f}** | {e.recommendation} |\n"
               )
           return report

       def _recommend(self, score: float) -> str:
           if score >= 4.0:
               return "✅ Adopt"
           elif score >= 3.0:
               return "🔄 Trial"
           elif score >= 2.0:
               return "⚠️ Assess"
           return "❌ Hold"
   ```

## Task 248: Data Platform Disaster Recovery and Business Continuity

### Objective

### Technical Requirements

- **Skills**: Disaster recovery planning, business continuity, infrastructure resilience, backup/restore

### Step-by-Step Guidance

1. **Define RTO/RPO for each data tier**:

2. **Design the DR architecture**:
   ```python
   # dr/disaster_recovery.py
   from dataclasses import dataclass, field
   from datetime import datetime
   from enum import Enum

   class DRStrategy(Enum):
       ACTIVE_ACTIVE = "active_active"        # Both regions serve traffic
       ACTIVE_PASSIVE = "active_passive"       # Standby region, manual failover
       BACKUP_RESTORE = "backup_restore"       # Periodic backups, restore on failure
       PILOT_LIGHT = "pilot_light"             # Minimal infra in DR, scale up on failure

   @dataclass
   class DRComponent:
       name: str
       tier: int  # 1-4
       strategy: DRStrategy
       rpo_hours: float
       rto_hours: float
       backup_frequency: str
       replication_method: str
       failover_type: str  # "automatic", "manual", "semi-automatic"
       last_tested: datetime = None
       test_result: str = ""

   class DisasterRecoveryPlan:
       def __init__(self):
           self.components = []

       def add_component(self, component: DRComponent):
           self.components.append(component)

       def define_platform_dr(self) -> list[DRComponent]:
           """Define DR strategy for each platform component."""
           return [
               DRComponent(
                   name="Snowflake",
                   tier=1,
                   strategy=DRStrategy.ACTIVE_PASSIVE,
                   rpo_hours=0,
                   rto_hours=2,
                   backup_frequency="Continuous (Time Travel + Fail-Safe)",
                   replication_method="Snowflake database replication to DR account",
                   failover_type="semi-automatic",
               ),
               DRComponent(
                   name="Kafka (Confluent)",
                   tier=1,
                   strategy=DRStrategy.ACTIVE_PASSIVE,
                   rpo_hours=0,
                   rto_hours=1,
                   backup_frequency="Continuous (multi-AZ replication)",
                   replication_method="Confluent cluster linking to DR region",
                   failover_type="manual",
               ),
               DRComponent(
                   name="S3 Data Lake",
                   tier=2,
                   strategy=DRStrategy.ACTIVE_PASSIVE,
                   rpo_hours=1,
                   rto_hours=4,
                   backup_frequency="Cross-region replication (async)",
                   replication_method="S3 Cross-Region Replication",
                   failover_type="manual",
               ),
               DRComponent(
                   name="dbt Transformations",
                   tier=2,
                   strategy=DRStrategy.BACKUP_RESTORE,
                   rpo_hours=0,
                   rto_hours=2,
                   backup_frequency="Code in Git (always recoverable)",
                   replication_method="Git repo (GitHub)",
                   failover_type="automatic",
               ),
               DRComponent(
                   name="Airflow/Orchestration",
                   tier=2,
                   strategy=DRStrategy.PILOT_LIGHT,
                   rpo_hours=1,
                   rto_hours=4,
                   backup_frequency="DAGs in Git, DB backup daily",
                   replication_method="Terraform recreate in DR region",
                   failover_type="manual",
               ),
               DRComponent(
                   name="Redis (Feature Store cache)",
                   tier=3,
                   strategy=DRStrategy.BACKUP_RESTORE,
                   rpo_hours=24,
                   rto_hours=8,
                   backup_frequency="Daily RDB snapshot",
                   replication_method="S3 backup, restore in DR",
                   failover_type="manual",
               ),
           ]

       def generate_runbook(self) -> str:
           """Generate failover runbook."""
           runbook = "# Data Platform Disaster Recovery Runbook\n\n"
           runbook += "## Activation Criteria\n"
           runbook += "Activate DR when ANY of the following occur:\n"
           runbook += "- AWS region outage > 30 minutes\n"
           runbook += "- Snowflake account outage > 1 hour\n"
           runbook += "- Primary Kafka cluster unavailable > 30 minutes\n"
           runbook += "- Data platform team declares DR\n\n"

           runbook += "## Failover Sequence\n\n"
           for i, component in enumerate(
               sorted(self.components, key=lambda c: c.tier), 1
           ):
               runbook += f"### Step {i}: {component.name} "
               runbook += f"(Tier {component.tier})\n"
               runbook += f"- **Strategy**: {component.strategy.value}\n"
               runbook += f"- **Failover**: {component.failover_type}\n"
               runbook += f"- **Expected RTO**: {component.rto_hours}h\n"
               runbook += f"- **Procedure**:\n"
               runbook += f"  1. Verify primary is truly unavailable\n"
               runbook += f"  2. Execute failover: [specific steps]\n"
               runbook += f"  3. Verify DR component is healthy\n"
               runbook += f"  4. Update DNS/connection strings\n"
               runbook += f"  5. Smoke test: run validation queries\n\n"

           return runbook
   ```

3. **Build automated backup verification**:
   ```python
   # dr/backup_verifier.py
   class BackupVerifier:
       """Verify backups are valid and restorable."""

       def __init__(self, snowflake_conn, s3_client):
           self.sf = snowflake_conn
           self.s3 = s3_client

       def verify_snowflake_replication(self) -> dict:
           """Check Snowflake replication lag."""
           result = self.sf.cursor().execute("""
               SELECT
                   database_name,
                   replication_state,
                   primary_account,
                   secondary_account,
                   DATEDIFF(second, 
                       replication_last_refreshed, 
                       CURRENT_TIMESTAMP()
                   ) AS lag_seconds
               FROM TABLE(information_schema.database_replication_status())
           """).fetchall()

           status = []
           for row in result:
               lag = row[4]
               status.append({
                   "database": row[0],
                   "state": row[1],
                   "lag_seconds": lag,
                   "healthy": lag < 3600,  # Alert if >1 hour behind
               })
           return {
               "timestamp": datetime.utcnow().isoformat(),
               "replicas": status,
               "all_healthy": all(s["healthy"] for s in status),
           }

       def verify_s3_replication(self, source_bucket: str,
                                  dr_bucket: str) -> dict:
           """Compare source and DR bucket object counts."""
           source_count = self._count_objects(source_bucket)
           dr_count = self._count_objects(dr_bucket)

           return {
               "source_bucket": source_bucket,
               "dr_bucket": dr_bucket,
               "source_objects": source_count,
               "dr_objects": dr_count,
               "delta": source_count - dr_count,
               "healthy": abs(source_count - dr_count) < 100,
           }

       def run_dr_drill(self) -> dict:
           """Monthly DR drill: attempt to restore from backup."""
           results = {
               "drill_date": datetime.utcnow().isoformat(),
               "components": [],
           }

           # Test 1: Snowflake failover (read-only query on replica)
           try:
               self.sf.cursor().execute(
                   "SELECT COUNT(*) FROM GOLD.FACT_ORDERS"
               )
               results["components"].append({
                   "component": "snowflake_replica",
                   "status": "passed",
                   "test": "Read query on replica",
               })
           except Exception as e:
               results["components"].append({
                   "component": "snowflake_replica",
                   "status": "failed",
                   "error": str(e),
               })

           return results

       def _count_objects(self, bucket: str) -> int:
           paginator = self.s3.get_paginator('list_objects_v2')
           count = 0
           for page in paginator.paginate(Bucket=bucket):
               count += page.get('KeyCount', 0)
           return count
   ```

   ```python
   from dataclasses import dataclass

   @dataclass
   class CostOptimization:
       category: str
       description: str
       current_cost: float
       projected_cost: float
       savings: float
       effort: str  # "low", "medium", "high"
       risk: str    # "low", "medium", "high"
       automated: bool

   class FinOpsOptimizer:
       def __init__(self, snowflake_conn, aws_client):
           self.sf = snowflake_conn
           self.aws = aws_client

       def identify_optimizations(self) -> list[CostOptimization]:
           """Scan for all cost optimization opportunities."""
           optimizations = []

           # 1. Idle warehouses
           idle = self._find_idle_warehouses()
           optimizations.extend(idle)

           # 2. Over-sized warehouses
           oversized = self._find_oversized_warehouses()
           optimizations.extend(oversized)

           # 3. Redundant data copies
           redundant = self._find_redundant_storage()
           optimizations.extend(redundant)

           # 4. Unused tables (no queries in 90 days)
           unused = self._find_unused_tables()
           optimizations.extend(unused)

           return sorted(optimizations, key=lambda o: o.savings, reverse=True)

       def _find_idle_warehouses(self) -> list[CostOptimization]:
           """Warehouses running but not used."""
           result = self.sf.cursor().execute("""
               SELECT warehouse_name, 
                      SUM(credits_used) AS credits,
                      COUNT(DISTINCT DATE(start_time)) AS active_days
               FROM snowflake.account_usage.warehouse_metering_history
               WHERE start_time >= DATEADD(day, -30, CURRENT_TIMESTAMP())
               GROUP BY warehouse_name
               HAVING active_days < 5  -- Used less than 5 days last month
           """).fetchall()

           return [
               CostOptimization(
                   category="idle_warehouse",
                   description=f"Warehouse {r[0]} used only {r[2]} days",
                   current_cost=round(r[1] * 3, 2),
                   projected_cost=0,
                   savings=round(r[1] * 3, 2),
                   effort="low",
                   risk="low",
                   automated=True,
               )
               for r in result
           ]

       def _find_oversized_warehouses(self) -> list[CostOptimization]:
           result = self.sf.cursor().execute("""
               SELECT warehouse_name, warehouse_size,
                      AVG(bytes_scanned) AS avg_bytes,
                      AVG(total_elapsed_time) AS avg_ms,
                      SUM(credits_used_cloud_services) AS credits
               FROM snowflake.account_usage.query_history
               WHERE start_time >= DATEADD(day, -30, CURRENT_TIMESTAMP())
                   AND warehouse_size IN ('X-Large', '2X-Large', '3X-Large')
               GROUP BY warehouse_name, warehouse_size
               HAVING avg_bytes < 5e9  -- Avg scan < 5GB
           """).fetchall()

           return [
               CostOptimization(
                   category="oversized_warehouse",
                   description=(
                       f"Warehouse {r[0]} ({r[1]}) scans avg "
                       f"{r[2] / 1e9:.1f} GB — downsize to Medium"
                   ),
                   current_cost=round(r[4] * 3, 2),
                   projected_cost=round(r[4] * 3 * 0.25, 2),  # ~75% savings
                   savings=round(r[4] * 3 * 0.75, 2),
                   effort="medium",
                   risk="medium",
                   automated=False,
               )
               for r in result
           ]

       def _find_unused_tables(self) -> list[CostOptimization]:
           result = self.sf.cursor().execute("""
               SELECT t.table_schema, t.table_name, t.bytes,
                      t.last_altered
               FROM information_schema.tables t
               LEFT JOIN (
                   SELECT DISTINCT 
                       f.value:objectName::STRING AS table_name
                   FROM snowflake.account_usage.access_history,
                        LATERAL FLATTEN(base_objects_accessed) f
                   WHERE query_start_time >= DATEADD(day, -90, CURRENT_TIMESTAMP())
               ) ah ON ah.table_name LIKE '%' || t.table_name || '%'
               WHERE ah.table_name IS NULL
                   AND t.bytes > 1e8  -- > 100MB
               ORDER BY t.bytes DESC
           """).fetchall()

           return [
               CostOptimization(
                   category="unused_table",
                   description=(
                       f"Table {r[0]}.{r[1]} ({r[2] / 1e9:.1f} GB) "
                       f"not queried in 90 days"
                   ),
                   current_cost=round(r[2] / 1e12 * 23 * 30, 2),  # Storage cost
                   projected_cost=0,
                   savings=round(r[2] / 1e12 * 23 * 30, 2),
                   effort="low",
                   risk="low",
                   automated=True,
               )
               for r in result
           ]
   ```

   ```python
   # finops/roi_tracker.py
   @dataclass
   class DataInitiative:
       name: str
       investment_usd: float
       annual_value_usd: float
       value_type: str  # "revenue", "cost_savings", "risk_reduction"

       def calculate_platform_roi(self) -> dict:

           total_value = sum(
               i.annual_value_usd * i.confidence for i in self.initiatives
           )

           return {
               "total_investment": total_investment,
               "total_value_confidence_adjusted": round(total_value, 2),
               "roi_multiple": (
                   round(total_value / total_investment, 1)
                   if total_investment > 0 else 0
               ),
               "payback_months": (
                   round(total_investment / (total_value / 12), 1)
                   if total_value > 0 else float("inf")
               ),
               "by_type": self._value_by_type(),
           }

       def _value_by_type(self) -> dict:
           by_type = {}
           for i in self.initiatives:
               if i.value_type not in by_type:
                   by_type[i.value_type] = 0
               by_type[i.value_type] += i.annual_value_usd * i.confidence
           return by_type
   ```

   ```python
   class SecurityMonitor:
       """Continuous security monitoring for the data platform."""

       def __init__(self, snowflake_conn):
           self.sf = snowflake_conn

       def detect_anomalous_access(self) -> list[dict]:
           """Detect unusual data access patterns."""
           results = self.sf.cursor().execute("""
               WITH user_baselines AS (
                   SELECT
                       user_name,
                       AVG(bytes_scanned) AS avg_bytes,
                       STDDEV(bytes_scanned) AS std_bytes,
                       COUNT(DISTINCT query_type) AS usual_query_types
                   FROM snowflake.account_usage.query_history
                   WHERE start_time BETWEEN 
                       DATEADD(day, -30, CURRENT_TIMESTAMP()) AND
                       DATEADD(day, -1, CURRENT_TIMESTAMP())
                   GROUP BY user_name
               )
               SELECT
                   qh.user_name,
                   qh.query_id,
                   qh.query_type,
                   qh.bytes_scanned,
                   ub.avg_bytes,
                   (qh.bytes_scanned - ub.avg_bytes) / NULLIF(ub.std_bytes, 0) 
                       AS z_score,
                   qh.query_text
               FROM snowflake.account_usage.query_history qh
               JOIN user_baselines ub ON qh.user_name = ub.user_name
               WHERE qh.start_time >= DATEADD(day, -1, CURRENT_TIMESTAMP())
                   AND (qh.bytes_scanned - ub.avg_bytes) / 
                       NULLIF(ub.std_bytes, 0) > 3
               ORDER BY z_score DESC
           """).fetchall()

           return [
               {
                   "user": r[0],
                   "query_id": r[1],
                   "anomaly": f"Scanned {r[3] / 1e9:.1f} GB (usual: {r[4] / 1e9:.1f} GB)",
                   "z_score": round(r[5], 2),
                   "risk": "high" if r[5] > 5 else "medium",
               }
               for r in results
   ```

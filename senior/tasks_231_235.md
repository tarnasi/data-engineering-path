3. **Build the metric catalog and validation**:
   ```python
   # semantic_layer/metric_catalog.py
   import yaml
   from pathlib import Path

   class MetricCatalog:
       def __init__(self, metrics_dir: str):
           self.metrics = {}
           for f in Path(metrics_dir).glob("**/*.yaml"):
               with open(f) as fh:
                   for doc in yaml.safe_load_all(fh):
                       if doc and "metric" in doc:
                           m = doc["metric"]
                           self.metrics[m["name"]] = m

       def get_metric_sql(self, name: str, dimensions: list[str] = None,
                          filters: dict = None, time_grain: str = "day") -> str:
           """Generate SQL for a metric query."""
           metric = self.metrics.get(name)
           if not metric:
               raise ValueError(f"Unknown metric: {name}")

           # Build SELECT
           agg = f"{metric['type'].upper()}({metric['sql']})"
           time_col = metric["timestamp"]

           select_cols = [f"DATE_TRUNC('{time_grain}', {time_col}) AS period"]
           group_cols = ["period"]

           if dimensions:
               for dim in dimensions:
                   if dim not in metric.get("dimensions", []):
                       raise ValueError(
                           f"Dimension '{dim}' not available for metric '{name}'. "
                           f"Available: {metric['dimensions']}"
                       )
                   select_cols.append(dim)
                   group_cols.append(dim)

           select_cols.append(f"{agg} AS {name}")

           # Build WHERE
           where_clauses = []
           for f in metric.get("filters", []):
               val = f["value"]
               if isinstance(val, list):
                   quoted = ", ".join(f"'{v}'" for v in val)
                   where_clauses.append(f"{f['field']} {f['operator']} ({quoted})")
               elif isinstance(val, bool):
                   where_clauses.append(f"{f['field']} = {str(val).upper()}")
               else:
                   where_clauses.append(f"{f['field']} {f['operator']} '{val}'")

           if filters:
               for key, val in filters.items():
                   where_clauses.append(f"{key} = '{val}'")

           sql = f"SELECT {', '.join(select_cols)}\n"
           sql += f"FROM {metric['model']}\n"
           if where_clauses:
               sql += f"WHERE {' AND '.join(where_clauses)}\n"
           sql += f"GROUP BY {', '.join(group_cols)}\n"
           sql += "ORDER BY period"

           return sql

       def validate_all(self) -> list[dict]:
           """Validate metric definitions for completeness."""
           issues = []
           for name, metric in self.metrics.items():
               if not metric.get("description"):
                   issues.append({"metric": name, "issue": "Missing description"})
               if not metric.get("owner"):
                   issues.append({"metric": name, "issue": "Missing owner"})
               if not metric.get("dimensions"):
                   issues.append({"metric": name, "issue": "No dimensions defined"})
               if metric.get("certified") and not metric.get("certified_by"):
                   issues.append({"metric": name, "issue": "Certified but no certifier"})
           return issues
   ```

   ```python
   # ci/smart_ci.py
   """Only build and test models affected by the PR."""
   import subprocess
   import json

   class SmartCI:
       def __init__(self, changed_files: list[str]):
           self.changed_files = changed_files

       def determine_affected_projects(self) -> list[str]:
           """Map changed files to dbt projects."""
           projects = set()
           for f in self.changed_files:
               if f.startswith("dbt-platform/"):
                   projects.add("dbt-platform")
               elif f.startswith("dbt-commerce/"):
                   projects.add("dbt-commerce")
               elif f.startswith("dbt-finance/"):
                   projects.add("dbt-finance")
               elif f.startswith("dbt-marketing/"):
                   projects.add("dbt-marketing")
               elif f.startswith("dbt-ml-features/"):
                   projects.add("dbt-ml-features")
               elif f.startswith("dbt-packages/"):
                   projects.update(["dbt-platform", "dbt-commerce",
                                    "dbt-finance", "dbt-marketing",
                                    "dbt-ml-features"])
           return list(projects)

       def get_modified_models(self, project: str) -> list[str]:
           """Get models modified in the PR for a specific project."""
           models = []
           for f in self.changed_files:
               if f.startswith(f"{project}/models/") and f.endswith(".sql"):
                   model_name = f.split("/")[-1].replace(".sql", "")
                   models.append(model_name)
           return models

       def run_slim_ci(self, project: str, models: list[str]) -> dict:
           """Run dbt for only modified models + downstream."""
           select = " ".join(f"{m}+" for m in models)  # model + downstream
           result = subprocess.run(
               ["dbt", "build", "--select", select,
                "--project-dir", project,
                "--target", "ci"],
               capture_output=True, text=True,
           )
           return {
               "project": project,
               "models_selected": select,
               "exit_code": result.returncode,
               "stdout": result.stdout,
               "stderr": result.stderr,
           }

       def run_contract_check(self, project: str) -> dict:
           """Verify model contracts aren't broken."""
           result = subprocess.run(
               ["dbt", "parse", "--project-dir", project],
               capture_output=True, text=True,
           )
           # Check for contract violations in parse output
           violations = [
               line for line in result.stderr.split("\n")
               if "contract" in line.lower()
           ]
           return {
               "project": project,
               "contract_violations": violations,
               "status": "pass" if not violations else "fail",
           }
   ```

   ```python
   # platform_product/nps_survey.py
   from dataclasses import dataclass

   @dataclass
   class NPSSurveyResult:
       respondent_team: str
       respondent_role: str
       nps_score: int          # 0-10
       satisfaction_areas: dict  # "reliability": 8, "docs": 5, etc.
       biggest_pain: str
       feature_request: str
       verbatim_feedback: str

   class NPSAnalyzer:
       def __init__(self, responses: list[NPSSurveyResult]):
           self.responses = responses

       def calculate_nps(self) -> dict:
           promoters = sum(1 for r in self.responses if r.nps_score >= 9)
           passives = sum(1 for r in self.responses if 7 <= r.nps_score <= 8)
           detractors = sum(1 for r in self.responses if r.nps_score <= 6)
           total = len(self.responses)

           nps = ((promoters - detractors) / total) * 100 if total > 0 else 0

           return {
               "nps_score": round(nps),
               "promoters": promoters,
               "passives": passives,
               "detractors": detractors,
               "total_responses": total,
               "response_rate": f"{total}/32",  # 32 platform users
           }

       def pain_point_ranking(self) -> list[tuple[str, int]]:
           """Rank pain points by frequency."""
           pains = {}
           for r in self.responses:
               pain = r.biggest_pain.lower().strip()
               pains[pain] = pains.get(pain, 0) + 1
           return sorted(pains.items(), key=lambda x: x[1], reverse=True)

       def feature_request_ranking(self) -> list[tuple[str, int]]:
           requests = {}
           for r in self.responses:
               req = r.feature_request.lower().strip()
               requests[req] = requests.get(req, 0) + 1
           return sorted(requests.items(), key=lambda x: x[1], reverse=True)

       def satisfaction_heatmap(self) -> dict:
           """Average satisfaction by area and team."""
           areas = {}
           for r in self.responses:
               for area, score in r.satisfaction_areas.items():
                   if area not in areas:
                       areas[area] = []
                   areas[area].append(score)
           return {
               area: round(sum(scores) / len(scores), 1)
               for area, scores in areas.items()
           }
   ```

## Task 234: Graph-Based Data Lineage — Impact Analysis at Scale

   ```python
   # lineage/graph_model.py
   from dataclasses import dataclass, field
   from enum import Enum

   class NodeType(Enum):
       SOURCE = "source"           # PostgreSQL table, API, file
       KAFKA_TOPIC = "kafka_topic"
       DBT_MODEL = "dbt_model"
       SNOWFLAKE_TABLE = "snowflake_table"
       DASHBOARD = "dashboard"
       ML_MODEL = "ml_model"
       API_ENDPOINT = "api_endpoint"
       COLUMN = "column"           # Column within a table/model

   class EdgeType(Enum):
       DERIVES_FROM = "derives_from"     # Column A computed from Column B
       LOADS_INTO = "loads_into"         # CDC loads source into raw
       TRANSFORMS = "transforms"         # dbt model transforms
       MATERIALIZES = "materializes"     # dbt model → Snowflake table
       READS_FROM = "reads_from"         # Dashboard reads from table
       FEATURES_FROM = "features_from"   # ML model uses feature

   @dataclass
   class LineageNode:
       id: str                  # Unique: "dbt_model:stg_orders"
       name: str
       node_type: NodeType
       project: str             # dbt project or system name
       owner: str
       metadata: dict = field(default_factory=dict)

   @dataclass
   class LineageEdge:
       source_id: str
       target_id: str
       edge_type: EdgeType
       transformation: str = ""  # SQL or description of how data flows
   ```

2. **Build the lineage graph**:
   ```python
   # lineage/graph.py
   import networkx as nx
   from collections import defaultdict

   class LineageGraph:
       def __init__(self):
           self.graph = nx.DiGraph()

       def add_node(self, node: LineageNode):
           self.graph.add_node(
               node.id,
               name=node.name,
               node_type=node.node_type.value,
               project=node.project,
               owner=node.owner,
               **node.metadata,
           )

       def add_edge(self, edge: LineageEdge):
           self.graph.add_edge(
               edge.source_id,
               edge.target_id,
               edge_type=edge.edge_type.value,
               transformation=edge.transformation,
           )

       def impact_analysis(self, node_id: str) -> dict:
           """What breaks if this node changes?"""
           downstream = nx.descendants(self.graph, node_id)
           affected = []
           for d in downstream:
               data = self.graph.nodes[d]
               affected.append({
                   "id": d,
                   "name": data.get("name"),
                   "type": data.get("node_type"),
                   "owner": data.get("owner"),
                   "distance": nx.shortest_path_length(
                       self.graph, node_id, d
                   ),
               })

           # Group by owner for notification
           by_owner = defaultdict(list)
           for a in affected:
               by_owner[a["owner"]].append(a)

           return {
               "node": node_id,
               "total_downstream": len(downstream),
               "affected_nodes": sorted(affected, key=lambda x: x["distance"]),
               "teams_to_notify": dict(by_owner),
               "critical_paths": self._find_critical_paths(node_id, downstream),
           }

       def _find_critical_paths(self, source: str, downstream: set) -> list:
           """Find paths to dashboards and APIs (end-user-facing)."""
           critical = []
           for d in downstream:
               node_type = self.graph.nodes[d].get("node_type")
               if node_type in ("dashboard", "api_endpoint", "ml_model"):
                   path = nx.shortest_path(self.graph, source, d)
                   critical.append({
                       "target": d,
                       "type": node_type,
                       "path": path,
                       "hops": len(path) - 1,
                   })
           return critical

       def upstream_trace(self, node_id: str) -> dict:
           """Where does this data come from?"""
           upstream = nx.ancestors(self.graph, node_id)
           sources = [
               u for u in upstream
               if self.graph.nodes[u].get("node_type") == "source"
           ]
           return {
               "node": node_id,
               "total_upstream": len(upstream),
               "original_sources": sources,
               "full_lineage": list(
                   nx.all_simple_paths(self.graph, s, node_id)
                   for s in sources
               ),
           }

       def column_impact(self, table_id: str, column_name: str) -> dict:
           """Column-level impact analysis."""
           column_id = f"{table_id}.{column_name}"
           if column_id not in self.graph:
               return {"error": f"Column {column_id} not found in lineage"}
           return self.impact_analysis(column_id)
   ```

3. **Parse dbt artifacts for lineage**:
   ```python
   # lineage/dbt_parser.py
   import json
   from pathlib import Path

   class DbtLineageParser:
       """Extract lineage from dbt manifest.json."""

       def __init__(self, manifest_path: str):
           with open(manifest_path) as f:
               self.manifest = json.load(f)

       def extract_nodes(self) -> list[LineageNode]:
           nodes = []
           for node_id, node_data in self.manifest.get("nodes", {}).items():
               if node_data["resource_type"] in ("model", "source", "seed"):
                   nodes.append(LineageNode(
                       id=node_id,
                       name=node_data["name"],
                       node_type=NodeType.DBT_MODEL
                       if node_data["resource_type"] == "model"
                       else NodeType.SOURCE,
                       project=node_data.get("package_name", "unknown"),
                       owner=node_data.get("config", {}).get("meta", {}).get(
                           "owner", "unknown"
                       ),
                       metadata={
                           "schema": node_data.get("schema"),
                           "database": node_data.get("database"),
                           "materialized": node_data.get("config", {}).get(
                               "materialized"
                           ),
                       },
                   ))

                   # Add column-level nodes
                   for col_name, col_data in node_data.get("columns", {}).items():
                       nodes.append(LineageNode(
                           id=f"{node_id}.{col_name}",
                           name=col_name,
                           node_type=NodeType.COLUMN,
                           project=node_data.get("package_name", "unknown"),
                           owner=node_data.get("config", {}).get("meta", {}).get(
                               "owner", "unknown"
                           ),
                       ))
           return nodes

       def extract_edges(self) -> list[LineageEdge]:
           edges = []
           for node_id, node_data in self.manifest.get("nodes", {}).items():
               for dep in node_data.get("depends_on", {}).get("nodes", []):
                   edges.append(LineageEdge(
                       source_id=dep,
                       target_id=node_id,
                       edge_type=EdgeType.TRANSFORMS,
                   ))
           return edges
   ```

   ```python
   # lineage/ci_impact.py
   class CIImpactCheck:
       def __init__(self, lineage_graph: LineageGraph):
           self.graph = lineage_graph

       def check_pr(self, changed_models: list[str]) -> dict:
           """Run impact analysis on PR changes."""
           all_affected = set()
           notifications = {}

           for model in changed_models:
               impact = self.graph.impact_analysis(model)
               all_affected.update(
                   a["id"] for a in impact["affected_nodes"]
               )

               for team, nodes in impact["teams_to_notify"].items():
                   if team not in notifications:
                       notifications[team] = []
                   notifications[team].extend(nodes)

           critical = any(
               self.graph.graph.nodes[n].get("node_type") in
               ("dashboard", "api_endpoint")
               for n in all_affected
           )

           return {
               "models_changed": changed_models,
               "total_downstream_affected": len(all_affected),
               "teams_to_notify": list(notifications.keys()),
               "has_critical_impact": critical,
               "recommendation": (
                   "BLOCK: Critical impact — requires review from affected teams"
                   if critical
                   else "WARN: Downstream impact — notify affected teams"
               ),
           }
   ```

   ```sql
   -- Top 20 most expensive queries (last 30 days)
   SELECT
       query_id,

       warehouse_size,

       warehouse_name,
       warehouse_size,

       ROUND(SUM(credits_used_cloud_services), 2) AS total_credits,
       ROUND(AVG(total_elapsed_time) / 1000, 2) AS avg_seconds,
       ROUND(AVG(bytes_scanned) / 1e9, 2) AS avg_gb_scanned,
       ROUND(SUM(credits_used_cloud_services) * 3, 2) AS est_cost_usd
   FROM snowflake.account_usage.query_history

   SELECT
       warehouse_name,
       ROUND(SUM(credits_used), 2) AS credits_used,

       ROUND(SUM(credits_used) / COUNT(DISTINCT DATE(start_time)), 2)
   ```

2. **Build the query optimizer**:
   ```python
   # performance/query_optimizer.py
   from dataclasses import dataclass

   @dataclass
   class OptimizationRecommendation:
       query_id: str
       category: str
       severity: str  # "high", "medium", "low"
       description: str
       estimated_savings_pct: float
       action: str

   class QueryOptimizer:
       def __init__(self, snowflake_conn):
           self.conn = snowflake_conn

       def analyze_expensive_queries(self, days: int = 30,
                                     top_n: int = 50) -> list:
           """Find and analyze the most expensive queries."""
           queries = self.conn.cursor().execute(f"""
               SELECT query_id, query_text, warehouse_name, warehouse_size,
                      credits_used_cloud_services, bytes_scanned,
                      partitions_scanned, partitions_total,
                      bytes_spilled_to_local_storage,
                      bytes_spilled_to_remote_storage,
                      total_elapsed_time, user_name
               FROM snowflake.account_usage.query_history
               WHERE start_time >= DATEADD(day, -{days}, CURRENT_TIMESTAMP())
                   AND credits_used_cloud_services > 0.01
               ORDER BY credits_used_cloud_services DESC
               LIMIT {top_n}
           """).fetchall()

           recommendations = []
           for q in queries:
               recs = self._analyze_query(q)
               recommendations.extend(recs)

           return sorted(recommendations,
                        key=lambda r: r.estimated_savings_pct,
                        reverse=True)

       def _analyze_query(self, query_row) -> list[OptimizationRecommendation]:
           recs = []
           query_id = query_row[0]
           query_text = query_row[1]
           warehouse_size = query_row[3]
           bytes_scanned = query_row[5]
           partitions_scanned = query_row[6]
           partitions_total = query_row[7]
           bytes_spilled_local = query_row[8]
           bytes_spilled_remote = query_row[9]

           # Check 1: Full table scan
           if partitions_total and partitions_total > 0:
               scan_pct = (partitions_scanned / partitions_total) * 100
               if scan_pct > 80:
                   recs.append(OptimizationRecommendation(
                       query_id=query_id,
                       category="pruning",
                       severity="high",
                       description=(
                           f"Full table scan: {scan_pct:.0f}% of partitions scanned. "
                           f"Add clustering key or filter on partition column."
                       ),
                       estimated_savings_pct=50,
                       action="Add CLUSTER BY on commonly filtered columns",
                   ))

           # Check 2: Spillage (query too large for warehouse)
           if bytes_spilled_remote and bytes_spilled_remote > 0:
               recs.append(OptimizationRecommendation(
                   query_id=query_id,
                   category="spillage",
                   severity="high",
                   description=(
                       f"Query spilled to remote storage "
                       f"({bytes_spilled_remote / 1e9:.1f} GB). "
                       f"Needs larger warehouse or query rewrite."
                   ),
                   estimated_savings_pct=30,
                   action="Rewrite to reduce memory usage or use larger warehouse temporarily",
               ))

           # Check 3: SELECT *
           if query_text and "SELECT *" in query_text.upper():
               recs.append(OptimizationRecommendation(
                   query_id=query_id,
                   category="select_star",
                   severity="medium",
                   description="SELECT * on large table; specify needed columns",
                   estimated_savings_pct=40,
                   action="Replace SELECT * with specific column list",
               ))

           # Check 4: Over-sized warehouse
           size_order = ["X-Small", "Small", "Medium", "Large",
                         "X-Large", "2X-Large", "3X-Large"]
           if (warehouse_size in size_order[3:]  # Large or bigger
               and bytes_scanned and bytes_scanned < 1e9):  # < 1GB scanned
               recs.append(OptimizationRecommendation(
                   query_id=query_id,
                   category="warehouse_sizing",
                   severity="medium",
                   description=(
                       f"Using {warehouse_size} warehouse for "
                       f"{bytes_scanned / 1e9:.1f} GB scan. Downsize."
                   ),
                   estimated_savings_pct=50,
                   action=f"Move to Small or Medium warehouse",
               ))

           return recs

       def generate_report(self, recommendations: list) -> str:
           report = "# Snowflake Cost Optimization Report\n\n"
           by_category = {}
           for r in recommendations:
               by_category.setdefault(r.category, []).append(r)

           for cat, recs in by_category.items():
               savings = sum(r.estimated_savings_pct for r in recs) / len(recs)
               report += f"## {cat.replace('_', ' ').title()} ({len(recs)} findings)\n"
               report += f"Avg estimated savings: {savings:.0f}%\n\n"
               for r in recs[:5]:
                   report += f"- [{r.severity.upper()}] {r.description}\n"
                   report += f"  Action: {r.action}\n\n"

           return report
   ```

   ```sql
   -- Resource monitors to prevent runaway costs
   ```

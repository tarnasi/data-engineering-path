## Task 221: Data Contracts at Scale — Enforcing Boundaries Between Producers and Consumers

### Business Context
NovaMart now has 8 domain teams producing data and 15+ consumers (dashboards, ML models, APIs, partner feeds). Last month, the Logistics team renamed a Kafka topic field from `delivery_eta` to `estimated_delivery_time`. Result: the ML team's delivery prediction model broke, the Customer Service dashboard showed blank ETAs for 6 hours, and the partner API returned nulls to 3 external partners. Nobody was notified in advance. Marcus is furious: "We can't have upstream teams breaking downstream consumers. Fix this."

### Objective
Implement a company-wide data contracts system that enforces schema agreements between data producers and consumers, with automated validation and breaking-change governance.

### Technical Requirements
- **Tools**: Python, JSON Schema / Protobuf, GitHub Actions, Kafka Schema Registry, YAML
- **Skills**: Data contracts, schema governance, API design principles, organizational change management

2. **Build the contract validation engine**:
   ```python
   # contracts/validator.py
   import yaml
   import json
   from dataclasses import dataclass
   from enum import Enum
   from pathlib import Path

   class ChangeType(Enum):
       COMPATIBLE = "compatible"
       BREAKING = "breaking"
       DEPRECATION = "deprecation"

   @dataclass
   class SchemaChange:
       field: str
       change_type: ChangeType
       description: str
       affected_consumers: list[str]

   class ContractValidator:
       def __init__(self, contracts_dir: str):
           self.contracts_dir = Path(contracts_dir)

       def load_contract(self, path: str) -> dict:
           with open(path) as f:
               return yaml.safe_load(f)

       def diff_contracts(self, old: dict, new: dict) -> list[SchemaChange]:
           """Compare two contract versions and identify changes."""
           changes = []
           old_fields = {f["name"]: f for f in old["schema"]["fields"]}
           new_fields = {f["name"]: f for f in new["schema"]["fields"]}

           # Removed fields = breaking
           for name in old_fields:
               if name not in new_fields:
                   affected = self._find_consumers_using_field(old, name)
                   changes.append(SchemaChange(
                       field=name,
                       change_type=ChangeType.BREAKING,
                       description=f"Field '{name}' removed",
                       affected_consumers=affected,
                   ))

           # Type changes = breaking
           for name in old_fields:
               if name in new_fields:
                   if old_fields[name]["type"] != new_fields[name]["type"]:
                       affected = self._find_consumers_using_field(old, name)
                       changes.append(SchemaChange(
                           field=name,
                           change_type=ChangeType.BREAKING,
                           description=(
                               f"Field '{name}' type changed: "
                               f"{old_fields[name]['type']} → "
                               f"{new_fields[name]['type']}"
                           ),
                           affected_consumers=affected,
                       ))

           # New required fields = breaking (old data won't have them)
           for name in new_fields:
               if name not in old_fields and new_fields[name].get("required"):
                   changes.append(SchemaChange(
                       field=name,
                       change_type=ChangeType.BREAKING,
                       description=f"New required field '{name}' added",
                       affected_consumers=[],
                   ))

           # New optional fields = compatible
           for name in new_fields:
               if name not in old_fields and not new_fields[name].get("required"):
                   changes.append(SchemaChange(
                       field=name,
                       change_type=ChangeType.COMPATIBLE,
                       description=f"New optional field '{name}' added",
                       affected_consumers=[],
                   ))

           return changes

       def _find_consumers_using_field(self, contract: dict,
                                        field: str) -> list[str]:
           return [
               c["team"]
               for c in contract.get("consumers", [])
               if field in c.get("fields_used", [])
           ]

       def validate_change(self, old_path: str, new_path: str) -> dict:
           """Full validation for CI/CD pipeline."""
           old = self.load_contract(old_path)
           new = self.load_contract(new_path)
           changes = self.diff_contracts(old, new)

           breaking = [c for c in changes if c.change_type == ChangeType.BREAKING]
           result = {
               "contract": new["metadata"]["name"],
               "old_version": old["metadata"]["version"],
               "new_version": new["metadata"]["version"],
               "total_changes": len(changes),
               "breaking_changes": len(breaking),
               "requires_approval": len(breaking) > 0,
               "affected_consumers": list(set(
                   consumer
                   for change in breaking
                   for consumer in change.affected_consumers
               )),
               "changes": [
                   {
                       "field": c.field,
                       "type": c.change_type.value,
                       "description": c.description,
                       "affected": c.affected_consumers,
                   }
                   for c in changes
               ],
           }
           return result
   ```

3. **Integrate into CI/CD**:
   ```yaml
   # .github/workflows/contract-validation.yaml
   name: Data Contract Validation
   on:
     pull_request:
       paths: ['contracts/**']

   jobs:
     validate:
       runs-on: ubuntu-latest
       steps:
         - uses: actions/checkout@v4
           with:
             fetch-depth: 0

         - name: Find changed contracts
           id: changed
           run: |
             CHANGED=$(git diff --name-only origin/main -- 'contracts/**/*.yaml')
             echo "files=$CHANGED" >> $GITHUB_OUTPUT

         - name: Validate contracts
           run: |
             python -m contracts.ci_validate \
               --changed-files "${{ steps.changed.outputs.files }}"

         - name: Block breaking changes without approval
           if: steps.validate.outputs.has_breaking == 'true'
           run: |
             echo "::error::Breaking changes detected. Requires approval from affected consumers."
             echo "Affected teams: ${{ steps.validate.outputs.affected_teams }}"
             exit 1

         - name: Notify consumers
           if: steps.validate.outputs.has_changes == 'true'
           run: |
             python -m contracts.notify_consumers \
               --changes "${{ steps.validate.outputs.changes_json }}"
   ```

4. **Build the consumer notification system**:
   ```python
   # contracts/notifier.py
   import json
   import requests

   class ContractNotifier:
       def __init__(self, slack_webhook: str):
           self.webhook = slack_webhook

       def notify_breaking_change(self, validation_result: dict):
           """Notify affected consumers of breaking changes."""
           contract = validation_result["contract"]
           version = validation_result["new_version"]
           affected = validation_result["affected_consumers"]
           changes = [
               c for c in validation_result["changes"]
               if c["type"] == "breaking"
           ]

           blocks = [
               {
                   "type": "header",
                   "text": {
                       "type": "plain_text",
                       "text": f"⚠️ Breaking Change: {contract} → v{version}"
                   }
               },
               {
                   "type": "section",
                   "text": {
                       "type": "mrkdwn",
                       "text": (
                           f"*Contract*: `{contract}`\n"
                           f"*New Version*: `{version}`\n"
                           f"*Breaking Changes*: {len(changes)}\n"
                           f"*Affected Teams*: {', '.join(affected)}"
                       )
                   }
               },
           ]

           for change in changes:
               blocks.append({
                   "type": "section",
                   "text": {
                       "type": "mrkdwn",
                       "text": (
                           f"• *{change['field']}*: {change['description']}\n"
                           f"  Consumers affected: {', '.join(change['affected'])}"
                       )
                   }
               })

           requests.post(self.webhook, json={"blocks": blocks}, timeout=10)
   ```

2. **Build the growth projection model**:
   ```python
   # capacity/projections.py
   import pandas as pd
   import numpy as np
   from dataclasses import dataclass

   @dataclass
   class GrowthScenario:
       name: str
       order_multiplier: float
       employee_multiplier: float
       event_multiplier: float
       storage_growth_rate: float  # Monthly compound rate

   SCENARIOS = {
       "conservative": GrowthScenario(
           name="Conservative (organic growth)",
           order_multiplier=1.8,
           employee_multiplier=1.5,
           event_multiplier=2.0,
           storage_growth_rate=0.08,
       ),
       "expected": GrowthScenario(
           name="Expected (LATAM expansion + hiring)",
           order_multiplier=3.0,
           employee_multiplier=2.0,
           event_multiplier=3.5,
           storage_growth_rate=0.12,
       ),
       "aggressive": GrowthScenario(
           name="Aggressive (post-IPO surge)",
           order_multiplier=5.0,
           employee_multiplier=3.0,
           event_multiplier=6.0,
           storage_growth_rate=0.18,
       ),
   }

   class CapacityModel:
       def __init__(self, baseline: PlatformBaseline):
           self.baseline = baseline

       def project(self, scenario: GrowthScenario,
                   months: int = 18) -> pd.DataFrame:
           """Project resource needs over time."""
           rows = []
           for month in range(months + 1):
               progress = month / months  # 0.0 to 1.0
               # S-curve growth (growth accelerates then plateaus)
               growth_factor = 1 / (1 + np.exp(-10 * (progress - 0.5)))

               order_mult = 1 + (scenario.order_multiplier - 1) * growth_factor
               event_mult = 1 + (scenario.event_multiplier - 1) * growth_factor
               emp_mult = 1 + (scenario.employee_multiplier - 1) * growth_factor

               storage_tb = (
                   self.baseline.snowflake_storage_tb
                   * (1 + scenario.storage_growth_rate) ** month
               )

               # Snowflake credits scale with queries AND data volume
               sf_credits = (
                   self.baseline.snowflake_credits_per_day
                   * (order_mult * 0.6 + emp_mult * 0.4)  # Weighted
               )

               # Flink scales with event throughput
               flink_tms = max(
                   self.baseline.flink_taskmanagers,
                   int(np.ceil(
                       self.baseline.flink_taskmanagers
                       * event_mult
                       * (self.baseline.flink_cpu_utilization_pct / 70)
                   ))
               )

               rows.append({
                   "month": month,
                   "orders_per_day": int(self.baseline.orders_per_day * order_mult),
                   "events_per_sec_peak": int(
                       self.baseline.events_per_second_peak * event_mult
                   ),
                   "snowflake_credits_day": round(sf_credits, 1),
                   "snowflake_storage_tb": round(storage_tb, 1),
                   "s3_storage_tb": round(
                       self.baseline.s3_storage_tb
                       * (1 + scenario.storage_growth_rate) ** month,
                       1
                   ),
                   "flink_taskmanagers": flink_tms,
                   "dashboard_users_peak": int(
                       self.baseline.dashboard_concurrent_users_peak * emp_mult
                   ),
                   "monthly_cost_est": round(
                       sf_credits * 30 * 3.0  # $3/credit
                       + storage_tb * 23  # $23/TB/month Snowflake
                       + flink_tms * 800  # ~$800/TM/month
                       + self.baseline.s3_storage_tb
                       * (1 + scenario.storage_growth_rate) ** month * 23
                       + 15_000,  # Base (Kafka, Airflow, monitoring)
                       -2  # Round to nearest 100
                   ),
               })

           return pd.DataFrame(rows)

       def find_bottlenecks(self, projections: pd.DataFrame) -> list[dict]:
           """Identify components that will hit limits first."""
           bottlenecks = []
           limits = {
               "snowflake_credits_day": {
                   "limit": 500,
                   "name": "Snowflake compute",
                   "mitigation": "Upgrade warehouse size or add clustering",
               },
               "flink_taskmanagers": {
                   "limit": 32,
                   "name": "Flink cluster capacity",
                   "mitigation": "Add Kubernetes node pool or increase TM memory",
               },
               "events_per_sec_peak": {
                   "limit": 10_000,
                   "name": "Kafka throughput",
                   "mitigation": "Add partitions and brokers",
               },
               "dashboard_users_peak": {
                   "limit": 300,
                   "name": "BI tool concurrency",
                   "mitigation": "Add Looker capacity or implement caching layer",
               },
           }
           for col, info in limits.items():
               breach_month = projections[projections[col] > info["limit"]]
               if len(breach_month) > 0:
                   bottlenecks.append({
                       "component": info["name"],
                       "current": projections.iloc[0][col],
                       "limit": info["limit"],
                       "breaches_at_month": int(breach_month.iloc[0]["month"]),
                       "mitigation": info["mitigation"],
                   })
           return sorted(bottlenecks, key=lambda b: b["breaches_at_month"])
   ```

3. **Generate the capacity report**:
   ```python
   # capacity/report.py
   class CapacityReport:
       def generate(self, baseline, model, scenarios) -> str:
           report = "# Data Platform Capacity Plan — 18 Month Projection\n\n"
           report += f"## Baseline (measured {baseline.measured_at.date()})\n\n"
           report += f"- Orders/day: {baseline.orders_per_day:,}\n"
           report += f"- Events/sec (peak): {baseline.events_per_second_peak:,}\n"
           report += f"- Snowflake: {baseline.snowflake_credits_per_day} credits/day\n"
           report += f"- Storage: {baseline.snowflake_storage_tb} TB (SF) + {baseline.s3_storage_tb} TB (S3)\n"
           report += f"- Monthly cost: ~$127,000\n\n"

           for name, scenario in scenarios.items():
               projections = model.project(scenario)
               bottlenecks = model.find_bottlenecks(projections)
               final = projections.iloc[-1]

               report += f"## Scenario: {scenario.name}\n\n"
               report += f"| Metric | Now | Month 18 |\n"
               report += f"|--------|-----|----------|\n"
               report += f"| Orders/day | {baseline.orders_per_day:,} | {int(final['orders_per_day']):,} |\n"
               report += f"| Events/sec (peak) | {baseline.events_per_second_peak:,} | {int(final['events_per_sec_peak']):,} |\n"
               report += f"| SF credits/day | {baseline.snowflake_credits_per_day} | {final['snowflake_credits_day']} |\n"
               report += f"| Storage (SF) | {baseline.snowflake_storage_tb} TB | {final['snowflake_storage_tb']} TB |\n"
               report += f"| Monthly cost | $127K | ${int(final['monthly_cost_est']/1000)}K |\n\n"

               if bottlenecks:
                   report += "### Bottlenecks\n\n"
                   for b in bottlenecks:
                       report += (f"- **{b['component']}**: hits limit at month "
                                 f"{b['breaches_at_month']} "
                                 f"(current: {b['current']}, limit: {b['limit']}). "
                                 f"Mitigation: {b['mitigation']}\n")
                   report += "\n"

           return report
   ```

## Task 223: Organizational Influence — RFC Process and Technical Strategy

### Business Context
NovaMart now has 15 engineers writing data code across 8 domain teams. Without coordination, they're making incompatible decisions: the Commerce team chose Protobuf for events while Logistics uses Avro. Marketing built a custom CDC connector instead of using the platform's. The ML team deployed their own Airflow instance. Marcus says: "You're the Senior DE. It's your job to set technical direction that everyone follows — not by telling them what to do, but by making the right thing the easy thing."

### Objective
Establish an RFC (Request for Comments) process that scales technical decision-making across 8 domain teams while preserving their autonomy.

### Technical Requirements
- **Tools**: Markdown, GitHub, YAML
- **Skills**: Technical leadership, RFC process design, organizational influence, consensus building

### Step-by-Step Guidance

1. **Design the RFC process**:
   ```markdown
   ## NovaMart Data Engineering RFC Process

   ### When to Write an RFC
   An RFC is required when:
   - Introducing a new technology to the data stack
   - Changing a shared standard (serialization format, naming conventions)
   - Making an architectural decision affecting 2+ teams
   - Any change with estimated cost > $5K/month
   - Any change to data contracts or SLAs

   An RFC is NOT required for:
   - Internal team decisions (how a team builds their own models)
   - Bug fixes and operational work
   - Changes within an existing approved architecture

   ### RFC Lifecycle
   ```mermaid
   stateDiagram-v2
       [*] --> Draft: Author creates RFC
       Draft --> Review: Author requests review
       Review --> Approved: Consensus reached
       Review --> Revised: Feedback requires changes
       Revised --> Review: Author addresses feedback
       Review --> Rejected: Consensus against
       Approved --> Implemented: Work begins
       Implemented --> [*]: Complete
       Rejected --> [*]: Archived with rationale
   ```

   ### Roles
   | Role | Responsibility |
   |------|---------------|
   | Author | Writes RFC, addresses feedback, drives to consensus |
   | Reviewers | Provide technical feedback within 5 business days |
   | Approver | Senior DE or Staff DE signs off on final decision |
   | Stakeholders | Affected teams — must be notified and can raise concerns |

   ### Timeline
   - Draft → Review: Author decides (typically 1-3 days)
   - Review period: 5 business days minimum
   - Review → Decision: Within 2 days of review close
   - Total: ~2 weeks for non-urgent, 3 days for urgent (with justification)
   ```

2. **Create the RFC template**:

   | **Decision Date** | YYYY-MM-DD |

   One paragraph describing the proposal.

   ## Motivation

   |------------|------|------|---------|
   | Option A | ... | ... | ... |
   | Option B | ... | ... | ... |

   ## Migration Plan

   What's the rollback plan if it doesn't work?

3. **Write a real RFC — standardizing on Protobuf**:
   ```markdown
   # RFC-007: Standardize Event Serialization on Protocol Buffers

   | Field | Value |
   |-------|-------|
   | **Status** | review |
   | **Author** | [You], Data Platform |
   | **Reviewers** | Priya (Staff DE), Commerce Lead, Logistics Lead, ML Lead |
   | **Created** | 2026-04-10 |

   ## Summary
   Adopt Protocol Buffers (protobuf) as the standard serialization 
   format for all Kafka events and data contracts. Currently, 3 teams 
   use Avro, 2 use Protobuf, and 1 uses JSON. This fragmentation causes 
   tooling overhead, consumer confusion, and Schema Registry conflicts.

   ## Motivation

   ```python
   # rfc/tracker.py
   import os
   import yaml
   from pathlib import Path
   from datetime import datetime, timedelta

   class RFCTracker:
       def __init__(self, rfc_dir: str):
           self.rfc_dir = Path(rfc_dir)

       def list_rfcs(self, status: str = None) -> list[dict]:
           rfcs = []
           for f in sorted(self.rfc_dir.glob("rfc-*.md")):
               metadata = self._parse_metadata(f)
               if status is None or metadata.get("status") == status:
                   rfcs.append(metadata)
           return rfcs

       def stale_rfcs(self, days: int = 14) -> list[dict]:
           """Find RFCs in review for too long."""
           stale = []
           for rfc in self.list_rfcs(status="review"):
               created = datetime.strptime(rfc["created"], "%Y-%m-%d")
               if datetime.now() - created > timedelta(days=days):
                   stale.append(rfc)
           return stale

       def weekly_digest(self) -> str:
           """Generate weekly RFC status update."""
           digest = "## RFC Weekly Digest\n\n"
           for status in ["review", "approved", "draft"]:
               rfcs = self.list_rfcs(status=status)
               if rfcs:
                   digest += f"### {status.title()} ({len(rfcs)})\n"
                   for rfc in rfcs:
                       digest += (f"- **{rfc['title']}** by {rfc['author']} "
                                 f"(created {rfc['created']})\n")
                   digest += "\n"
           stale = self.stale_rfcs()
           if stale:
               digest += f"### ⚠️ Stale ({len(stale)} RFCs in review > 14 days)\n"
               for rfc in stale:
                   digest += f"- {rfc['title']} — needs decision\n"
           return digest

       def _parse_metadata(self, path: Path) -> dict:
           content = path.read_text()
           # Parse the metadata table from the RFC
           metadata = {"file": path.name}
           for line in content.split("\n"):
               if line.startswith("# RFC-"):
                   metadata["title"] = line.lstrip("# ")
               if "**Status**" in line:
                   metadata["status"] = line.split("|")[-2].strip().split("/")[0].strip()
               if "**Author**" in line:
                   metadata["author"] = line.split("|")[-2].strip()
               if "**Created**" in line:
                   metadata["created"] = line.split("|")[-2].strip()
           return metadata
   ```

   GRANT USAGE ON ALL SCHEMAS IN DATABASE novamart_gold TO ROLE data_reader;

   GRANT SELECT ON ALL TABLES IN SCHEMA novamart_gold.mart_revenue TO ROLE data_reader;

   # security/field_encryption.py
   from cryptography.fernet import Fernet
   from functools import lru_cache
   import json
   import os

   class FieldEncryptor:
       """Encrypt PII fields before producing to Kafka."""

       PII_FIELDS = {"email", "phone", "address", "full_name", "ip_address"}

       def __init__(self):
           self._key = os.environ["FIELD_ENCRYPTION_KEY"].encode()
           self._fernet = Fernet(self._key)

       def encrypt_message(self, message: dict, schema: dict) -> dict:
           """Encrypt PII fields in a Kafka message."""
           encrypted = message.copy()
           for field_name, value in message.items():
               if field_name in self.PII_FIELDS and value is not None:
                   encrypted[field_name] = self._fernet.encrypt(
                       str(value).encode()
                   ).decode()
           return encrypted

       def decrypt_message(self, message: dict) -> dict:
           """Decrypt PII fields (only authorized consumers)."""
           decrypted = message.copy()
           for field_name, value in message.items():
               if field_name in self.PII_FIELDS and value is not None:
                   try:
                       decrypted[field_name] = self._fernet.decrypt(
                           value.encode()
                       ).decode()
                   except Exception:
                       pass  # Field wasn't encrypted (backward compat)
           return decrypted
   ```

                   actor STRING NOT NULL,
                   actor_role STRING NOT NULL,

                   resource STRING NOT NULL,

                   rows_affected INTEGER,
                   pii_accessed BOOLEAN DEFAULT FALSE,
                   source_ip STRING,
                   result STRING NOT NULL,
                   metadata VARIANT
               )
           """)

       def log_access(self, actor: str, role: str, action: str,
                      resource: str, **kwargs):
           self.conn.cursor().execute("""
               INSERT INTO audit.data_access_log 
               (actor, actor_role, action, resource, query_text,
                rows_affected, pii_accessed, source_ip, result, metadata)
               VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, 
                       PARSE_JSON(%s))
           """, (
               actor, role, action, resource,
               kwargs.get("query_text"),
               kwargs.get("rows_affected"),
               kwargs.get("pii_accessed", False),
               kwargs.get("source_ip"),
               kwargs.get("result", "success"),
               json.dumps(kwargs.get("metadata", {})),
           ))

       def query_suspicious_access(self, days: int = 7) -> list:
           """Find unusual access patterns."""
           result = self.conn.cursor().execute(f"""
               SELECT actor, actor_role, action, resource,
                      COUNT(*) AS access_count,
                      COUNT(DISTINCT resource) AS unique_resources,
                      SUM(CASE WHEN pii_accessed THEN 1 ELSE 0 END) AS pii_accesses
               FROM audit.data_access_log
               WHERE event_time >= DATEADD(day, -{days}, CURRENT_TIMESTAMP())
               GROUP BY actor, actor_role, action, resource
               HAVING pii_accesses > 10
                  OR access_count > 1000
                  OR unique_resources > 50
               ORDER BY pii_accesses DESC
           """).fetchall()
           return result

   import subprocess
   import json

   class SecurityScanner:
       """Automated security checks run on every PR."""

       def scan_terraform(self, tf_dir: str) -> list[dict]:
           """Check Terraform for security misconfigurations."""
           findings = []

           # Check for overly permissive S3 policies
           result = subprocess.run(
               ["tfsec", tf_dir, "--format", "json", "--soft-fail"],

           for finding in tfsec_results.get("results", []):
               findings.append({
                   "source": "tfsec",
                   "severity": finding["severity"],
                   "description": finding["description"],
                   "file": finding["location"]["filename"],
                   "line": finding["location"]["start_line"],
               })
           return findings

       def scan_sql_permissions(self, sql_files: list[str]) -> list[dict]:
           """Check SQL for overly broad permission grants."""
           findings = []
           dangerous_patterns = [
               "GRANT ALL ON DATABASE",
               "GRANT ACCOUNTADMIN",
               "GRANT SYSADMIN",
               "GRANT ALL PRIVILEGES",
               "COPY INTO .* FROM @",    # Uncontrolled external stage access
           ]
           for sql_file in sql_files:
               with open(sql_file) as f:
                   content = f.read().upper()
                   for pattern in dangerous_patterns:
                       if pattern.upper() in content:
                           findings.append({
                               "source": "sql_scanner",
                               "severity": "HIGH",
                               "description": f"Dangerous pattern: {pattern}",
                               "file": sql_file,
                           })
           return findings

   # mentorship/quality_tracker.py
   from dataclasses import dataclass
   from datetime import datetime

   @dataclass
   class TeamQualityScore:
       team: str
       date: datetime
       test_coverage_pct: float
       sla_defined: bool
       sla_compliance_pct: float | None
       runbook_coverage_pct: float
       oncall_active: bool
       incidents_last_30d: int
       mean_time_to_detect_min: float | None
       documentation_score: float  # 0-100

       @property
       def overall_grade(self) -> str:
           score = 0
           score += min(self.test_coverage_pct, 100) * 0.25
           score += (100 if self.sla_defined else 0) * 0.15
           score += (self.sla_compliance_pct or 0) * 0.20
           score += self.runbook_coverage_pct * 0.15
           score += (100 if self.oncall_active else 0) * 0.10
           score += self.documentation_score * 0.15

           if score >= 90:
               return "A"
           elif score >= 80:
               return "B+"
           elif score >= 70:
               return "B"
           elif score >= 60:
               return "C+"
           elif score >= 50:
               return "C"
           else:
               return "D"

   class QualityBenchmark:
       def __init__(self):
           self.history: list[TeamQualityScore] = []

       def add_assessment(self, score: TeamQualityScore):
           self.history.append(score)

       def team_progress(self, team: str) -> list[dict]:
           team_scores = [s for s in self.history if s.team == team]
           return [
               {
                   "date": s.date.isoformat(),
                   "grade": s.overall_grade,
                   "test_coverage": s.test_coverage_pct,
                   "sla_compliance": s.sla_compliance_pct,
               }
               for s in sorted(team_scores, key=lambda x: x.date)
           ]

       def leaderboard(self) -> str:
           """Generate quality leaderboard for monthly review."""
           latest = {}
           for score in sorted(self.history, key=lambda x: x.date):
               latest[score.team] = score

           board = "## Data Engineering Quality Leaderboard\n\n"
           board += "| Rank | Team | Grade | Tests | SLA | Runbooks |\n"
           board += "|:----:|------|:-----:|:-----:|:---:|:--------:|\n"

           ranked = sorted(latest.values(),
                          key=lambda s: s.test_coverage_pct
                          + (s.sla_compliance_pct or 0),
                          reverse=True)

           for i, score in enumerate(ranked, 1):
               board += (
                   f"| {i} | {score.team} | {score.overall_grade} | "
                   f"{score.test_coverage_pct:.0f}% | "
                   f"{score.sla_compliance_pct or 'N/A'}% | "
                   f"{score.runbook_coverage_pct:.0f}% |\n"
               )

## Task 251: Open Table Formats — Apache Iceberg Migration Strategy

### Business Context
Snowflake is NovaMart's largest vendor at $468K/year. The CFO asks: "Could we move some workloads off Snowflake to reduce cost?" Meanwhile, the ML team uses Databricks, which can't read Snowflake tables natively. Every time they need data, it's an export/import process. Priya proposes: "Apache Iceberg gives us an open table format. Our data lives in S3 in an open format. Snowflake, Databricks, Spark, Flink — they all read the same files. We reduce lock-in and enable multi-engine access."

### Objective
Evaluate and pilot Apache Iceberg as an open table format layer, enabling multi-engine access to the data lake while reducing Snowflake lock-in.

   ```python
   # iceberg/multi_engine.py
   from pyiceberg.catalog import load_catalog
   from pyiceberg.expressions import GreaterThanOrEqual

   class IcebergDataAccess:
       """Unified data access across engines via Iceberg."""

       def __init__(self, catalog_name: str = "novamart"):
           self.catalog = load_catalog(
               catalog_name,
               **{
                   "type": "rest",
                   "uri": "https://iceberg-catalog.novamart.internal",
                   "s3.endpoint": "https://s3.us-east-1.amazonaws.com",
                   "s3.region": "us-east-1",
               }
           )

       def list_tables(self, namespace: str = "gold") -> list[str]:
           """List all Iceberg tables in a namespace."""
           return [
               str(t) for t in self.catalog.list_tables(namespace)
           ]

       def read_table(self, namespace: str, table_name: str,
                      row_filter=None) -> "pyarrow.Table":
           """Read an Iceberg table as PyArrow (zero-copy)."""
           table = self.catalog.load_table(f"{namespace}.{table_name}")
           scan = table.scan()
           if row_filter:
               scan = scan.filter(row_filter)
           return scan.to_arrow()

       def get_table_history(self, namespace: str, 
                             table_name: str) -> list[dict]:
           """Get table snapshot history (time travel)."""
           table = self.catalog.load_table(f"{namespace}.{table_name}")
           return [
               {
                   "snapshot_id": s.snapshot_id,
                   "timestamp_ms": s.timestamp_ms,
                   "operation": s.summary.get("operation", "unknown"),
                   "added_rows": s.summary.get("added-data-files", 0),
               }
               for s in table.history()
           ]

       def time_travel_read(self, namespace: str, table_name: str,
                            snapshot_id: int) -> "pyarrow.Table":
           """Read a table as of a specific snapshot (time travel)."""
           table = self.catalog.load_table(f"{namespace}.{table_name}")
           return table.scan(snapshot_id=snapshot_id).to_arrow()

       def schema_evolution_safe(self, namespace: str, table_name: str,
                                  new_column: str, dtype: str) -> dict:
           """Safely add a column (Iceberg supports schema evolution)."""
           table = self.catalog.load_table(f"{namespace}.{table_name}")
           with table.update_schema() as update:
               update.add_column(new_column, dtype)
           return {
               "table": f"{namespace}.{table_name}",
               "action": "column_added",
               "column": new_column,
               "type": dtype,
           }
   ```

   ```python
   # observability/structured_logging.py
   import json
   import logging
   import time
   from contextlib import contextmanager
   from functools import wraps

   class StructuredLogger:
       """Structured JSON logging with correlation IDs."""

       def __init__(self, service: str, environment: str):
           self.service = service
           self.environment = environment
           self.logger = logging.getLogger(service)
           handler = logging.StreamHandler()
           handler.setFormatter(logging.Formatter("%(message)s"))
           self.logger.addHandler(handler)
           self.logger.setLevel(logging.INFO)

       def log(self, level: str, message: str, **kwargs):
           entry = {
               "timestamp": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
               "service": self.service,
               "environment": self.environment,
               "level": level,
               "message": message,
               **kwargs,
           }
           self.logger.log(
               getattr(logging, level.upper()), json.dumps(entry)
           )

       @contextmanager
       def timed_operation(self, operation: str, **context):
           """Time an operation and log duration."""
           start = time.time()
           self.log("info", f"Starting {operation}", operation=operation,
                    phase="start", **context)
           try:
               yield
               duration = time.time() - start
               self.log("info", f"Completed {operation}",
                        operation=operation, phase="end",
                        duration_seconds=round(duration, 3), **context)
           except Exception as e:
               duration = time.time() - start
               self.log("error", f"Failed {operation}: {e}",
                        operation=operation, phase="error",
                        duration_seconds=round(duration, 3),
                        error=str(e), **context)
               raise

       def pipeline_metric(self, pipeline: str, metric: str,
                           value: float, **tags):
           """Emit a pipeline-specific metric."""
           self.log("info", f"metric:{metric}={value}",
                    pipeline=pipeline, metric_name=metric,
                    metric_value=value, metric_type="gauge", **tags)
   ```

   ```python
   # observability/platform_health.py
   from dataclasses import dataclass
   from datetime import datetime

   @dataclass
   class PlatformHealthSnapshot:
       timestamp: datetime
       freshness_sli: float       # % of tables within SLO
       quality_sli: float         # % of tests passing
       performance_sli: float     # % of pipelines within time SLO
       error_budget_remaining: dict  # SLO -> remaining budget
       active_incidents: int
       cost_today_usd: float

   class PlatformHealthMonitor:
       def __init__(self, snowflake_conn, kafka_admin):
           self.sf = snowflake_conn
           self.kafka = kafka_admin

       def get_health_snapshot(self) -> PlatformHealthSnapshot:
           freshness = self._check_freshness_sli()
           quality = self._check_quality_sli()
           performance = self._check_performance_sli()

           return PlatformHealthSnapshot(
               timestamp=datetime.utcnow(),
               freshness_sli=freshness["pct_within_slo"],
               quality_sli=quality["test_pass_rate"],
               performance_sli=performance["pct_within_slo"],
               error_budget_remaining=self._calculate_error_budgets(),
               active_incidents=0,
               cost_today_usd=self._get_today_cost(),
           )

       def _check_freshness_sli(self) -> dict:
           result = self.sf.cursor().execute("""
               SELECT
                   COUNT(*) AS total_tables,
                   COUNT(CASE 
                       WHEN DATEDIFF(hour, last_altered, CURRENT_TIMESTAMP()) <= 2 
                       THEN 1 END) AS fresh_tables
               FROM information_schema.tables
               WHERE table_schema = 'GOLD'
           """).fetchone()
           total, fresh = result
           return {
               "total_tables": total,
               "fresh_tables": fresh,
               "pct_within_slo": round(fresh / max(total, 1) * 100, 2),
           }

       def _check_quality_sli(self) -> dict:
           # Query dbt test results from the most recent run
           result = self.sf.cursor().execute("""
               SELECT
                   COUNT(*) AS total_tests,
                   COUNT(CASE WHEN status = 'pass' THEN 1 END) AS passed
               FROM DBT_ARTIFACTS.TEST_RESULTS
               WHERE run_started_at = (
                   SELECT MAX(run_started_at) FROM DBT_ARTIFACTS.TEST_RESULTS
               )
           """).fetchone()
           total, passed = result
           return {
               "total_tests": total,
               "passed": passed,
               "test_pass_rate": round(passed / max(total, 1) * 100, 2),
           }

       def _check_performance_sli(self) -> dict:
           result = self.sf.cursor().execute("""
               SELECT
                   COUNT(*) AS total_runs,
                   COUNT(CASE 
                       WHEN DATEDIFF(minute, start_time, end_time) <= 30 
                       THEN 1 END) AS within_slo
               FROM DBT_ARTIFACTS.RUN_RESULTS
               WHERE run_started_at >= DATEADD(day, -7, CURRENT_TIMESTAMP())
           """).fetchone()
           total, within = result
           return {
               "total_runs": total,
               "within_slo": within,
               "pct_within_slo": round(within / max(total, 1) * 100, 2),
           }

       def _calculate_error_budgets(self) -> dict:
           # Error budget = allowed failures per month based on SLO
           return {
               "freshness_99.5": "38 minutes remaining of 43 min budget",
               "quality_99.5": "1 failure remaining of 2 allowed",
               "performance": "2 breaches remaining of 3 allowed",
           }

       def _get_today_cost(self) -> float:
           result = self.sf.cursor().execute("""
               SELECT ROUND(SUM(credits_used) * 3, 2)
               FROM snowflake.account_usage.warehouse_metering_history
               WHERE start_time >= CURRENT_DATE()
           """).fetchone()
           return result[0] if result and result[0] else 0

       def detect_gradual_degradation(self, days: int = 30) -> list[dict]:
           """Detect slowly worsening metrics (the silent killer)."""
           result = self.sf.cursor().execute(f"""
               WITH daily_metrics AS (
                   SELECT
                       DATE(run_started_at) AS run_date,
                       AVG(DATEDIFF(minute, start_time, end_time)) AS avg_duration
                   FROM DBT_ARTIFACTS.RUN_RESULTS
                   WHERE run_started_at >= DATEADD(day, -{days}, CURRENT_TIMESTAMP())
                   GROUP BY run_date
               )
               SELECT
                   REGR_SLOPE(avg_duration, DATEDIFF(day, MIN(run_date) OVER (), 
                       run_date)) AS daily_increase_minutes,
                   AVG(avg_duration) AS current_avg,
                   MIN(avg_duration) AS best_day,
                   MAX(avg_duration) AS worst_day
               FROM daily_metrics
           """).fetchone()

           alerts = []
           if result and result[0] and result[0] > 0.5:  # >30 sec/day increase
               alerts.append({
                   "metric": "dbt_run_duration",
                   "trend": f"+{result[0]:.1f} min/day",
                   "current": f"{result[1]:.0f} min avg",
                   "projected_slo_breach": f"In {max(0, (30 - result[1]) / result[0]):.0f} days",
                   "severity": "warning",
               })
           return alerts
   ```

## Task 253: Advanced Orchestration — Event-Driven Pipeline Architecture

### Business Context
NovaMart's pipelines run on fixed schedules: dbt runs every hour, Flink jobs run continuously, data quality checks run after each dbt run. But the world is event-driven: a customer places an order (event), that order gets shipped (event), the payment clears (event). Running pipelines on clocks when the world produces events is fundamentally mismatched. Result: unnecessary compute (hourly runs when no new data arrived) and delayed processing (new data waits up to 59 minutes for the next scheduled run).

### Objective
Redesign the pipeline architecture from schedule-driven to event-driven: pipelines trigger based on data arrival events, not wall-clock schedules.

### Technical Requirements
- **Tools**: Python, Kafka (events), Airflow/Dagster (orchestration), S3 event notifications
- **Skills**: Event-driven architecture, orchestration patterns, DAG design, sensor patterns

### Step-by-Step Guidance

1. **Audit current schedule vs event alignment**:
   ```markdown
   ## Schedule vs Reality Audit

   | Pipeline | Current Schedule | Data Arrival Pattern | Waste | Delay |
   |----------|:---------------:|---------------------|:-----:|:-----:|
   | Raw → Staging (CDC) | Every 15 min | Continuous (Kafka) | Low | 7.5 min avg |
   | Staging → Gold (dbt) | Hourly | After CDC completes | 40% idle runs | 30 min avg |
   | Quality checks | After dbt | After dbt completes | None | Sequential |
   | ML features | Every 6 hours | After Gold updates | 80% idle runs | 3 hr avg |
   | Dashboards | Every 30 min | After Gold + Quality | 50% idle runs | 15 min avg |
   | Financial reconciliation | Daily 6 AM | After month-end close | 29 days idle | 0 |

   ### Cost of Schedule-Driven Architecture
   - **Wasted compute**: ~40% of dbt runs process no new data ($7K/month wasted)
   - **Unnecessary latency**: Data available in Kafka in seconds, but reaches
     dashboards in 30-90 minutes due to scheduled processing
   - **Cascading delays**: If dbt is 10 minutes late, everything downstream
     shifts by 10 minutes
   ```

2. **Design event-driven pipeline architecture**:
   ```python
   # orchestration/event_driven.py
   from dataclasses import dataclass
   from enum import Enum
   from datetime import datetime

   class TriggerType(Enum):
       DATA_ARRIVAL = "data_arrival"       # New data landed
       PIPELINE_COMPLETE = "pipeline_complete"  # Upstream pipeline finished
       SCHEDULE = "schedule"               # Fixed schedule (still useful for some)
       MANUAL = "manual"                   # Human triggered
       SLA_DEADLINE = "sla_deadline"       # Must run by this time

   @dataclass
   class PipelineEvent:
       event_id: str
       event_type: TriggerType
       source: str             # Which system produced this event
       payload: dict           # Event-specific data
       timestamp: datetime

   @dataclass
   class PipelineTrigger:
       pipeline_name: str
       trigger_type: TriggerType
       trigger_source: str      # What event triggers this
       conditions: list[str]    # All conditions must be true
       timeout_minutes: int     # If trigger doesn't fire, force-run
       dependencies: list[str]  # Other pipelines that must complete first

   class EventDrivenOrchestrator:
       def __init__(self):
           self.triggers = {}
           self.event_log = []

       def register_trigger(self, trigger: PipelineTrigger):
           self.triggers[trigger.pipeline_name] = trigger

       def on_event(self, event: PipelineEvent) -> list[str]:
           """Process an event and return pipelines to trigger."""
           self.event_log.append(event)
           pipelines_to_run = []

           for name, trigger in self.triggers.items():
               if self._should_trigger(trigger, event):
                   pipelines_to_run.append(name)

           return pipelines_to_run

       def _should_trigger(self, trigger: PipelineTrigger,
                           event: PipelineEvent) -> bool:
           # Check event type matches
           if trigger.trigger_type != event.event_type:
               return False

           # Check source matches
           if trigger.trigger_source != event.source:
               return False

           # Check all dependencies met
           for dep in trigger.dependencies:
               if not self._dependency_met(dep):
                   return False

           return True

       def _dependency_met(self, pipeline_name: str) -> bool:
           """Check if a dependency pipeline completed recently."""
           for event in reversed(self.event_log):
               if (event.event_type == TriggerType.PIPELINE_COMPLETE and
                   event.source == pipeline_name):
                   # Completed in last 24 hours
                   age = (datetime.utcnow() - event.timestamp).total_seconds()
                   return age < 86400
           return False

       def define_novamart_pipelines(self) -> list[PipelineTrigger]:
           """Define NovaMart's event-driven pipeline topology."""
           return [
               # CDC: triggered by S3 landing notification
               PipelineTrigger(
                   pipeline_name="raw_to_staging",
                   trigger_type=TriggerType.DATA_ARRIVAL,
                   trigger_source="s3://novamart-raw/cdc/",
                   conditions=["file_count > 0"],
                   timeout_minutes=60,  # Force run hourly if no events
                   dependencies=[],
               ),
               # dbt: triggered by staging completion
               PipelineTrigger(
                   pipeline_name="staging_to_gold",
                   trigger_type=TriggerType.PIPELINE_COMPLETE,
                   trigger_source="raw_to_staging",
                   conditions=["rows_processed > 0"],
                   timeout_minutes=120,
                   dependencies=["raw_to_staging"],
               ),
               # Quality: triggered by dbt completion
               PipelineTrigger(
                   pipeline_name="quality_checks",
                   trigger_type=TriggerType.PIPELINE_COMPLETE,
                   trigger_source="staging_to_gold",
                   conditions=[],
                   timeout_minutes=30,
                   dependencies=["staging_to_gold"],
               ),
               # ML features: triggered by quality passing
               PipelineTrigger(
                   pipeline_name="ml_feature_refresh",
                   trigger_type=TriggerType.PIPELINE_COMPLETE,
                   trigger_source="quality_checks",
                   conditions=["quality_pass_rate > 99"],
                   timeout_minutes=360,
                   dependencies=["quality_checks"],
               ),
               # Financial reconciliation: schedule + event hybrid
               PipelineTrigger(
                   pipeline_name="financial_reconciliation",
                   trigger_type=TriggerType.SCHEDULE,
                   trigger_source="cron:0 6 * * *",  # 6 AM daily
                   conditions=["business_day == true"],
                   timeout_minutes=0,  # Strict schedule
                   dependencies=["staging_to_gold", "quality_checks"],
               ),
           ]
   ```

3. **Implement S3 event notifications**:
   ```python
   # orchestration/s3_events.py
   import json

   class S3EventProcessor:
       """Process S3 event notifications to trigger pipelines."""

       def __init__(self, orchestrator: EventDrivenOrchestrator):
           self.orchestrator = orchestrator

       def handle_s3_event(self, event_record: dict) -> list[str]:
           """Process an S3 event notification (from SQS/SNS)."""
           bucket = event_record["s3"]["bucket"]["name"]
           key = event_record["s3"]["object"]["key"]
           size = event_record["s3"]["object"].get("size", 0)
           event_time = event_record["eventTime"]

           pipeline_event = PipelineEvent(
               event_id=event_record["responseElements"]["x-amz-request-id"],
               event_type=TriggerType.DATA_ARRIVAL,
               source=f"s3://{bucket}/{'/'.join(key.split('/')[:-1])}/",
               payload={
                   "bucket": bucket,
                   "key": key,
                   "size_bytes": size,
               },
               timestamp=datetime.fromisoformat(
                   event_time.replace("Z", "+00:00")
               ),
           )

           return self.orchestrator.on_event(pipeline_event)

       def handle_sqs_batch(self, records: list[dict]) -> dict:
           """Handle a batch of SQS messages containing S3 events."""
           triggered = []
           for record in records:
               body = json.loads(record["body"])
               for s3_record in body.get("Records", []):
                   pipelines = self.handle_s3_event(s3_record)
                   triggered.extend(pipelines)

           return {
               "records_processed": len(records),
               "pipelines_triggered": list(set(triggered)),
   ```

   ```python
   # sharing/partner_manager.py
   from dataclasses import dataclass, field
   from datetime import datetime
   from enum import Enum

   class ShareStatus(Enum):
       ACTIVE = "active"
       SUSPENDED = "suspended"
       TERMINATED = "terminated"

   @dataclass
   class DataShare:
       share_id: str
       partner_name: str
       partner_account: str
       data_scope: str           # Description of what's shared
       shared_views: list[str]
       status: ShareStatus
       created_at: datetime
       last_accessed: datetime = None
       access_count_30d: int = 0
       pii_included: bool = False  # Must always be False for external
       audit_log: list = field(default_factory=list)

   class PartnerSharingManager:
       def __init__(self, snowflake_conn):
           self.conn = snowflake_conn
           self.shares = {}

       def create_share(self, share: DataShare) -> dict:
           """Create a new data share with governance checks."""
           # Governance check: no PII
           if share.pii_included:
               raise ValueError("Cannot share PII data externally")

           # Verify views don't contain PII columns
           for view in share.shared_views:
               pii_check = self._check_for_pii(view)
               if pii_check:
                   raise ValueError(
                       f"View {view} contains PII columns: {pii_check}"
                   )

           self.shares[share.share_id] = share
           return {"status": "created", "share_id": share.share_id}

       def audit_share_access(self, share_id: str) -> list[dict]:
           """Get access audit for a share."""
           result = self.conn.cursor().execute(f"""
               SELECT
                   query_start_time,
                   user_name,
                   role_name,
                   query_type,
                   rows_produced,
                   query_text
               FROM snowflake.account_usage.query_history
               WHERE query_text LIKE '%SHARING.%'
                   AND query_start_time >= DATEADD(day, -30, CURRENT_TIMESTAMP())
               ORDER BY query_start_time DESC
           """).fetchall()

           return [
               {
                   "timestamp": str(r[0]),
                   "user": r[1],
                   "role": r[2],
                   "query_type": r[3],
                   "rows_accessed": r[4],
               }
               for r in result
           ]

       def suspend_share(self, share_id: str, reason: str) -> dict:
           """Immediately suspend a data share."""
           share = self.shares.get(share_id)
           if not share:
               return {"error": "Share not found"}

           share.status = ShareStatus.SUSPENDED
           # Revoke access in Snowflake
           self.conn.cursor().execute(f"""
               ALTER SHARE {share_id} REMOVE ACCOUNTS = {share.partner_account}
           """)
           return {
               "status": "suspended",
               "share_id": share_id,
               "reason": reason,
           }

       def _check_for_pii(self, view_name: str) -> list[str]:
           """Check if a view contains PII columns."""
           pii_patterns = [
               "email", "phone", "name", "address", "ssn",
               "credit_card", "ip_address",
           ]
           result = self.conn.cursor().execute(f"""
               SELECT column_name
               FROM information_schema.columns
               WHERE table_name = '{view_name}'
           """).fetchall()

           pii_found = []
           for (col,) in result:
               for pattern in pii_patterns:
                   if pattern in col.lower():
                       pii_found.append(col)
           return pii_found
   ```

4. **Build data clean room for sensitive analysis**:
   ```python
   # sharing/clean_room.py
   class DataCleanRoom:
       """Secure computation on combined datasets without exposing raw data."""

       def __init__(self, snowflake_conn):
           self.conn = snowflake_conn

       def create_clean_room_analysis(self, partner: str,
                                       analysis_type: str) -> dict:
           """Run pre-approved analysis queries in a clean room."""
           approved_analyses = {
               "overlap_count": """
                   -- How many customers are in both datasets?
                   -- Returns ONLY the count, not the customer list
                   SELECT COUNT(DISTINCT n.customer_hash)
                   FROM NOVAMART.CUSTOMERS n
                   INNER JOIN PARTNER.{partner}.CUSTOMERS p
                       ON n.customer_hash = p.customer_hash
               """,
               "aggregate_behavior": """
                   -- What's the combined behavior profile?
                   -- Returns ONLY aggregates, not individual records
                   SELECT
                       n.region,
                       COUNT(*) AS overlap_customers,
                       AVG(n.lifetime_value) AS avg_ltv,
                       AVG(p.engagement_score) AS avg_engagement
                   FROM NOVAMART.CUSTOMERS n
                   INNER JOIN PARTNER.{partner}.CUSTOMERS p
                       ON n.customer_hash = p.customer_hash
                   GROUP BY n.region
               """,
           }

           if analysis_type not in approved_analyses:
               return {
                   "error": f"Analysis '{analysis_type}' not approved. "
                   f"Available: {list(approved_analyses.keys())}",
               }

           query = approved_analyses[analysis_type].format(partner=partner)

           # Audit log
           self._log_clean_room_query(partner, analysis_type)

           result = self.conn.cursor().execute(query).fetchall()
           return {
               "analysis": analysis_type,
               "partner": partner,
               "result": result,
               "note": "Only aggregate data returned. No individual records exposed.",
           }

       def _log_clean_room_query(self, partner: str, analysis: str):
           self.conn.cursor().execute("""
               INSERT INTO COMPLIANCE.CLEAN_ROOM_LOG
               (partner, analysis_type, executed_at, executed_by)
               VALUES (%s, %s, CURRENT_TIMESTAMP(), CURRENT_USER())
           """, (partner, analysis))
   ```

   ```python
   # retro/improvement_tracker.py
   @dataclass
   class ImprovementInitiative:
       name: str
       category: str
       current_state: str
       target_state: str
       effort_weeks: float
       impact_hours_per_week: float
       owner: str
       status: str  # "proposed", "approved", "in-progress", "completed"

       completed_at: datetime = None

   class ImprovementTracker:
       def __init__(self):
           self.initiatives = []

       def prioritize(self) -> list[ImprovementInitiative]:

           for init in self.initiatives:
               init._roi = init.impact_hours_per_week / max(init.effort_weeks, 0.1)
           return sorted(self.initiatives, key=lambda i: i._roi, reverse=True)

       def quarterly_report(self) -> str:
           completed = [i for i in self.initiatives if i.status == "completed"]
           total_saved = sum(i.impact_hours_per_week for i in completed)

           report = "# Quarterly Improvement Report\n\n"
           report += f"## Initiatives completed: {len(completed)}\n"
           report += f"## Engineering hours saved per week: {total_saved}\n"
           report += f"## Annualized savings: {total_saved * 52} hours "
           report += f"(~{total_saved * 52 / 2080:.1f} FTE)\n\n"

           report += "| Initiative | Hours Saved/Week | Status |\n"
           report += "|-----------|:----------------:|--------|\n"
           for i in self.initiatives:
               report += f"| {i.name} | {i.impact_hours_per_week} | {i.status} |\n"
   ```

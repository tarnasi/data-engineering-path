5. **Build the cross-domain data product validator**:
   ```python
   # mesh/product_validator.py
   import yaml

   REQUIRED_METADATA = ["name", "domain", "owner_email", "description", "version"]
   REQUIRED_SCHEMA = ["table_name", "columns", "primary_key", "grain"]
   REQUIRED_QUALITY = ["freshness_sla", "completeness_sla", "tests"]
   REQUIRED_GOVERNANCE = ["classification", "retention_days", "pii_columns"]

   class DataProductValidator:
       def validate(self, product_spec: dict) -> dict:
           errors = []
           warnings = []

           # Metadata validation
           metadata = product_spec.get("metadata", {})
           for field in REQUIRED_METADATA:
               if field not in metadata:
                   errors.append(f"Missing required metadata field: {field}")

           # Naming convention: domain.product_name
           name = metadata.get("name", "")
           if "." not in name:
               errors.append(
                   f"Product name '{name}' must follow 'domain.product_name' format"
               )

           # Schema validation
           schema = product_spec.get("schema", {})
           for field in REQUIRED_SCHEMA:
               if field not in schema:
                   errors.append(f"Missing required schema field: {field}")

           # Column validation
           for col in schema.get("columns", []):
               if "description" not in col:
                   warnings.append(f"Column '{col.get('name')}' has no description")
               if "type" not in col:
                   errors.append(f"Column '{col.get('name')}' has no type")

           # Quality validation
           quality = product_spec.get("quality", {})
           for field in REQUIRED_QUALITY:
               if field not in quality:
                   errors.append(f"Missing required quality field: {field}")

           tests = quality.get("tests", [])
           test_types = [t.get("type") for t in tests]
           if "unique" not in test_types:
               errors.append("Must include a 'unique' test on primary key")
           if "freshness" not in test_types:
               warnings.append("No freshness test defined (recommended)")

           # Governance validation
           governance = product_spec.get("governance", {})
           for field in REQUIRED_GOVERNANCE:
               if field not in governance:
                   errors.append(f"Missing required governance field: {field}")

           valid_classifications = ["public", "internal", "confidential", "restricted"]
           if governance.get("classification") not in valid_classifications:
               errors.append(
                   f"Invalid classification. Must be one of: {valid_classifications}"
               )

           return {
               "valid": len(errors) == 0,
               "errors": errors,
               "warnings": warnings,
               "product": metadata.get("name", "unknown"),
           }
   ```

## Task 207: Stateful Stream Processing at Production Scale

### Business Context
NovaMart's recommendation engine needs real-time user behavior signals: "customers who viewed this product in the last 10 minutes also bought X." The batch feature store (Task 191) has a 2-hour lag — too slow for real-time recommendations. The ML team needs streaming features with < 1 second latency. But NovaMart's clickstream is 50K events/minute at peak. Your Flink cluster needs to maintain state (user sessions, view counts, cart state) for 2M active users simultaneously.

### Objective
Build a production-grade stateful stream processing system using Apache Flink that computes real-time user features at scale, with proper state management, checkpointing, and failure recovery.

### Technical Requirements
- **Tools**: Apache Flink (Java/Python), Kafka, Redis, RocksDB
- **Skills**: Stateful stream processing, state backend tuning, exactly-once, watermarks, session windows

### Step-by-Step Guidance

1. **Understand the state challenge**:
   ```markdown
   ## Stateful Streaming at Scale

   ```mermaid
   graph LR
       A[Clickstream: 50K events/min] --> B[Flink Job]
       B --> C{State Backend: RocksDB}
       C --> D[User Session State: 2M users]
       C --> E[Windowed Aggregates]
       C --> F[Pattern Detection]
       B --> G[Redis: Real-time Features]
       B --> H[Kafka: Enriched Events]
   ```

   ### Why State is Hard
   | Challenge | Impact | Solution |
   |-----------|--------|---------|
   | 2M user sessions in memory | OOM risk | RocksDB (disk-backed state) |
   | State grows unbounded | Disk fills up | State TTL (expire after 24h) |
   | Exactly-once on failure | Duplicate features | Flink checkpointing + Kafka transactions |
   | Checkpoint size (~50GB) | Slow recovery | Incremental checkpoints |
   | Reprocessing after deploy | State schema migration | Compatible state evolution |
   ```

2. **Build the real-time feature computation**:
   ```python
   # streaming/user_features.py
   from pyflink.datastream import StreamExecutionEnvironment
   from pyflink.datastream.window import SessionWindowTimeGap
   from pyflink.common.time import Time
   from pyflink.datastream.state import (
       ValueStateDescriptor, MapStateDescriptor,
       StateTtlConfig
   )
   from pyflink.datastream.functions import KeyedProcessFunction
   from pyflink.common.typeinfo import Types
   import json

   class UserFeatureProcessor(KeyedProcessFunction):
       """Process clickstream events and maintain real-time user state."""

       def open(self, runtime_context):
           # State: last N product views per user
           ttl_config = StateTtlConfig.builder(Time.hours(24)) \
               .set_update_type(StateTtlConfig.UpdateType.OnReadAndWrite) \
               .build()

           views_descriptor = ValueStateDescriptor(
               "recent_views", Types.STRING()
           )
           views_descriptor.enable_time_to_live(ttl_config)
           self.recent_views = runtime_context.get_state(views_descriptor)

           # State: session metrics
           session_descriptor = ValueStateDescriptor(
               "session_metrics", Types.STRING()
           )
           session_descriptor.enable_time_to_live(ttl_config)
           self.session_metrics = runtime_context.get_state(session_descriptor)

           # State: purchase history (for "viewed then bought" features)
           purchases_descriptor = ValueStateDescriptor(
               "recent_purchases", Types.STRING()
           )
           purchases_descriptor.enable_time_to_live(ttl_config)
           self.recent_purchases = runtime_context.get_state(purchases_descriptor)

       def process_element(self, event, ctx):
           user_id = ctx.get_current_key()
           event_data = json.loads(event)
           event_type = event_data["event_type"]

           # Update session metrics
           session = json.loads(self.session_metrics.value() or '{}')
           session["page_views"] = session.get("page_views", 0) + 1
           session["last_event_ts"] = event_data["timestamp"]
           session.setdefault("session_start", event_data["timestamp"])

           if event_type == "product_view":
               # Update recent views (keep last 20)
               views = json.loads(self.recent_views.value() or '[]')
               views.append({
                   "product_id": event_data["product_id"],
                   "timestamp": event_data["timestamp"],
                   "category": event_data.get("category", "unknown"),
               })
               views = views[-20:]  # Keep last 20
               self.recent_views.update(json.dumps(views))

               session["product_views"] = session.get("product_views", 0) + 1

           elif event_type == "add_to_cart":
               session["cart_adds"] = session.get("cart_adds", 0) + 1
               session["cart_value"] = (session.get("cart_value", 0)
                                       + event_data.get("price", 0))

           elif event_type == "purchase":
               purchases = json.loads(self.recent_purchases.value() or '[]')
               purchases.append({
                   "product_id": event_data["product_id"],
                   "amount": event_data["amount"],
                   "timestamp": event_data["timestamp"],
               })
               purchases = purchases[-50:]
               self.recent_purchases.update(json.dumps(purchases))

           self.session_metrics.update(json.dumps(session))

           # Emit real-time features
           views = json.loads(self.recent_views.value() or '[]')
           features = {
               "user_id": user_id,
               "timestamp": event_data["timestamp"],
               "features": {
                   "session_page_views": session.get("page_views", 0),
                   "session_product_views": session.get("product_views", 0),
                   "session_cart_adds": session.get("cart_adds", 0),
                   "session_cart_value": session.get("cart_value", 0),
                   "recent_view_categories": list(set(
                       v["category"] for v in views[-10:]
                   )),
                   "views_last_10min": sum(
                       1 for v in views
                       if _within_minutes(v["timestamp"],
                                         event_data["timestamp"], 10)
                   ),
                   "is_browsing_heavily": session.get("product_views", 0) > 10,
               }
           }
           yield json.dumps(features)
   ```

3. **Configure the Flink job for production scale**:
   ```python
   # streaming/flink_job.py

   def build_flink_job():
       env = StreamExecutionEnvironment.get_execution_environment()

       # Checkpointing: exactly-once, every 60 seconds
       env.enable_checkpointing(60_000)  # 60s interval
       checkpoint_config = env.get_checkpoint_config()
       checkpoint_config.set_checkpointing_mode(
           CheckpointingMode.EXACTLY_ONCE
       )
       checkpoint_config.set_min_pause_between_checkpoints(30_000)
       checkpoint_config.set_checkpoint_timeout(120_000)
       checkpoint_config.set_max_concurrent_checkpoints(1)
       checkpoint_config.enable_unaligned_checkpoints()

       # State backend: RocksDB for large state
       env.set_state_backend(RocksDBStateBackend(
           "s3://novamart-flink/checkpoints",
           True  # Incremental checkpoints
       ))

       # Parallelism: 8 task slots for 50K events/min
       env.set_parallelism(8)

       # Source: Kafka clickstream
       kafka_source = KafkaSource.builder() \
           .set_bootstrap_servers("kafka:9092") \
           .set_topics("clickstream.events") \
           .set_group_id("user-features-v1") \
           .set_starting_offsets(KafkaOffsetsInitializer.latest()) \
           .set_value_only_deserializer(SimpleStringSchema()) \
           .build()

       # Process
       events = env.from_source(
           kafka_source, WatermarkStrategy.no_watermarks(), "Clickstream"
       )

       features = events \
           .key_by(lambda e: json.loads(e)["user_id"]) \
           .process(UserFeatureProcessor())

       # Sink: Redis for online serving
       features.add_sink(RedisSink())

       # Sink: Kafka for downstream consumers
       features.sink_to(
           KafkaSink.builder()
           .set_bootstrap_servers("kafka:9092")
           .set_record_serializer(
               KafkaRecordSerializationSchema.builder()
               .set_topic("user.features.realtime")
               .set_value_serialization_schema(SimpleStringSchema())
               .build()
           )
           .set_delivery_guarantee(DeliveryGuarantee.EXACTLY_ONCE)
           .build()
       )

       env.execute("User Real-Time Features")
   ```

### Common Mistakes
- Unbounded state — without TTL, state grows until the job OOMs
- Checkpoint too frequently — each checkpoint has I/O cost; balance freshness vs throughput
- Not using incremental checkpoints — full checkpoints of 50GB state take minutes; incremental takes seconds
- Ignoring state schema evolution — if you change the state format, you can't restore from old checkpoints
- Testing only with small data — state behavior changes dramatically at 2M keys vs 100 keys

   -- snowflake_costs/warehouse_analysis.sql
   -- Find the most expensive queries and warehouses

   -- Top warehouses by cost (last 30 days)
   SELECT
       warehouse_name,
       SUM(credits_used) AS total_credits,
       ROUND(SUM(credits_used) * 3.00, 2) AS estimated_cost_usd,
       COUNT(DISTINCT DATE_TRUNC('day', start_time)) AS active_days,
       ROUND(AVG(credits_used), 4) AS avg_credits_per_day,
       MAX(warehouse_size) AS max_size
   FROM SNOWFLAKE.ACCOUNT_USAGE.WAREHOUSE_METERING_HISTORY
   WHERE start_time >= DATEADD(day, -30, CURRENT_TIMESTAMP())
   GROUP BY warehouse_name
   ORDER BY total_credits DESC;

   -- Idle warehouse detection (running but no queries)
   WITH warehouse_usage AS (
       SELECT
           warehouse_name,
           DATE_TRUNC('hour', start_time) AS hour,
           SUM(credits_used) AS credits
       FROM SNOWFLAKE.ACCOUNT_USAGE.WAREHOUSE_METERING_HISTORY
       WHERE start_time >= DATEADD(day, -7, CURRENT_TIMESTAMP())
       GROUP BY 1, 2
   ),
   warehouse_queries AS (
       SELECT
           warehouse_name,
           DATE_TRUNC('hour', start_time) AS hour,
           COUNT(*) AS query_count
       FROM SNOWFLAKE.ACCOUNT_USAGE.QUERY_HISTORY
       WHERE start_time >= DATEADD(day, -7, CURRENT_TIMESTAMP())
       GROUP BY 1, 2
   )
   SELECT
       u.warehouse_name,
       u.hour,
       u.credits,
       COALESCE(q.query_count, 0) AS queries,
       ROUND(u.credits * 3.00, 2) AS wasted_cost_usd
   FROM warehouse_usage u
   LEFT JOIN warehouse_queries q
       ON u.warehouse_name = q.warehouse_name AND u.hour = q.hour
   WHERE COALESCE(q.query_count, 0) = 0
       AND u.credits > 0
   ORDER BY wasted_cost_usd DESC;

   -- Top expensive queries (candidates for optimization)
   SELECT
       query_id,
       user_name,
       warehouse_name,
       ROUND(total_elapsed_time / 1000, 1) AS duration_sec,
       ROUND(credits_used_cloud_services, 4) AS credits,
       bytes_scanned / (1024*1024*1024) AS gb_scanned,
       partitions_scanned,
       partitions_total,
       ROUND(partitions_scanned::FLOAT / NULLIF(partitions_total, 0) * 100, 1) 
           AS partition_scan_pct,
       SUBSTR(query_text, 1, 200) AS query_preview
   FROM SNOWFLAKE.ACCOUNT_USAGE.QUERY_HISTORY
   WHERE start_time >= DATEADD(day, -7, CURRENT_TIMESTAMP())
       AND total_elapsed_time > 30000  -- > 30 seconds
   ORDER BY total_elapsed_time DESC
   LIMIT 50;
   ```

2. **Build cost attribution by domain**:
   ```python
   # finops/cost_attribution.py
   import pandas as pd

   class CostAttribution:
       """Attribute platform costs to domain teams."""

       # Static costs split evenly across domains
       SHARED_COSTS = {
           "kafka_base": 4000,       # Base Confluent cluster
           "airflow_base": 2500,     # MWAA base cost
           "monitoring": 1500,       # Grafana + PagerDuty
           "platform_team": 0,       # Team cost not attributed to domains
       }

       def calculate_domain_costs(self, snowflake_usage: pd.DataFrame,
                                   s3_usage: pd.DataFrame,
                                   domain_pipeline_map: dict) -> pd.DataFrame:
           """
           Attribution rules:
           - Snowflake: by warehouse usage (tagged warehouses)
           - S3: by storage volume per domain prefix
           - Kafka: by topic throughput per domain
           - Shared costs: split proportionally by pipeline count
           """
           domains = list(domain_pipeline_map.keys())
           total_pipelines = sum(len(v) for v in domain_pipeline_map.values())

           results = []
           for domain in domains:
               pipeline_count = len(domain_pipeline_map[domain])
               share_pct = pipeline_count / total_pipelines

               # Snowflake: sum credits for domain-tagged warehouses
               domain_wh = snowflake_usage[
                   snowflake_usage["warehouse_name"].str.startswith(f"WH_{domain.upper()}")
               ]
               sf_cost = domain_wh["credits"].sum() * 3.00

               # S3: storage for domain prefix
               domain_s3 = s3_usage[s3_usage["prefix"].str.startswith(f"data/{domain}")]
               s3_cost = domain_s3["cost_usd"].sum()

               # Shared costs: proportional to pipeline count
               shared = sum(self.SHARED_COSTS.values()) * share_pct

               results.append({
                   "domain": domain,
                   "pipelines": pipeline_count,
                   "snowflake_cost": round(sf_cost, 2),
                   "s3_cost": round(s3_cost, 2),

## Task 209: Statistical Process Control — Anomaly Detection at Scale

### Business Context
NovaMart now has 80+ pipelines across 8 domains. Manual data quality checks don't scale. Last week, a currency conversion bug in the QuickShop integration went undetected for 3 days because the row counts looked normal — only the values were wrong. Priya says: "Row counts aren't enough. We need statistical monitoring that catches drift in *values*, not just counts. And it needs to work across all 80 pipelines without someone hand-tuning thresholds for each one."

### Objective
Build an automated anomaly detection system using statistical process control (SPC) that monitors data quality metrics across all pipelines, detects drift in both volume and values, and alerts with low false-positive rates.

### Technical Requirements
- **Tools**: Python (scipy, pandas, numpy), SQL (Snowflake), Grafana
- **Skills**: Statistical process control, anomaly detection, time series analysis, alert tuning

### Step-by-Step Guidance

1. **Understand SPC for data pipelines**:
   ```markdown
   ## Statistical Process Control for Data

   Traditional monitoring: "Alert if row count < 1000"
   Problem: What if normal is sometimes 800 on weekends?

   SPC approach: "Alert if row count deviates > 3σ from the 
   expected value for this day-of-week and time-of-day."

   ### Control Chart Types for Data Pipelines
   | Chart Type | What It Monitors | When to Use |
   |-----------|-----------------|-------------|
   | X-bar | Mean of a metric over time | Revenue, row counts |
   | R-chart | Variability of a metric | Consistency of batch sizes |
   | P-chart | Proportion (% rate) | Null rates, error rates |
   | CUSUM | Cumulative deviation from target | Gradual drift detection |
   | EWMA | Exponentially weighted moving average | Smooth trend changes |
   ```

2. **Build the anomaly detection engine**:
   ```python
   # anomaly/detector.py
   import numpy as np
   import pandas as pd
   from scipy import stats
   from dataclasses import dataclass
   from datetime import datetime, timezone

   @dataclass
   class AnomalyResult:
       metric_name: str
       pipeline: str
       current_value: float
       expected_value: float
       lower_bound: float
       upper_bound: float
       z_score: float
       is_anomaly: bool
       severity: str  # "warning" | "critical"
       detection_method: str
       timestamp: datetime

   class DataAnomalyDetector:
       """Multi-method anomaly detection for data pipeline metrics."""

       def __init__(self, history_days: int = 30, sigma_threshold: float = 3.0):
           self.history_days = history_days
           self.sigma_threshold = sigma_threshold

       def detect_zscore(self, metric_name: str, pipeline: str,
                         history: pd.Series, current: float) -> AnomalyResult:
           """Z-score based detection with day-of-week adjustment."""
           if len(history) < 7:
               return self._insufficient_data(metric_name, pipeline, current)

           mean = history.mean()
           std = history.std()

           if std == 0:
               # No variance — any deviation is anomalous
               is_anomaly = current != mean
               z = 0 if not is_anomaly else float('inf')
           else:
               z = abs(current - mean) / std
               is_anomaly = z > self.sigma_threshold

           return AnomalyResult(
               metric_name=metric_name,
               pipeline=pipeline,
               current_value=current,
               expected_value=round(mean, 2),
               lower_bound=round(mean - self.sigma_threshold * std, 2),
               upper_bound=round(mean + self.sigma_threshold * std, 2),
               z_score=round(z, 2),
               is_anomaly=is_anomaly,
               severity="critical" if z > 4 else "warning" if z > 3 else "normal",
               detection_method="z-score",
               timestamp=datetime.now(timezone.utc),
           )

       def detect_ewma(self, metric_name: str, pipeline: str,
                       history: pd.Series, current: float,
                       span: int = 7) -> AnomalyResult:
           """EWMA-based detection (better for trending metrics)."""
           ewma = history.ewm(span=span).mean()
           ewma_std = history.ewm(span=span).std()

           expected = ewma.iloc[-1]
           std = ewma_std.iloc[-1]

           if std == 0 or pd.isna(std):
               return self._insufficient_data(metric_name, pipeline, current)

           z = abs(current - expected) / std
           return AnomalyResult(
               metric_name=metric_name,
               pipeline=pipeline,
               current_value=current,
               expected_value=round(expected, 2),
               lower_bound=round(expected - self.sigma_threshold * std, 2),
               upper_bound=round(expected + self.sigma_threshold * std, 2),
               z_score=round(z, 2),
               is_anomaly=z > self.sigma_threshold,
               severity="critical" if z > 4 else "warning" if z > 3 else "normal",
               detection_method="ewma",
               timestamp=datetime.now(timezone.utc),
           )

       def detect_seasonal(self, metric_name: str, pipeline: str,
                           history: pd.DataFrame, current: float,
                           day_of_week: int) -> AnomalyResult:
           """Day-of-week adjusted detection (handles weekday/weekend patterns)."""
           # Filter history to same day of week
           same_dow = history[history.index.dayofweek == day_of_week]["value"]

           if len(same_dow) < 4:
               return self.detect_zscore(metric_name, pipeline,
                                        history["value"], current)

           return self.detect_zscore(metric_name, pipeline, same_dow, current)

       def detect_drift(self, metric_name: str, pipeline: str,
                        history: pd.Series, window: int = 7) -> dict:
           """Detect gradual drift using CUSUM (Cumulative Sum)."""
           target = history[:-window].mean()
           recent = history[-window:]

           cusum_pos = 0
           cusum_neg = 0
           threshold = self.sigma_threshold * history.std()

           alerts = []
           for i, val in enumerate(recent):
               cusum_pos = max(0, cusum_pos + (val - target - threshold / 2))
               cusum_neg = max(0, cusum_neg + (target - val - threshold / 2))

               if cusum_pos > threshold or cusum_neg > threshold:
                   alerts.append({
                       "index": i,
                       "direction": "increasing" if cusum_pos > threshold else "decreasing",
                       "magnitude": max(cusum_pos, cusum_neg),
                   })

           return {
               "metric": metric_name,
               "pipeline": pipeline,
               "drift_detected": len(alerts) > 0,
               "direction": alerts[-1]["direction"] if alerts else None,
               "details": alerts,
           }

       def _insufficient_data(self, metric_name, pipeline, current):
           return AnomalyResult(
               metric_name=metric_name, pipeline=pipeline,
               current_value=current, expected_value=0,
               lower_bound=0, upper_bound=0, z_score=0,
               is_anomaly=False, severity="normal",
               detection_method="insufficient_data",
               timestamp=datetime.now(timezone.utc),
           )
   ```

3. **Build the pipeline monitoring orchestrator**:
   ```python
   # anomaly/monitor.py
   class PipelineMonitor:
       """Run anomaly detection across all pipelines after each batch."""

       MONITORED_METRICS = [
           "row_count",
           "null_rate",
           "distinct_count",
           "total_amount_sum",
           "avg_value",
           "max_value",
           "freshness_minutes",
       ]

       def __init__(self, conn, detector: DataAnomalyDetector):
           self.conn = conn
           self.detector = detector

       def check_pipeline(self, pipeline: str,
                          metrics: dict[str, float]) -> list[AnomalyResult]:
           """Check all metrics for a single pipeline run."""
           results = []

           for metric_name, current_value in metrics.items():
               if metric_name not in self.MONITORED_METRICS:
                   continue

               # Get history
               history = self._get_metric_history(pipeline, metric_name)

               if len(history) < 7:
                   continue

               # Choose detection method based on metric type
               if metric_name in ("row_count", "total_amount_sum"):
                   result = self.detector.detect_seasonal(
                       metric_name, pipeline, history, current_value,
                       day_of_week=datetime.now(timezone.utc).weekday()
                   )
               elif metric_name in ("null_rate",):
                   result = self.detector.detect_ewma(
                       metric_name, pipeline, history["value"], current_value
                   )
               else:
                   result = self.detector.detect_zscore(
                       metric_name, pipeline, history["value"], current_value
                   )

               results.append(result)

           return results

       def check_all_pipelines(self) -> dict:
           """Run checks across all 80+ pipelines."""
           pipelines = self._get_active_pipelines()
           all_anomalies = []

           for pipeline in pipelines:
               metrics = self._get_latest_metrics(pipeline)
               anomalies = self.check_pipeline(pipeline, metrics)
               critical = [a for a in anomalies if a.is_anomaly]
               all_anomalies.extend(critical)

           return {
               "pipelines_checked": len(pipelines),
               "anomalies_found": len(all_anomalies),
               "critical": [a for a in all_anomalies if a.severity == "critical"],
               "warnings": [a for a in all_anomalies if a.severity == "warning"],
           }
   ```

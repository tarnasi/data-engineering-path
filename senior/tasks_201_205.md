4. **Build the IDP API**:
   ```python
   # idp/api.py
   from fastapi import FastAPI, HTTPException
   from pydantic import BaseModel, field_validator
   import yaml

   app = FastAPI(title="NovaMart IDP", version="1.0.0")

   class PipelineSpec(BaseModel):
       apiVersion: str
       kind: str
       metadata: dict
       spec: dict

       @field_validator("apiVersion")
       @classmethod
       def check_version(cls, v):
           if v != "novamart/v1":
               raise ValueError(f"Unsupported API version: {v}")
           return v

       @field_validator("kind")
       @classmethod
       def check_kind(cls, v):
           valid = ["DataPipeline", "DataProduct", "StreamingPipeline"]
           if v not in valid:
               raise ValueError(f"Invalid kind: {v}. Must be one of {valid}")
           return v

   class ProvisionResult(BaseModel):
       pipeline_name: str
       status: str
       resources_created: list[str]
       estimated_monthly_cost: float
       pr_url: str | None
       next_steps: list[str]

   @app.post("/api/v1/pipelines", response_model=ProvisionResult)
   async def provision_pipeline(spec: PipelineSpec):
       """Provision all infrastructure for a new data pipeline."""
       # 1. Validate against policies
       policy_result = await validate_policies(spec)
       if not policy_result.allowed:
           raise HTTPException(403, f"Policy violation: {policy_result.reason}")

       # 2. Estimate cost
       cost = estimate_monthly_cost(spec)
       if cost > 5000:
           raise HTTPException(400,
               f"Estimated cost ${cost}/mo exceeds $5K limit. "
               "Submit for architecture review.")

       # 3. Provision resources
       resources = []
       orchestrator = ProvisionOrchestrator()

       # Snowflake schema + tables
       sf_result = await orchestrator.provision_snowflake(spec)
       resources.extend(sf_result)

       # Kafka topics (if CDC/streaming)
       if spec.spec.get("ingestion", {}).get("kafka_topic"):
           kafka_result = await orchestrator.provision_kafka(spec)
           resources.extend(kafka_result)

       # dbt project scaffolding
       dbt_result = await orchestrator.scaffold_dbt(spec)
       resources.extend(dbt_result)

       # Airflow DAG
       dag_result = await orchestrator.generate_dag(spec)
       resources.extend(dag_result)

       # Monitoring dashboard
       monitor_result = await orchestrator.setup_monitoring(spec)
       resources.extend(monitor_result)

       # 4. Create PR
       pr_url = await orchestrator.create_pull_request(spec, resources)

       # 5. Audit log
       await audit_log(spec, resources, cost)

       return ProvisionResult(
           pipeline_name=spec.metadata["name"],
           status="provisioned",
           resources_created=resources,
           estimated_monthly_cost=cost,
           pr_url=pr_url,
           next_steps=[
               "Review the generated PR and add your business logic",
               f"Estimated monthly cost: ${cost:.2f}",
               "Pipeline will auto-deploy after PR approval",
               "Monitoring dashboard: grafana.novamart.io/d/{name}",
           ],
       )

   @app.get("/api/v1/pipelines/{name}/status")
   async def pipeline_status(name: str):
       """Get current status of a provisioned pipeline."""
       pipeline = await get_pipeline(name)
       if not pipeline:
           raise HTTPException(404, f"Pipeline '{name}' not found")
       return {
           "name": name,
           "status": pipeline["status"],
           "last_run": pipeline.get("last_run"),
           "health": pipeline.get("health", "unknown"),
           "cost_mtd": pipeline.get("cost_mtd", 0),
       }
   ```

5. **Build the provisioning orchestrator**:
   ```python
   # idp/orchestrator.py
   import subprocess
   import json
   from jinja2 import Environment, FileSystemLoader

   class ProvisionOrchestrator:
       def __init__(self):
           self.jinja = Environment(
               loader=FileSystemLoader("idp/templates")
           )

       async def provision_snowflake(self, spec) -> list[str]:
           """Create Snowflake schema, tables, and grants via Terraform."""
           tf_vars = {
               "schema_name": spec.spec["storage"]["snowflake_schema"],
               "tables": spec.spec["storage"]["tables"],
               "owner_role": f"{spec.metadata['domain']}_writer",
               "classification": spec.spec["governance"]["data_classification"],
               "pii_columns": spec.spec["governance"].get("pii_columns", []),
           }
           # Generate Terraform HCL from template
           template = self.jinja.get_template("snowflake_schema.tf.j2")
           tf_content = template.render(**tf_vars)

           # Apply Terraform
           result = await self._run_terraform(tf_content, spec.metadata["name"])
           return [f"snowflake_schema:{tf_vars['schema_name']}"] + [
               f"snowflake_table:{t['name']}" for t in tf_vars["tables"]
           ]

       async def provision_kafka(self, spec) -> list[str]:
           """Create Kafka topic with appropriate configuration."""
           topic_config = {
               "name": spec.spec["ingestion"]["kafka_topic"],
               "partitions": spec.spec["ingestion"].get("partitions", 6),
               "retention_ms": spec.spec["ingestion"].get(
                   "retention_days", 7) * 86400000,
               "cleanup_policy": "delete",
           }
           template = self.jinja.get_template("kafka_topic.tf.j2")
           tf_content = template.render(**topic_config)
           await self._run_terraform(tf_content, spec.metadata["name"])
           return [f"kafka_topic:{topic_config['name']}"]

       async def scaffold_dbt(self, spec) -> list[str]:
           """Generate dbt models, tests, and schema.yml from spec."""
           models = spec.spec.get("transform", {}).get("dbt", {}).get("models", [])
           quality = spec.spec.get("quality", {}).get("tests", [])
           resources = []

           for model in models:
               template = self.jinja.get_template("dbt_model.sql.j2")
               sql = template.render(model=model, source=spec.spec["storage"])
               resources.append(f"dbt_model:{model['name']}")

           # Generate schema.yml with tests
           schema_template = self.jinja.get_template("dbt_schema.yml.j2")
           schema = schema_template.render(models=models, tests=quality)
           resources.append("dbt_schema:schema.yml")
           return resources

       async def generate_dag(self, spec) -> list[str]:
           """Generate Airflow DAG from spec using DAG factory pattern."""
           dag_config = {
               "dag_id": f"pipeline_{spec.metadata['name'].replace('-', '_')}",
               "schedule": spec.spec["orchestration"]["schedule"],
               "sla_minutes": spec.spec["orchestration"]["sla_minutes"],
               "retries": spec.spec["orchestration"]["retries"],
               "dbt_models": [m["name"] for m in
                   spec.spec.get("transform", {}).get("dbt", {}).get("models", [])],
               "alert_channels": spec.spec["orchestration"]["alert_channels"],
           }
           template = self.jinja.get_template("airflow_dag.py.j2")
           dag_code = template.render(**dag_config)
           return [f"airflow_dag:{dag_config['dag_id']}"]

       async def setup_monitoring(self, spec) -> list[str]:
           """Create Grafana dashboard and PagerDuty service."""
           if not spec.spec.get("monitoring", {}).get("dashboard"):
               return []
           metrics = spec.spec["monitoring"].get("metrics", [])
           template = self.jinja.get_template("grafana_dashboard.json.j2")
           dashboard = template.render(
               name=spec.metadata["name"],
               metrics=metrics
           )
           return [f"grafana_dashboard:{spec.metadata['name']}"]
   ```

   ```python
   # idp/cli.py
   import click
   import yaml
   import httpx

   IDP_URL = "https://idp.novamart.internal/api/v1"

   @click.group()
   def cli():
       """novactl — NovaMart Data Platform CLI"""
       pass

   @cli.command()
   @click.argument("spec_file", type=click.Path(exists=True))
   @click.option("--dry-run", is_flag=True, help="Validate without provisioning")
   def create(spec_file: str, dry_run: bool):
       """Create a new data pipeline from a YAML spec."""
       with open(spec_file) as f:
           spec = yaml.safe_load(f)

       if dry_run:
           click.echo("🔍 Validating spec...")
           resp = httpx.post(f"{IDP_URL}/pipelines/validate", json=spec)
       else:
           click.echo("🚀 Provisioning pipeline...")
           resp = httpx.post(f"{IDP_URL}/pipelines", json=spec)

       if resp.status_code == 200:
           result = resp.json()
           click.echo(f"✅ Pipeline '{result['pipeline_name']}' provisioned!")
           click.echo(f"   Resources: {len(result['resources_created'])}")
           click.echo(f"   Est. cost: ${result['estimated_monthly_cost']:.2f}/mo")
           if result.get("pr_url"):
               click.echo(f"   PR: {result['pr_url']}")
           click.echo("\nNext steps:")
           for step in result["next_steps"]:
               click.echo(f"   → {step}")
       else:
           click.echo(f"❌ Error: {resp.json().get('detail', resp.text)}")

   @cli.command()
   @click.argument("name")
   def status(name: str):
       """Check pipeline status."""
       resp = httpx.get(f"{IDP_URL}/pipelines/{name}/status")
       if resp.status_code == 200:
           info = resp.json()
           click.echo(f"Pipeline: {info['name']}")
           click.echo(f"Status:   {info['status']}")
           click.echo(f"Health:   {info['health']}")
           click.echo(f"Last run: {info['last_run']}")
           click.echo(f"Cost MTD: ${info['cost_mtd']:.2f}")
       else:
           click.echo(f"❌ Pipeline '{name}' not found")

   @cli.command()
   def init():
       """Generate a starter pipeline spec."""
       template = {
           "apiVersion": "novamart/v1",
           "kind": "DataPipeline",
           "metadata": {
               "name": "my-pipeline",
               "domain": "my-domain",
               "owner": "my-team",
               "contact": "team@novamart.com",
               "description": "Description of your pipeline",
   ```

   ```python
   # adr/adr_manager.py
   import os
   import re
   from datetime import datetime, timezone

   class ADRManager:
       def __init__(self, adr_dir: str = "docs/adrs"):
           self.adr_dir = adr_dir

           next_num = self._next_number()

   - [Driver 2]

   **Pros**: 
   **Cons**: 
   **Cost**: 
   **Operational burden**:

   ### Positive
   ### Negative
   ### Risks

   ## Decision Authority
   - **Author**: 
   - **Approvers**: 
   - **Informed**: 

   ## Review Date
   [When to revisit]
   """
           with open(filepath, "w") as f:
               f.write(content)
           return filepath

       def list_adrs(self) -> list[dict]:
           """List all ADRs with status."""
           adrs = []
           for f in sorted(os.listdir(self.adr_dir)):
               if f.startswith("adr-") and f.endswith(".md"):
                   path = os.path.join(self.adr_dir, f)
                   with open(path) as fh:
                       content = fh.read()
                   status_match = re.search(r"## Status\n(.+)", content)
                   status = status_match.group(1).strip() if status_match else "Unknown"
                   title_match = re.search(r"# (ADR-\d+: .+)", content)
                   title = title_match.group(1) if title_match else f
                   adrs.append({"file": f, "title": title, "status": status})
           return adrs

       def _next_number(self) -> int:
           existing = [f for f in os.listdir(self.adr_dir)
                      if f.startswith("adr-")]
           if not existing:
               return 1
           numbers = [int(re.search(r"adr-(\d+)", f).group(1))
                     for f in existing if re.search(r"adr-(\d+)", f)]
           return max(numbers) + 1 if numbers else 1
   ```

   ```python
   # migration/dual_writer.py
   import pyarrow as pa
   import pyarrow.parquet as pq
   from pyiceberg.catalog import load_catalog
   from pyiceberg.schema import Schema
   from pyiceberg.types import (
       NestedField, StringType, TimestampType, DecimalType, LongType
   )

   class DualWriter:
       """Write to both Snowflake (native) and Iceberg simultaneously."""

       def __init__(self, snowflake_conn, iceberg_catalog_uri: str,
                    s3_warehouse: str):
           self.sf = snowflake_conn
           self.catalog = load_catalog(
               "novamart",
               **{
                   "uri": iceberg_catalog_uri,
                   "s3.endpoint": "https://s3.amazonaws.com",
                   "warehouse": s3_warehouse,
               }
           )

       def write_batch(self, table_name: str, df, partition_cols: list[str]):
           """Write a batch to both destinations."""
           # Write to Snowflake (existing path — unchanged)
           self._write_snowflake(table_name, df)

           # Write to Iceberg (new path — shadow)
           self._write_iceberg(table_name, df, partition_cols)

       def _write_snowflake(self, table_name: str, df):
           """Existing Snowflake write — no changes."""
           df.to_sql(table_name, self.sf, if_exists="append", index=False)

       def _write_iceberg(self, table_name: str, df, partition_cols: list[str]):
           """New Iceberg write path."""
           iceberg_table_name = f"novamart.iceberg_{table_name}"

           # Get or create Iceberg table
           try:
               table = self.catalog.load_table(iceberg_table_name)
           except Exception:
               arrow_schema = pa.Schema.from_pandas(df)
               table = self.catalog.create_table(
                   iceberg_table_name,
                   schema=arrow_schema,
               )

           # Append data
           arrow_table = pa.Table.from_pandas(df)
           table.append(arrow_table)

       def validate(self, table_name: str, date_range: tuple) -> dict:
           """Compare row counts and checksums between destinations."""
           start, end = date_range

           # Snowflake count + checksum
           sf_result = self.sf.execute(f"""
               SELECT COUNT(*) as cnt,
                      SUM(HASH(*)) as checksum
               FROM {table_name}
               WHERE _loaded_at BETWEEN '{start}' AND '{end}'
           """).fetchone()

           # Iceberg count + checksum
           iceberg_table = self.catalog.load_table(
               f"novamart.iceberg_{table_name}"
           )
           iceberg_df = iceberg_table.scan(
               row_filter=f"_loaded_at >= '{start}' AND _loaded_at <= '{end}'"
           ).to_pandas()

           return {
               "table": table_name,
               "snowflake_count": sf_result[0],
               "iceberg_count": len(iceberg_df),
               "counts_match": sf_result[0] == len(iceberg_df),
               "date_range": date_range,
           }
   ```

4. **Build the migration orchestrator**:
   ```python
   # migration/orchestrator.py
   from enum import Enum

   class MigrationPhase(Enum):
       SHADOW_WRITE = "shadow_write"
       DUAL_READ = "dual_read"
       CUTOVER = "cutover"
       CLEANUP = "cleanup"

   class MigrationOrchestrator:
       def __init__(self, tables: list[str]):
           self.tables = tables
           self.phase = MigrationPhase.SHADOW_WRITE
           self.validation_results = {}

       def advance_phase(self) -> bool:
           """Advance to next phase if all validations pass."""
           if self.phase == MigrationPhase.SHADOW_WRITE:
               # Check: all tables have matching counts for 7+ days
               if self._all_validations_passing(min_days=7):
                   self.phase = MigrationPhase.DUAL_READ
                   return True
               return False

           elif self.phase == MigrationPhase.DUAL_READ:
               # Check: all queries return matching results
               if self._query_results_matching(min_days=7):
                   self.phase = MigrationPhase.CUTOVER
                   return True
               return False

           elif self.phase == MigrationPhase.CUTOVER:
               # Check: 24 hours with no issues post-cutover
               if self._no_issues_since_cutover(hours=24):
                   self.phase = MigrationPhase.CLEANUP
                   return True
               return False

           return False

       def rollback(self):
           """Emergency rollback to previous phase."""
           phase_order = list(MigrationPhase)
           current_idx = phase_order.index(self.phase)
           if current_idx > 0:
               self.phase = phase_order[current_idx - 1]
               self._execute_rollback()

       def status(self) -> dict:
           return {
               "phase": self.phase.value,
               "tables_total": len(self.tables),
               "tables_migrated": sum(
                   1 for t in self.tables
                   if self.validation_results.get(t, {}).get("counts_match")
               ),
               "validation_results": self.validation_results,
               "can_advance": self.advance_phase.__doc__,
           }
   ```

   ```markdown
   1. What are the top 5 products by revenue in each country?
   2. Which customers have the highest growth rate (month over month)?
   3. What percentage of revenue comes from repeat customers?

   **Part 4: Production Readiness (30 min)**
   Add ONE of the following (your choice):
   - Unit tests for your transformations
   - Logging and error handling
   - Incremental loading logic
   - A brief write-up of how you'd deploy this to production

   ### Evaluation Rubric (shared with candidate)
   | Dimension | Weight | What We Look For |
   |-----------|:------:|-----------------|
   | Correctness | 30% | Does it produce the right output? |
   | Code quality | 25% | Readable, well-structured, idiomatic |
   | Data quality | 20% | Are edge cases handled? Nulls? Dupes? |
   | Production thinking | 15% | Tests, error handling, or deployment plan |
   | Communication | 10% | README explains approach and trade-offs |
   ```

4. **Create the system design rubric**:
   ```markdown
   ## System Design Interview Rubric

   ### Prompt (given to candidate)
   "Design a real-time analytics pipeline for an e-commerce company 
   with 100K orders/day across 5 countries. The CEO wants a dashboard 
   showing live revenue by region, updated every minute."

   ### Evaluation (0-4 scale for each)

   | Dimension | 0 (No Hire) | 2 (Hire) | 4 (Strong Hire) |
   |-----------|-------------|----------|-----------------|
   | **Requirements** | Jumps to solution | Asks good questions, identifies constraints | Drives requirements, uncovers hidden needs |
   | **Architecture** | Single-box, no scaling | Reasonable components, basic flow | Scalable, handles failures, clear data flow |
   | **Trade-offs** | No trade-offs discussed | Mentions 1-2 trade-offs | Analyzes multiple options with pros/cons |
   | **Deep dive** | Superficial | Can explain one component in detail | Deep technical knowledge in multiple areas |
   | **Communication** | Unclear, disorganized | Clear explanation | Structured, engages interviewer, uses diagrams |

   ### Level Calibration
   - **L3 (Junior)**: Score 8-11 total. Reasonable architecture, some trade-offs
   - **L4 (Mid)**: Score 12-15 total. Good architecture, clear trade-offs, 1 deep dive
   - **L5 (Senior)**: Score 16-20 total. Excellent architecture, multiple trade-offs, drives discussion
   ```

5. **Design interviewer training**:
   ```markdown
   ## Interviewer Training Program

   ### Before First Interview
   1. Shadow 2 interviews conducted by an experienced interviewer
   2. Read the rubric and evaluation criteria
   3. Practice giving scores on 3 past candidate recordings
   4. Calibration session: discuss scores with hiring committee

   ### Interview Best Practices
   
   **DO:**
   - Start with a 2-minute warmup (name, role, what you work on)
   - Give the candidate the full problem upfront
   - Use hints when stuck (note: "needed hint on X" — not auto-reject)
   - Evaluate adaptability when they hit roadblocks
   - Write your scorecard within 1 hour of the interview

   **DO NOT:**
   - Ask brain teasers or trick questions
   - Ask about age, family, religion, disability, or protected status
   - Compare candidates to each other ("better than the last person")
   - Let one dimension override everything (brilliant jerk problem)
   - Change the problem mid-interview
   - Ghost the candidate — always provide timely feedback

   ### Bias Awareness
   - **Similarity bias**: Don't prefer candidates who remind you of yourself
   - **Halo effect**: Don't let one great answer inflate all scores
   - **Anchoring**: Score each dimension independently
   - **Recency bias**: Use notes, not memory, for scoring
   ```

### Common Mistakes
- Hiring for current needs only — hire for where the team needs to be in 12 months
- No rubric — gut-feel hiring leads to bias and inconsistency
- Take-home projects that take 10+ hours — you'll lose good candidates who value their time
- Not sharing the rubric with candidates — transparency builds trust and attracts better people
- Only testing technical skills — a brilliant engineer who can't collaborate is a net negative

### Stretch Goal
Run a "bar raiser" program: one interviewer in every loop is from outside the team, focused solely on maintaining the hiring bar across the organization.

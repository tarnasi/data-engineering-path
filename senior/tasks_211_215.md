   ```python
   # integration/entity_resolver.py
   import hashlib
   from dataclasses import dataclass
   from enum import Enum

   class MatchConfidence(Enum):
       EXACT = "exact"        # Same email, same entity
       HIGH = "high"          # Fuzzy match, high confidence
       LOW = "low"            # Possible match, needs review
       NO_MATCH = "no_match"  # Different entities

   @dataclass
   class EntityMatch:
       shipfast_id: str
       novamart_id: str | None
       entity_type: str
       confidence: MatchConfidence
       match_reason: str
       unified_id: str

   class EntityResolver:
       """Resolve ShipFast entities to NovaMart entities."""

       def __init__(self, novamart_conn):
           self.conn = novamart_conn
           self._id_prefix = "SF"  # ShipFast prefix for new unified IDs

       def resolve_customer(self, shipfast_customer: dict) -> EntityMatch:
           """Match ShipFast customer to NovaMart customer."""
           email = shipfast_customer.get("email", "").lower().strip()

           # Strategy 1: Exact email match
           if email:
               novamart_match = self._find_by_email(email)
               if novamart_match:
                   return EntityMatch(
                       shipfast_id=str(shipfast_customer["id"]),
                       novamart_id=novamart_match["customer_id"],
                       entity_type="customer",
                       confidence=MatchConfidence.EXACT,
                       match_reason=f"Email match: {email}",
                       unified_id=novamart_match["customer_id"],
                   )

           # Strategy 2: Name + address fuzzy match
           name = shipfast_customer.get("name", "")
           address = shipfast_customer.get("address", "")
           if name and address:
               fuzzy_match = self._fuzzy_match_customer(name, address)
               if fuzzy_match:
                   return EntityMatch(
                       shipfast_id=str(shipfast_customer["id"]),
                       novamart_id=fuzzy_match["customer_id"],
                       entity_type="customer",
                       confidence=MatchConfidence.HIGH,
                       match_reason=f"Fuzzy match: name+address",
                       unified_id=fuzzy_match["customer_id"],
                   )

           # Strategy 3: No match — create new unified ID
           unified_id = self._generate_unified_id(
               "customer", shipfast_customer["id"]
           )
           return EntityMatch(
               shipfast_id=str(shipfast_customer["id"]),
               novamart_id=None,
               entity_type="customer",
               confidence=MatchConfidence.NO_MATCH,
               match_reason="No match found — new customer",
               unified_id=unified_id,
           )

       def _generate_unified_id(self, entity_type: str, source_id) -> str:
           """Generate a deterministic unified ID for new entities."""
           raw = f"shipfast:{entity_type}:{source_id}"
           return f"{self._id_prefix}_{hashlib.sha256(raw.encode()).hexdigest()[:12]}"

       def _find_by_email(self, email: str) -> dict | None:
           result = self.conn.execute(
               "SELECT customer_id FROM gold_commerce.dim_customers "
               "WHERE LOWER(email_hash) = %s",
               (hashlib.sha256(email.encode()).hexdigest(),)
           ).fetchone()
           return {"customer_id": result[0]} if result else None

       def build_id_mapping_table(self, matches: list[EntityMatch]):
           """Create the master ID mapping table."""
           rows = [
               (m.shipfast_id, m.novamart_id, m.entity_type,
                m.confidence.value, m.match_reason, m.unified_id)
               for m in matches
           ]
           self.conn.executemany("""
               INSERT INTO integration.entity_id_mapping 
               (source_id, novamart_id, entity_type, confidence, 
                match_reason, unified_id)
               VALUES (%s, %s, %s, %s, %s, %s)
           """, rows)
   ```

   # benchmarks/runner.py
   import time
   import json
   import subprocess
   from dataclasses import dataclass, asdict
   from pathlib import Path

   @dataclass
   class BenchmarkResult:
       model_name: str
       execution_time_ms: float
       rows_produced: int
       run_timestamp: str
       git_sha: str
       branch: str
       is_incremental: bool
       warehouse: str  # "duckdb" or "snowflake"

   class BenchmarkRunner:
       """Run dbt models and measure execution time."""

       def __init__(self, project_dir: str, target: str = "benchmark"):
           self.project_dir = project_dir
           self.target = target

       def run_model(self, model_name: str, full_refresh: bool = False) -> BenchmarkResult:
           """Run a single dbt model and measure time."""
           cmd = ["dbt", "run", "--select", model_name, "--target", self.target]
           if full_refresh:
               cmd.append("--full-refresh")

           start = time.perf_counter()
           result = subprocess.run(
               cmd, cwd=self.project_dir,
               capture_output=True, text=True
           )
           elapsed_ms = (time.perf_counter() - start) * 1000

           if result.returncode != 0:
               raise RuntimeError(f"dbt run failed: {result.stderr}")

           rows = self._extract_row_count(result.stdout, model_name)
           git_sha = self._get_git_sha()
           branch = self._get_git_branch()

           return BenchmarkResult(
               model_name=model_name,
               execution_time_ms=round(elapsed_ms, 2),
               rows_produced=rows,
               run_timestamp=time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
               git_sha=git_sha,
               branch=branch,
               is_incremental=not full_refresh,
               warehouse=self.target,
           )

       def run_all_changed_models(self, base_branch: str = "main") -> list[BenchmarkResult]:
           """Run benchmarks only for models changed in this PR."""
           # Find changed dbt models
           diff_cmd = f"git diff --name-only {base_branch}...HEAD -- models/"
           result = subprocess.run(
               diff_cmd.split(), capture_output=True, text=True,
               cwd=self.project_dir
           )
           changed_files = result.stdout.strip().split("\n")
           model_names = [
               Path(f).stem for f in changed_files
               if f.endswith(".sql") and f.startswith("models/")
           ]

           results = []
           for model in model_names:
               try:
                   bench = self.run_model(model)
                   results.append(bench)
               except RuntimeError as e:
                   print(f"Skipping {model}: {e}")
           return results
   ```

3. **Build the regression detector**:
   ```python
   # benchmarks/regression_detector.py
   import json
   from pathlib import Path

   class RegressionDetector:
       """Compare benchmark results against baselines."""

       def __init__(self, baseline_path: str, threshold_pct: float = 20.0):
           self.baseline_path = Path(baseline_path)
           self.threshold_pct = threshold_pct
           self.baselines = self._load_baselines()

       def _load_baselines(self) -> dict:
           if self.baseline_path.exists():
               with open(self.baseline_path) as f:
                   return json.load(f)
           return {}

       def check(self, results: list) -> dict:
           """Check benchmarks against baselines."""
           regressions = []
           improvements = []
           new_models = []

           for result in results:
               model = result.model_name
               current_ms = result.execution_time_ms

               if model not in self.baselines:
                   new_models.append({
                       "model": model,
                       "time_ms": current_ms,
                   })
                   continue

               baseline_ms = self.baselines[model]["execution_time_ms"]
               change_pct = ((current_ms - baseline_ms) / baseline_ms) * 100

               if change_pct > self.threshold_pct:
                   regressions.append({
                       "model": model,
                       "baseline_ms": baseline_ms,
                       "current_ms": current_ms,
                       "change_pct": round(change_pct, 1),
                       "severity": "critical" if change_pct > 100 else "warning",
                   })
               elif change_pct < -self.threshold_pct:
                   improvements.append({
                       "model": model,
                       "baseline_ms": baseline_ms,
                       "current_ms": current_ms,
                       "change_pct": round(change_pct, 1),
                   })

           return {
               "pass": len(regressions) == 0,
               "regressions": regressions,
               "improvements": improvements,
               "new_models": new_models,
               "models_checked": len(results),
           }

       def update_baselines(self, results: list):
           """Update baselines with new results (run on main merge)."""
           for result in results:
               self.baselines[result.model_name] = {
                   "execution_time_ms": result.execution_time_ms,
                   "rows_produced": result.rows_produced,
                   "updated_at": result.run_timestamp,
                   "git_sha": result.git_sha,
               }
           with open(self.baseline_path, "w") as f:
               json.dump(self.baselines, f, indent=2)

       def format_pr_comment(self, check_result: dict) -> str:
           """Generate a GitHub PR comment summarizing results."""
           lines = ["## dbt Performance Benchmark Results\n"]

           if check_result["pass"]:
               lines.append("**Status: PASSED** — No performance regressions detected.\n")
           else:
               lines.append("**Status: FAILED** — Performance regressions detected!\n")

           if check_result["regressions"]:
               lines.append("### Regressions\n")
               lines.append("| Model | Baseline | Current | Change |")
               lines.append("|-------|----------|---------|--------|")
               for r in check_result["regressions"]:
                   emoji = "🔴" if r["severity"] == "critical" else "🟡"
                   lines.append(
                       f"| {emoji} `{r['model']}` | {r['baseline_ms']:.0f}ms | "
                       f"{r['current_ms']:.0f}ms | +{r['change_pct']}% |"
                   )

           if check_result["improvements"]:
               lines.append("\n### Improvements\n")
               lines.append("| Model | Baseline | Current | Change |")
               lines.append("|-------|----------|---------|--------|")
               for i in check_result["improvements"]:
                   lines.append(
                       f"| 🟢 `{i['model']}` | {i['baseline_ms']:.0f}ms | "
                       f"{i['current_ms']:.0f}ms | {i['change_pct']}% |"
                   )

           if check_result["new_models"]:
               lines.append(f"\n### New Models ({len(check_result['new_models'])})")
               lines.append("No baseline exists yet. Baseline will be set on merge.")

           return "\n".join(lines)
   ```

     benchmark:
       runs-on: ubuntu-latest

         - uses: actions/checkout@v4
           with:
             fetch-depth: 0  # Need full history for diff

         - name: Set up Python
           uses: actions/setup-python@v5
           with:
             python-version: '3.11'

         - name: Install dependencies
           run: |
             pip install dbt-duckdb pandas
             pip install -r requirements-bench.txt

         - name: Run benchmarks on changed models
           run: |
             python benchmarks/ci_runner.py \
               --base-branch main \
               --output benchmark_results.json

         - name: Check for regressions
           id: check
           run: |
             python benchmarks/ci_check.py \
               --results benchmark_results.json \
               --baselines benchmarks/baselines.json \
               --threshold 20 \
               --output pr_comment.md

         - name: Comment on PR
           if: always()
           uses: actions/github-script@v7
           with:
             script: |
               const fs = require('fs');
               const comment = fs.readFileSync('pr_comment.md', 'utf8');
               github.rest.issues.createComment({
                 owner: context.repo.owner,
                 repo: context.repo.repo,
                 issue_number: context.issue.number,
                 body: comment,
               });

         - name: Fail if regression detected
           if: steps.check.outputs.has_regression == 'true'
           run: exit 1
   ```

### Common Mistakes
- Benchmarking on different data sizes — CI uses sample data; production uses full data; results may not correlate
- Too-tight thresholds — 5% variance is normal; 20% is a reasonable starting threshold
- Not benchmarking incremental models — a full-refresh is fast but incremental may be slow (or vice versa)
- Only checking execution time — also track rows produced; a model that returns 0 rows is "fast" but broken
- Flaky benchmarks — DuckDB in CI has different performance characteristics; use relative comparison, not absolute

   """NovaMart Data Platform SDK."""
   __version__ = "1.0.0"

   # novamart-sdk/src/novamart/config.py
   import os
   from functools import lru_cache

   class Config:
       """Environment-aware configuration with sensible defaults."""

       ENVIRONMENTS = {"dev", "staging", "production"}

       def __init__(self, env: str | None = None):
           self.env = env or os.getenv("NOVAMART_ENV", "dev")
           if self.env not in self.ENVIRONMENTS:
               raise ValueError(
                   f"Invalid environment '{self.env}'. "
                   f"Must be one of: {self.ENVIRONMENTS}"
               )

       @lru_cache(maxsize=128)
       def get(self, key: str, default: str | None = None) -> str | None:
           """Get config value from environment variables."""
           # Environment-specific override first
           env_key = f"NOVAMART_{self.env.upper()}_{key}"
           value = os.getenv(env_key)
           if value:
               return value
           # Then generic key
           return os.getenv(f"NOVAMART_{key}", default)

       @property
       def snowflake_account(self) -> str:
           return self.get("SNOWFLAKE_ACCOUNT", "novamart.us-east-1")

       @property
       def kafka_brokers(self) -> str:
           return self.get("KAFKA_BROKERS", "localhost:9092")

       @property
       def is_production(self) -> bool:
           return self.env == "production"
   ```

   ```python
   # novamart-sdk/src/novamart/snowflake.py
   import snowflake.connector
   import pandas as pd
   from contextlib import contextmanager
   from .config import Config

   class SnowflakeClient:
       """Opinionated Snowflake client with connection pooling and defaults."""

       def __init__(self, config: Config | None = None,
                    warehouse: str | None = None,
                    role: str | None = None):
           self.config = config or Config()
           self.warehouse = warehouse or self.config.get(
               "SNOWFLAKE_WAREHOUSE", "WH_DEFAULT"
           )
           self.role = role or self.config.get("SNOWFLAKE_ROLE", "DATA_READER")
           self._conn = None

       @contextmanager
       def connection(self):
           """Context-managed Snowflake connection."""
           conn = snowflake.connector.connect(
               account=self.config.snowflake_account,
               user=self.config.get("SNOWFLAKE_USER"),
               password=self.config.get("SNOWFLAKE_PASSWORD"),
               warehouse=self.warehouse,
               role=self.role,
               database=self.config.get("SNOWFLAKE_DATABASE", "NOVAMART"),
           )
           try:
               yield conn
           finally:
               conn.close()

       def query(self, sql: str, params: dict | None = None) -> pd.DataFrame:
           """Execute a query and return a DataFrame."""
           with self.connection() as conn:
               if params:
                   return pd.read_sql(sql, conn, params=params)
               return pd.read_sql(sql, conn)

       def execute(self, sql: str, params: tuple | None = None):
           """Execute a statement (DDL/DML) without returning results."""
           with self.connection() as conn:
               cursor = conn.cursor()
               try:
                   cursor.execute(sql, params)
               finally:
                   cursor.close()

       def table_exists(self, schema: str, table: str) -> bool:
           """Check if a table exists."""
           result = self.query(
               "SELECT COUNT(*) as cnt FROM INFORMATION_SCHEMA.TABLES "
               "WHERE TABLE_SCHEMA = %(schema)s AND TABLE_NAME = %(table)s",
               {"schema": schema.upper(), "table": table.upper()},
           )
           return result.iloc[0]["cnt"] > 0
   ```

   # novamart-sdk/src/novamart/alerts.py
   import httpx
   import logging

   logger = logging.getLogger(__name__)

   class AlertClient:
       """Send alerts to Slack and PagerDuty."""

       def __init__(self, config=None):
           from .config import Config
           self.config = config or Config()

       def send_slack(self, channel: str, message: str,
                      severity: str = "info") -> bool:
           """Send a message to a Slack channel."""
           webhook = self.config.get("SLACK_WEBHOOK_URL")
           if not webhook:
               logger.warning("SLACK_WEBHOOK_URL not set; alert not sent")
               return False

           color_map = {
               "info": "#36a64f",
               "warning": "#ff9900",
               "critical": "#ff0000",
           }
           payload = {
               "channel": channel,
               "attachments": [{
                   "color": color_map.get(severity, "#36a64f"),
                   "title": f"[{severity.upper()}] Data Platform Alert",
                   "text": message,
                   "footer": "novamart-sdk",
               }]
           }
           resp = httpx.post(webhook, json=payload, timeout=10)
           return resp.status_code == 200

       def page(self, service_key: str, description: str,
                details: dict | None = None) -> bool:
           """Trigger a PagerDuty incident (use sparingly!)."""
           if not self.config.is_production:
               logger.info(f"PagerDuty page suppressed (env={self.config.env}): "
                          f"{description}")
               return False

           pd_url = "https://events.pagerduty.com/v2/enqueue"
           payload = {
               "routing_key": service_key,
               "event_action": "trigger",
               "payload": {
                   "summary": description,
                   "severity": "critical",
                   "source": "novamart-sdk",
                   "custom_details": details or {},
               },
           }
           resp = httpx.post(pd_url, json=payload, timeout=10)
           return resp.status_code == 202

   # novamart-sdk/src/novamart/metrics.py
   import time
   import logging
   from contextlib import contextmanager

   logger = logging.getLogger(__name__)

   class MetricsClient:
       """Record pipeline metrics for monitoring and SLA tracking."""

       def __init__(self, pipeline_name: str, config=None):
           self.pipeline_name = pipeline_name
           from .config import Config
           self.config = config or Config()
           self._metrics = {}

       def record(self, metric_name: str, value: float,
                  tags: dict | None = None):
           """Record a metric value."""
           self._metrics[metric_name] = {
               "value": value,
               "timestamp": time.time(),
               "pipeline": self.pipeline_name,
               "tags": tags or {},
           }
           logger.info(f"Metric: {self.pipeline_name}.{metric_name} = {value}")

       @contextmanager
       def timer(self, metric_name: str):
           """Time a block of code and record as a metric."""
           start = time.perf_counter()
           yield
           elapsed_ms = (time.perf_counter() - start) * 1000
           self.record(metric_name, elapsed_ms, {"unit": "ms"})

       def publish(self):
           """Push all recorded metrics to the metrics store."""
           if not self._metrics:
               return
           # Push to Snowflake metrics table
           from .snowflake import SnowflakeClient
           sf = SnowflakeClient(self.config)
           for name, data in self._metrics.items():
               sf.execute(
                   "INSERT INTO platform_metrics.pipeline_metrics "
                   "(pipeline_name, metric_name, metric_value, recorded_at) "
                   "VALUES (%s, %s, %s, CURRENT_TIMESTAMP())",
                   (self.pipeline_name, name, data["value"]),
               )
           logger.info(
               f"Published {len(self._metrics)} metrics for {self.pipeline_name}"
           )
           self._metrics.clear()
   ```

   ## Technology Evaluation: LLM-Powered Analytics

   ### Problem It Solves
   Business users submit 15-20 ad-hoc data requests per week to the 
   analytics team. Most are simple aggregations that could be self-serve 
   if users could "ask questions in English" instead of writing SQL.

   ### What We're Evaluating

   ```python
   # poc/text_to_sql.py
   from langchain_community.utilities import SQLDatabase
   from langchain_openai import ChatOpenAI

   class AnalyticsAssistant:
       def __init__(self, snowflake_uri: str):
           self.db = SQLDatabase.from_uri(
               snowflake_uri,
               include_tables=[
                   "fact_orders", "dim_customers", "dim_products",
                   "mart_daily_revenue", "mart_customer_360"
               ],
               sample_rows_in_table_info=3,
           )
           self.llm = ChatOpenAI(model="gpt-4", temperature=0)

       def ask(self, question: str) -> dict:
           """Convert natural language question to SQL and execute."""
           # Generate SQL
           prompt = f"""Given the following database schema:
           {self.db.get_table_info()}
           
           Generate a SQL query to answer: {question}
           
           Rules:
           - Use only the tables listed above
           - Always include appropriate WHERE clauses
           - Limit results to 100 rows maximum
           - Do not modify data (SELECT only)
           - Use CURRENT_DATE() for date references
           """
           
           sql = self.llm.invoke(prompt).content
           
           # Safety check: only allow SELECT
           if not sql.strip().upper().startswith("SELECT"):
               return {"error": "Generated query is not a SELECT statement"}
           
           # Execute
           result = self.db.run(sql)
           
           return {
               "question": question,
               "sql": sql,
               "result": result,
               "confidence": "high" if "fact_orders" in sql else "medium",
           }
   ```

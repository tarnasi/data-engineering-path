## Task 256: Advanced Testing Strategy — Contract, Property-Based, and Chaos Testing

### Business Context
The retro (Task 255) identified testing as the biggest capability gap (score: 3/5). Currently NovaMart has dbt tests (not null, unique, accepted values) and some integration tests. But: (1) No contract tests between services, (2) No property-based tests that catch edge cases humans wouldn't think of, (3) No end-to-end data flow tests. A recent incident: the orders API changed a field from camelCase to snake_case. The dbt pipeline ingested it silently, creating null values in GOLD. A contract test would have caught this instantly.

### Objective
Build a comprehensive testing strategy covering all layers: unit tests, contract tests, property-based tests, integration tests, and end-to-end data flow tests.

### Technical Requirements
- **Tools**: Python (pytest, hypothesis), dbt tests, JSON Schema, Great Expectations
- **Skills**: Testing pyramid, contract testing, property-based testing, test automation

   ```python
   # testing/contract_tests.py
   import json
   import jsonschema
   from dataclasses import dataclass

   @dataclass
   class DataContract:
       name: str
       version: str
       producer: str
       consumer: str
       schema: dict  # JSON Schema
       sla: dict     # Freshness, quality guarantees

   class ContractTestSuite:
       """Test data contracts between producers and consumers."""

       def __init__(self):
           self.contracts = {}

       def register_contract(self, contract: DataContract):
           self.contracts[contract.name] = contract

       def validate_record(self, contract_name: str,
                           record: dict) -> dict:
           """Validate a single record against its contract."""
           contract = self.contracts.get(contract_name)
           if not contract:
               return {"valid": False, "error": "Contract not found"}

           try:
               jsonschema.validate(record, contract.schema)
               return {"valid": True, "contract": contract_name}
           except jsonschema.ValidationError as e:
               return {
                   "valid": False,
                   "contract": contract_name,
                   "error": e.message,
                   "path": list(e.path),
               }

       def validate_batch(self, contract_name: str,
                          records: list[dict]) -> dict:
           """Validate a batch of records."""
           results = [self.validate_record(contract_name, r) for r in records]
           failures = [r for r in results if not r["valid"]]

           return {
               "contract": contract_name,
               "total_records": len(records),
               "passed": len(records) - len(failures),
               "failed": len(failures),
               "pass_rate": round(
                   (len(records) - len(failures)) / max(len(records), 1) * 100, 2
               ),
               "failures": failures[:10],  # First 10 failures for debugging
           }

   # Define NovaMart's core data contracts
   ORDERS_API_CONTRACT = DataContract(
       name="orders_api_v2",
       version="2.0",
       producer="Commerce API",
       consumer="Data Platform (CDC)",
       schema={
           "type": "object",
           "required": ["order_id", "customer_id", "total_amount",
                        "currency", "status", "created_at"],
           "properties": {
               "order_id": {"type": "string", "pattern": "^ORD-[0-9]+$"},
               "customer_id": {"type": "string", "pattern": "^CUS-[0-9]+$"},
               "total_amount": {"type": "number", "minimum": 0},
               "currency": {"type": "string", "enum": ["USD", "EUR", "GBP",
                            "JPY", "AUD", "CAD", "BRL", "MXN", "INR"]},
               "status": {"type": "string", "enum": ["pending", "confirmed",
                          "shipped", "delivered", "cancelled", "refunded"]},
               "created_at": {"type": "string", "format": "date-time"},
               "items": {
                   "type": "array",
                   "minItems": 1,
                   "items": {
                       "type": "object",
                       "required": ["sku", "quantity", "unit_price"],
                       "properties": {
                           "sku": {"type": "string"},
   ```

3. **Build property-based tests**:
   ```python
   # testing/property_tests.py
   from hypothesis import given, strategies as st, settings

   # Property: Revenue should always be >= 0
   @given(
       quantity=st.integers(min_value=1, max_value=10000),
       unit_price=st.floats(min_value=0, max_value=100000,
                             allow_nan=False, allow_infinity=False),
       discount_pct=st.floats(min_value=0, max_value=100,
                               allow_nan=False, allow_infinity=False),
   )
   @settings(max_examples=1000)
   def test_revenue_calculation_never_negative(quantity, unit_price,
                                                discount_pct):
       """Revenue = quantity * unit_price * (1 - discount/100). Always >= 0."""
       revenue = quantity * unit_price * (1 - discount_pct / 100)
       assert revenue >= 0, (
           f"Negative revenue: {quantity} * {unit_price} * "
           f"(1 - {discount_pct}/100) = {revenue}"
       )

   # Property: Data deduplication should be idempotent
   @given(
       records=st.lists(
           st.fixed_dictionaries({
               "id": st.integers(min_value=1, max_value=100),
               "value": st.text(min_size=1, max_size=50),
               "updated_at": st.datetimes(),
           }),
           min_size=1,
           max_size=100,
       )
   )
   def test_dedup_is_idempotent(records):
       """Deduplicating twice should give the same result as once."""
       def dedup(recs):
           seen = {}
           for r in recs:
               key = r["id"]
               if key not in seen or r["updated_at"] > seen[key]["updated_at"]:
                   seen[key] = r
           return sorted(seen.values(), key=lambda x: x["id"])

       once = dedup(records)
       twice = dedup(once)
       assert once == twice, "Dedup is not idempotent"

   # Property: Aggregations should preserve row counts
   @given(
       rows=st.lists(
           st.fixed_dictionaries({
               "category": st.sampled_from(["A", "B", "C"]),
               "amount": st.floats(min_value=0, max_value=1e6,
                                    allow_nan=False, allow_infinity=False),
           }),
           min_size=1,
           max_size=1000,
       )
   )
   def test_aggregation_preserves_total(rows):
       """Sum of group sums should equal total sum."""
       total = sum(r["amount"] for r in rows)

       groups = {}
       for r in rows:
           groups.setdefault(r["category"], 0)
           groups[r["category"]] += r["amount"]

       group_total = sum(groups.values())
       assert abs(total - group_total) < 0.01, (
           f"Total {total} != group sum {group_total}"
       )
   ```

4. **Build end-to-end data flow tests**:

   ```python
   # testing/e2e_tests.py
   class E2EDataFlowTest:
       """End-to-end tests that verify complete data flows."""

       def __init__(self, source_conn, snowflake_conn, kafka_producer):
           self.source = source_conn
           self.sf = snowflake_conn
           self.kafka = kafka_producer

       def test_order_flow(self) -> dict:
           """Test: Order created in source → visible in GOLD within SLA."""
           import time
           import uuid

           # Step 1: Insert a test order in the source system
           test_order_id = f"TEST-{uuid.uuid4().hex[:8].upper()}"
           self.kafka.produce("orders.raw", {
               "order_id": test_order_id,
               "customer_id": "CUS-TEST-001",
               "total_amount": 99.99,
               "currency": "USD",
               "status": "confirmed",
               "created_at": time.strftime("%Y-%m-%dT%H:%M:%SZ"),
               "items": [{"sku": "TEST-SKU", "quantity": 1,
                          "unit_price": 99.99}],
               "_test": True,  # Flag for cleanup
           })

           # Step 2: Wait for pipeline to process (with timeout)
           max_wait = 300  # 5 minutes
           poll_interval = 15
           elapsed = 0
           found = False

           while elapsed < max_wait:
               time.sleep(poll_interval)
               elapsed += poll_interval

               result = self.sf.cursor().execute(
                   "SELECT COUNT(*) FROM GOLD.FACT_ORDERS "
                   "WHERE order_id = %s", (test_order_id,)
               ).fetchone()

               if result and result[0] > 0:
                   found = True
                   break

           # Step 3: Verify data integrity
           if found:
               row = self.sf.cursor().execute(
                   "SELECT total_amount_usd, status "
                   "FROM GOLD.FACT_ORDERS WHERE order_id = %s",
                   (test_order_id,)
               ).fetchone()

               return {
                   "test": "order_flow_e2e",
                   "passed": row[0] == 99.99 and row[1] == "confirmed",
                   "latency_seconds": elapsed,
                   "within_sla": elapsed <= 120,  # 2-minute SLA
               }

           return {
               "test": "order_flow_e2e",
               "passed": False,
               "error": f"Order not found in GOLD after {max_wait}s",
           }

       def cleanup_test_data(self):
           """Remove test records from all layers."""
           self.sf.cursor().execute(
               "DELETE FROM GOLD.FACT_ORDERS WHERE order_id LIKE 'TEST-%'"
           )
           self.sf.cursor().execute(
               "DELETE FROM STAGING.STG_ORDERS WHERE order_id LIKE 'TEST-%'"
           )
   ```

### Common Mistakes
- Only dbt tests — dbt tests check the warehouse layer; contract tests catch issues at ingestion before they reach the warehouse

- Property tests without invariants — "it doesn't crash" is not a useful property; define business invariants
- No test environment — tests should run against staging, never production

   ```python
   # data_products/api.py
   from fastapi import FastAPI, Depends, HTTPException
   from datetime import datetime

   app = FastAPI(title="NovaMart Seller Analytics API")

   class UsageMeter:
       """Track and enforce API usage limits."""

       TIER_LIMITS = {
           "basic": {"api_calls_per_day": 100, "endpoints": ["sales", "ratings"]},
           "professional": {
               "api_calls_per_day": 1000,
               "endpoints": ["sales", "ratings", "benchmarks", "demographics"],
           },
           "enterprise": {
               "api_calls_per_day": 10000,
               "endpoints": ["sales", "ratings", "benchmarks", "demographics",
                            "forecasting", "pricing", "custom"],
           },
       }

       def __init__(self, db_conn):
           self.db = db_conn

       def check_access(self, seller_id: str, endpoint: str) -> dict:
           """Check if seller has access to this endpoint."""
           tier = self._get_seller_tier(seller_id)
           limits = self.TIER_LIMITS[tier]

           if endpoint not in limits["endpoints"]:
               return {"allowed": False, "reason": f"Upgrade to access {endpoint}"}

           usage_today = self._get_usage_today(seller_id)
           if usage_today >= limits["api_calls_per_day"]:
               return {"allowed": False, "reason": "Daily limit reached"}

           self._record_usage(seller_id, endpoint)
           return {"allowed": True, "remaining": limits["api_calls_per_day"] - usage_today - 1}

       def _get_seller_tier(self, seller_id: str) -> str:
           result = self.db.execute(
               "SELECT tier FROM seller_subscriptions WHERE seller_id = %s",
               (seller_id,)
           ).fetchone()
           return result[0] if result else "basic"

       def _get_usage_today(self, seller_id: str) -> int:
           result = self.db.execute(
               "SELECT COUNT(*) FROM api_usage_log "
               "WHERE seller_id = %s AND DATE(called_at) = CURRENT_DATE",
               (seller_id,)
           ).fetchone()
           return result[0] if result else 0

       def _record_usage(self, seller_id: str, endpoint: str):
           self.db.execute(
               "INSERT INTO api_usage_log (seller_id, endpoint, called_at) "
               "VALUES (%s, %s, %s)",
               (seller_id, endpoint, datetime.utcnow())
           )

   @app.get("/api/v1/sellers/{seller_id}/sales")
   ```

   ```python
   # data_products/revenue_tracker.py
   @dataclass
   class DataProductRevenue:
       month: str
       tier: str
       seller_count: int
       mrr: float  # Monthly Recurring Revenue
       api_calls: int
       cost_to_serve: float
       gross_margin: float

   class DataProductRevenueTracker:
       def __init__(self, db_conn):
           self.db = db_conn

       def calculate_monthly_revenue(self, month: str) -> dict:
           """Calculate data product revenue for a given month."""
           result = self.db.execute("""
               SELECT
                   tier,
                   COUNT(DISTINCT seller_id) AS sellers,
                   SUM(monthly_price) AS mrr
               FROM seller_subscriptions
               WHERE subscription_month = %s AND status = 'active'
               GROUP BY tier
           """, (month,)).fetchall()

           tiers = {}
           total_mrr = 0
           for row in result:
               tier_name, count, mrr = row
               tiers[tier_name] = {"sellers": count, "mrr": mrr}
               total_mrr += mrr

           # Estimate cost to serve
           api_calls = self._total_api_calls(month)
           cost_per_call = 0.001  # $0.001 per API call (compute + storage)
           cost_to_serve = api_calls * cost_per_call + 500  # Base infra cost

           return {
               "month": month,
               "total_mrr": total_mrr,
               "tiers": tiers,
               "api_calls": api_calls,
               "cost_to_serve": round(cost_to_serve, 2),
               "gross_margin_pct": round(
                   (total_mrr - cost_to_serve) / max(total_mrr, 1) * 100, 1
               ),
               "arr_projection": round(total_mrr * 12, 2),
           }

       def _total_api_calls(self, month: str) -> int:
           result = self.db.execute(
               "SELECT COUNT(*) FROM api_usage_log "
               "WHERE TO_CHAR(called_at, 'YYYY-MM') = %s",
               (month,)
           ).fetchone()
           return result[0] if result else 0
   ```

2. **Build the deprecation workflow**:
   ```python
   # deprecation/workflow.py
   class DeprecationWorkflow:
       """Manage the lifecycle of system deprecation."""

       PHASE_ORDER = [
           DeprecationPhase.IDENTIFIED,
           DeprecationPhase.ANNOUNCED,
           DeprecationPhase.MIGRATION,
           DeprecationPhase.READ_ONLY,
           DeprecationPhase.SUNSET,
           DeprecationPhase.DECOMMISSIONED,
       ]

       def advance_phase(self, system: LegacySystem) -> dict:
           """Move a system to the next deprecation phase."""
           current_idx = self.PHASE_ORDER.index(system.phase)
           if current_idx >= len(self.PHASE_ORDER) - 1:
               return {"error": "Already fully decommissioned"}

           next_phase = self.PHASE_ORDER[current_idx + 1]

           # Pre-flight checks for each phase transition
           checks = self._pre_flight_checks(system, next_phase)
           if not checks["passed"]:
               return {"error": f"Pre-flight check failed: {checks['reason']}"}

           system.phase = next_phase
           return {
               "system": system.name,
               "new_phase": next_phase.value,
               "next_steps": self._next_steps(system, next_phase),
           }

       def _pre_flight_checks(self, system: LegacySystem,
                               target_phase: DeprecationPhase) -> dict:
           if target_phase == DeprecationPhase.ANNOUNCED:
               return {"passed": True, "reason": ""}

           if target_phase == DeprecationPhase.MIGRATION:
               # Verify replacement system is ready
               return {"passed": True, "reason": ""}

           if target_phase == DeprecationPhase.READ_ONLY:
               # Verify no active writers
               if system.users:
                   return {
                       "passed": False,
                       "reason": f"Active users: {system.users}",
                   }
               return {"passed": True, "reason": ""}

           if target_phase == DeprecationPhase.SUNSET:
               # Verify no queries in 30 days
               return {"passed": True, "reason": ""}

           if target_phase == DeprecationPhase.DECOMMISSIONED:
               # Final check: backup exists, no dependencies
               if system.dependencies:
                   return {
                       "passed": False,
                       "reason": f"Dependencies remain: {system.dependencies}",
                   }
               return {"passed": True, "reason": ""}

           return {"passed": True, "reason": ""}

       def _next_steps(self, system: LegacySystem,
                       phase: DeprecationPhase) -> list[str]:
           steps = {
               DeprecationPhase.ANNOUNCED: [
                   f"Notify users: {system.users}",
                   "Set deprecation date (usually 90 days from now)",
                   f"Document migration path to: {system.replacement}",
               ],
               DeprecationPhase.MIGRATION: [
                   "Migrate users to replacement system",
                   "Run both systems in parallel for validation",
                   "Compare outputs for consistency",
               ],
               DeprecationPhase.READ_ONLY: [
                   "Disable write access",
                   "Monitor for access attempts",
                   "Keep running for 30 days",
               ],
               DeprecationPhase.SUNSET: [
                   "Stop scheduled jobs",
                   "Create final backup",
                   "Keep infrastructure for 30 days (rollback)",
               ],
               DeprecationPhase.DECOMMISSIONED: [
                   "Delete infrastructure",
                   "Remove monitoring alerts",
                   "Archive documentation",
                   f"Monthly savings: ${system.monthly_cost}",
               ],
           }
           return steps.get(phase, [])

       def generate_communication(self, system: LegacySystem) -> str:
           """Generate deprecation announcement."""
           return f"""
   Subject: [Action Required] {system.name} Deprecation Notice

   Hi team,

   We are deprecating **{system.name}** as part of our platform
   simplification effort.

   **What**: {system.description}
   **Why**: Replaced by {system.replacement}
   **When**: Migration window: 90 days from this notice
   **Impact**: {len(system.users)} users need to migrate
   **Cost savings**: ${system.monthly_cost}/month

   **Action required**: Please migrate your queries/workflows
   to {system.replacement} by [DATE].

   We will provide migration support. Reply to this email or
   reach out in #data-platform-support.

   — Data Platform Team
   """
   ```

- Deleting without warning — always announce, provide migration time, and keep rollback capability

   ```python
   # kafka_ops/cluster_manager.py
   from confluent_kafka.admin import (
       AdminClient, NewTopic, NewPartitions, ConfigResource,
       ResourceType,
   )
   from dataclasses import dataclass

   @dataclass
   class TopicConfig:
       name: str
       partitions: int
       replication_factor: int
       retention_ms: int
       cleanup_policy: str  # "delete" or "compact"
       min_insync_replicas: int
       owner_team: str
       tier: str  # "critical", "standard", "development"

   class KafkaClusterManager:
       TIER_DEFAULTS = {
           "critical": {
               "replication_factor": 3,
               "min_insync_replicas": 2,
               "retention_ms": 7 * 24 * 3600 * 1000,  # 7 days
           },
           "standard": {
               "replication_factor": 3,
               "min_insync_replicas": 2,
               "retention_ms": 3 * 24 * 3600 * 1000,  # 3 days
           },
           "development": {
               "replication_factor": 2,
               "min_insync_replicas": 1,
               "retention_ms": 1 * 24 * 3600 * 1000,  # 1 day
           },
       }

       def __init__(self, bootstrap_servers: str):
           self.admin = AdminClient({
               "bootstrap.servers": bootstrap_servers,
           })

       def create_topic(self, config: TopicConfig) -> dict:
           """Create a topic with governance-compliant configuration."""
           # Apply tier defaults
           tier_config = self.TIER_DEFAULTS[config.tier]

           # Naming convention enforcement
           if not self._validate_topic_name(config.name, config.owner_team):
               return {
                   "error": f"Topic name must follow pattern: "
                   f"{config.owner_team}.domain.event_name"
               }

           topic = NewTopic(
               config.name,
               num_partitions=config.partitions,
               replication_factor=tier_config["replication_factor"],
               config={
                   "retention.ms": str(tier_config["retention_ms"]),
                   "cleanup.policy": config.cleanup_policy,
                   "min.insync.replicas": str(
                       tier_config["min_insync_replicas"]
                   ),
               },
           )

           futures = self.admin.create_topics([topic])
           for name, future in futures.items():
               try:
                   future.result()
                   return {"topic": name, "status": "created"}
               except Exception as e:
                   return {"topic": name, "error": str(e)}

       def get_cluster_health(self) -> dict:
           """Get overall cluster health metrics."""
           metadata = self.admin.list_topics(timeout=10)

           topics = {}
           total_partitions = 0
           under_replicated = 0

           for topic_name, topic_meta in metadata.topics.items():
               if topic_name.startswith("__"):
                   continue  # Skip internal topics
               partition_count = len(topic_meta.partitions)
               total_partitions += partition_count

               for pid, partition in topic_meta.partitions.items():
                   if len(partition.isrs) < len(partition.replicas):
                       under_replicated += 1

               topics[topic_name] = {
                   "partitions": partition_count,
                   "healthy": all(
                       len(p.isrs) == len(p.replicas)
                       for p in topic_meta.partitions.values()
                   ),
               }

           return {
               "broker_count": len(metadata.brokers),
               "topic_count": len(topics),
               "total_partitions": total_partitions,
               "under_replicated_partitions": under_replicated,
               "healthy": under_replicated == 0,
           }

       def _validate_topic_name(self, name: str, team: str) -> bool:
           """Enforce naming convention: team.domain.event_name."""
           parts = name.split(".")
           return len(parts) >= 3 and parts[0] == team
   ```

2. **Build consumer group monitoring**:
   ```python
   # kafka_ops/consumer_monitor.py
   from confluent_kafka.admin import AdminClient

   @dataclass
   class ConsumerGroupProfile:
       group_id: str
       processing_type: str  # "real-time", "micro-batch", "batch"
       expected_lag_range: tuple[int, int]  # (min, max) acceptable lag
       owner_team: str
       alert_threshold: int

   class ConsumerGroupMonitor:
       """Intelligent consumer group monitoring with profile-aware alerting."""

       def __init__(self, admin_client: AdminClient):
           self.admin = admin_client
           self.profiles = {}

       def register_profile(self, profile: ConsumerGroupProfile):
           self.profiles[profile.group_id] = profile

       def check_consumer_health(self) -> list[dict]:
           """Check all consumer groups against their profiles."""
           groups = self.admin.list_consumer_groups()
           alerts = []

           for group in groups.result().valid:
               group_id = group.group_id
               if group_id.startswith("__"):
                   continue

               lag = self._get_consumer_lag(group_id)
               profile = self.profiles.get(group_id)

               if profile:
                   # Profile-aware alerting
                   if lag > profile.alert_threshold:
                       alerts.append({
                           "group": group_id,
                           "lag": lag,
                           "threshold": profile.alert_threshold,
                           "type": profile.processing_type,
                           "severity": self._severity(lag, profile),
                       })
               else:
                   # Unknown consumer group — always alert
                   if lag > 10000:
                       alerts.append({
                           "group": group_id,
                           "lag": lag,
                           "threshold": "unknown (no profile registered)",
                           "severity": "warning",
                       })

           return alerts

       def _get_consumer_lag(self, group_id: str) -> int:
           """Calculate total lag for a consumer group."""
           try:
               offsets = self.admin.list_consumer_group_offsets(
                   [group_id]
               )
               # Sum lag across all partitions
               total_lag = 0
               for topic_partition, offset_meta in offsets.items():
                   # Compare committed offset to log end offset
                   total_lag += max(0, offset_meta.offset)
               return total_lag
           except Exception:
               return -1

       def _severity(self, lag: int, profile: ConsumerGroupProfile) -> str:
           if lag > profile.alert_threshold * 5:
               return "critical"
           elif lag > profile.alert_threshold * 2:
               return "high"
           return "medium"
   ```

3. **Build multi-tenant resource governance**:
   ```python
   # kafka_ops/governance.py
   class KafkaGovernance:
       """Multi-tenant governance for shared Kafka cluster."""

       TEAM_QUOTAS = {
           "commerce": {
               "max_topics": 30,
               "max_partitions": 300,
               "max_produce_bytes_per_sec": 50 * 1024 * 1024,  # 50 MB/s
               "max_consume_bytes_per_sec": 100 * 1024 * 1024,  # 100 MB/s
           },
           "logistics": {
               "max_topics": 15,
               "max_partitions": 150,
               "max_produce_bytes_per_sec": 30 * 1024 * 1024,
               "max_consume_bytes_per_sec": 60 * 1024 * 1024,
           },
           "marketing": {
               "max_topics": 10,
               "max_partitions": 100,
               "max_produce_bytes_per_sec": 10 * 1024 * 1024,
               "max_consume_bytes_per_sec": 20 * 1024 * 1024,
           },
       }

       def __init__(self, cluster_manager: KafkaClusterManager):
           self.cluster = cluster_manager

       def check_quota(self, team: str, action: str,
                       config: TopicConfig = None) -> dict:
           """Check if a team's action is within quota."""
           quota = self.TEAM_QUOTAS.get(team)
           if not quota:
               return {"allowed": False, "reason": f"Unknown team: {team}"}

           current = self._get_team_usage(team)

           if action == "create_topic" and config:
               if current["topics"] >= quota["max_topics"]:
                   return {
                       "allowed": False,
                       "reason": f"Topic limit reached: {current['topics']}/{quota['max_topics']}",
                   }
               new_partitions = current["partitions"] + config.partitions
               if new_partitions > quota["max_partitions"]:
                   return {
                       "allowed": False,
                       "reason": f"Partition limit would be exceeded: {new_partitions}/{quota['max_partitions']}",
                   }

           return {"allowed": True, "usage": current, "quota": quota}

       def _get_team_usage(self, team: str) -> dict:
           """Get current resource usage for a team."""
           health = self.cluster.get_cluster_health()
           team_topics = [
               t for t in health.get("topics", {})
               if t.startswith(f"{team}.")
           ]
           return {
               "topics": len(team_topics),
               "partitions": sum(
                   health.get("topics", {}).get(t, {}).get("partitions", 0)
                   for t in team_topics
               ),
           }
   ```

   ```python
   # strategy/investment_model.py
   @dataclass
   class StrategyInitiative:
       name: str
       pillar: str
       investment_usd: float        # Total cost
       timeline_quarters: int
       expected_value_usd: float    # Expected annual return
       value_type: str              # "revenue", "savings", "risk_reduction"
       confidence: float            # 0-1
       dependencies: list[str]

   class StrategyInvestmentModel:
       def __init__(self, initiatives: list[StrategyInitiative]):
           self.initiatives = initiatives

       def portfolio_analysis(self) -> dict:
           """Analyze the strategic investment portfolio."""
           total_investment = sum(i.investment_usd for i in self.initiatives)
           total_value = sum(
               i.expected_value_usd * i.confidence for i in self.initiatives
           )

           by_pillar = {}
           for i in self.initiatives:
               if i.pillar not in by_pillar:
                   by_pillar[i.pillar] = {
                       "investment": 0, "value": 0, "count": 0
                   }
               by_pillar[i.pillar]["investment"] += i.investment_usd
               by_pillar[i.pillar]["value"] += (
                   i.expected_value_usd * i.confidence
               )
               by_pillar[i.pillar]["count"] += 1

           return {
               "total_investment_2yr": total_investment,
               "expected_annual_value": round(total_value, 2),
               "portfolio_roi": round(total_value / max(total_investment, 1), 2),
               "by_pillar": by_pillar,
               "payback_months": round(
                   total_investment / max(total_value / 12, 1), 1
               ),
           }

       def generate_executive_summary(self) -> str:
           analysis = self.portfolio_analysis()
           summary = "# Data Platform 2-Year Strategy — Executive Summary\n\n"
           summary += f"**Total Investment**: ${analysis['total_investment_2yr']:,.0f}\n"
           summary += f"**Expected Annual Return**: ${analysis['expected_annual_value']:,.0f}\n"
           summary += f"**Portfolio ROI**: {analysis['portfolio_roi']}x\n"
           summary += f"**Payback Period**: {analysis['payback_months']} months\n\n"

           summary += "## Investment by Strategic Pillar\n\n"
           summary += "| Pillar | Investment | Expected Value | Initiatives |\n"
   ```

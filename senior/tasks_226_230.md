   ```python
   # stream_quality/validator.py
   from dataclasses import dataclass, field
   from enum import Enum
   from typing import Callable
   import time
   import json

   class Severity(Enum):
       WARNING = "warning"      # Log but pass through
       ERROR = "error"          # Send to DLQ
       CRITICAL = "critical"    # Send to DLQ + alert

   @dataclass
   class QualityRule:
       name: str
       description: str
       severity: Severity
       check: Callable[[dict], bool]  # Returns True if valid

   @dataclass
   class ValidationResult:
       record: dict
       is_valid: bool
       failures: list[str] = field(default_factory=list)
       warnings: list[str] = field(default_factory=list)
       validation_time_ms: float = 0

   class StreamQualityGate:
       def __init__(self, rules: list[QualityRule]):
           self.rules = rules
           self._metrics = {
               "total": 0, "passed": 0, "failed": 0,
               "warnings": 0, "dlq": 0,
           }

       def validate(self, record: dict) -> ValidationResult:
           start = time.monotonic()
           failures = []
           warnings = []

           for rule in self.rules:
               try:
                   if not rule.check(record):
                       if rule.severity == Severity.WARNING:
                           warnings.append(rule.name)
                       else:
                           failures.append(rule.name)
               except Exception as e:
                   failures.append(f"{rule.name}: exception {e}")

           elapsed = (time.monotonic() - start) * 1000
           is_valid = len(failures) == 0

           self._metrics["total"] += 1
           if is_valid:
               self._metrics["passed"] += 1
           else:
               self._metrics["failed"] += 1
               self._metrics["dlq"] += 1
           if warnings:
               self._metrics["warnings"] += 1

           return ValidationResult(
               record=record,
               is_valid=is_valid,
               failures=failures,
               warnings=warnings,
               validation_time_ms=elapsed,
           )

       @property
       def pass_rate(self) -> float:
           if self._metrics["total"] == 0:
               return 1.0
           return self._metrics["passed"] / self._metrics["total"]

   # Define rules for order events
   ORDER_RULES = [
       QualityRule(
           name="schema_order_id_present",
           description="order_id must be present and non-empty",
           severity=Severity.ERROR,
           check=lambda r: bool(r.get("order_id")),
       ),
       QualityRule(
           name="amount_positive",
           description="Order amount must be positive",
           severity=Severity.CRITICAL,
           check=lambda r: isinstance(r.get("amount"), (int, float))
           and r["amount"] > 0,
       ),
       QualityRule(
           name="currency_valid",
           description="Currency must be a valid ISO 4217 code",
           severity=Severity.ERROR,
           check=lambda r: r.get("currency") in {
               "USD", "GBP", "EUR", "BRL", "MXN", "JPY", "CAD", "AUD"
           },
       ),
       QualityRule(
           name="timestamp_recent",
           description="Event timestamp within last 24 hours",
           severity=Severity.WARNING,
           check=lambda r: (
               abs(time.time() - r.get("event_time", 0)) < 86400
           ),
       ),
       QualityRule(
           name="region_valid",
           description="Region must be a known NovaMart region",
           severity=Severity.ERROR,
           check=lambda r: r.get("region") in {
               "US", "UK", "DE", "FR", "ES", "IT", "PL", "BR", "MX", "JP"
   ```

3. **Build the circuit breaker for bad data floods**:
   ```python
   # stream_quality/circuit_breaker.py
   from collections import deque
   from datetime import datetime, timezone
   from enum import Enum
   import time

   class CircuitState(Enum):
       CLOSED = "closed"       # Normal operation
       OPEN = "open"           # Blocking all records
       HALF_OPEN = "half_open" # Testing if source recovered

   class DataCircuitBreaker:
       """Stop processing if error rate exceeds threshold.
       
       Prevents a bad source deployment from contaminating
       the entire downstream pipeline for hours.
       """

       def __init__(self, error_threshold_pct: float = 10.0,
                    window_seconds: int = 300,
                    min_samples: int = 100,
                    cooldown_seconds: int = 60):
           self.error_threshold_pct = error_threshold_pct
           self.window_seconds = window_seconds
           self.min_samples = min_samples
           self.cooldown_seconds = cooldown_seconds
           self.state = CircuitState.CLOSED
           self._results: deque = deque()
           self._opened_at: float | None = None

       def record_result(self, is_valid: bool):
           now = time.monotonic()
           self._results.append((now, is_valid))
           # Trim old results
           cutoff = now - self.window_seconds
           while self._results and self._results[0][0] < cutoff:
               self._results.popleft()

           self._evaluate()

       def _evaluate(self):
           if self.state == CircuitState.OPEN:
               # Check if cooldown has passed
               if (time.monotonic() - self._opened_at) > self.cooldown_seconds:
                   self.state = CircuitState.HALF_OPEN
               return

           if len(self._results) < self.min_samples:
               return  # Not enough data to judge

           errors = sum(1 for _, valid in self._results if not valid)
           error_pct = (errors / len(self._results)) * 100

           if error_pct > self.error_threshold_pct:
               self.state = CircuitState.OPEN
               self._opened_at = time.monotonic()

       def should_process(self) -> bool:
           if self.state == CircuitState.CLOSED:
               return True
           if self.state == CircuitState.HALF_OPEN:
               return True  # Let a few through to test
           return False  # OPEN — block everything

       def status(self) -> dict:
           total = len(self._results)
           errors = sum(1 for _, v in self._results if not v)
           return {
               "state": self.state.value,
               "window_records": total,
               "error_count": errors,
               "error_pct": round(errors / max(total, 1) * 100, 2),
               "threshold_pct": self.error_threshold_pct,
           }
   ```

4. **Build the DLQ processor**:
   ```python
   # stream_quality/dlq_processor.py
   from datetime import datetime, timezone

   class DLQProcessor:
       """Process records from the Dead Letter Queue."""

       def __init__(self, kafka_consumer, snowflake_conn):
           self.consumer = kafka_consumer
           self.conn = snowflake_conn

       def process_dlq_batch(self, topic: str, batch_size: int = 1000):
           """Read DLQ records, classify, and store for analysis."""
           records = self.consumer.poll(max_records=batch_size)
           classified = {"recoverable": [], "poison": [], "expired": []}

           for record in records:
               failure_reasons = record.headers.get("quality_failures", "")
               original_time = record.headers.get("original_timestamp", "")

               if self._is_recoverable(failure_reasons):
                   classified["recoverable"].append(record)
               elif self._is_expired(original_time):
                   classified["expired"].append(record)
               else:
                   classified["poison"].append(record)

           # Store in Snowflake for analysis
           self._store_dlq_records(classified)
           return {k: len(v) for k, v in classified.items()}

       def _is_recoverable(self, reasons: str) -> bool:
           """Some failures can be fixed and replayed."""
           recoverable_reasons = {"timestamp_recent", "currency_valid"}
           record_reasons = set(reasons.split(","))
           return record_reasons.issubset(recoverable_reasons)

       def _is_expired(self, timestamp: str) -> bool:
           """Records older than 24h aren't worth recovering."""
           if not timestamp:
               return True
           try:
               original = datetime.fromisoformat(timestamp)
               age_hours = (
                   datetime.now(timezone.utc) - original
               ).total_seconds() / 3600
               return age_hours > 24
           except (ValueError, TypeError):
               return True

       def _store_dlq_records(self, classified: dict):
           for category, records in classified.items():
               for record in records:
                   self.conn.cursor().execute("""
                       INSERT INTO ops.dlq_records 
                       (topic, category, record_key, failure_reasons, 
                        original_timestamp, processed_at)
                       VALUES (%s, %s, %s, %s, %s, CURRENT_TIMESTAMP())
                   """, (
                       record.topic,
                       category,
                       record.key,
                       record.headers.get("quality_failures", ""),
                       record.headers.get("original_timestamp", ""),
                   ))
   ```

   ```python
   # platform/storage_adapter.py
   from abc import ABC, abstractmethod
   from typing import BinaryIO

   class ObjectStorageAdapter(ABC):
       """Abstraction over cloud object storage."""

       @abstractmethod
       def upload(self, bucket: str, key: str, data: BinaryIO) -> str:
           pass

       @abstractmethod
       def download(self, bucket: str, key: str) -> BinaryIO:
           pass

       @abstractmethod
       def list_objects(self, bucket: str, prefix: str) -> list[str]:
           pass

       @abstractmethod
       def delete(self, bucket: str, key: str) -> bool:
           pass

   class S3Adapter(ObjectStorageAdapter):
       def __init__(self):
           import boto3
           self.client = boto3.client("s3")

       def upload(self, bucket: str, key: str, data: BinaryIO) -> str:
           self.client.upload_fileobj(data, bucket, key)
           return f"s3://{bucket}/{key}"

       def download(self, bucket: str, key: str) -> BinaryIO:
           import io
           buf = io.BytesIO()
           self.client.download_fileobj(bucket, key, buf)
           buf.seek(0)
           return buf

       def list_objects(self, bucket: str, prefix: str) -> list[str]:
           resp = self.client.list_objects_v2(Bucket=bucket, Prefix=prefix)
           return [obj["Key"] for obj in resp.get("Contents", [])]

       def delete(self, bucket: str, key: str) -> bool:
           self.client.delete_object(Bucket=bucket, Key=key)
           return True

   class GCSAdapter(ObjectStorageAdapter):
       def __init__(self):
           from google.cloud import storage
           self.client = storage.Client()

       def upload(self, bucket: str, key: str, data: BinaryIO) -> str:
           blob = self.client.bucket(bucket).blob(key)
           blob.upload_from_file(data)
           return f"gs://{bucket}/{key}"

       def download(self, bucket: str, key: str) -> BinaryIO:
           import io
           blob = self.client.bucket(bucket).blob(key)
           buf = io.BytesIO()
           blob.download_to_file(buf)
           buf.seek(0)
           return buf

       def list_objects(self, bucket: str, prefix: str) -> list[str]:
           blobs = self.client.bucket(bucket).list_blobs(prefix=prefix)
           return [b.name for b in blobs]

       def delete(self, bucket: str, key: str) -> bool:
           self.client.bucket(bucket).blob(key).delete()
           return True

   def get_storage_adapter(cloud: str = "aws") -> ObjectStorageAdapter:
       adapters = {"aws": S3Adapter, "gcp": GCSAdapter}
       return adapters[cloud]()
   ```

   ```python
   # platform/secrets_adapter.py
   from abc import ABC, abstractmethod
   import os

   class SecretsAdapter(ABC):
       @abstractmethod
       def get_secret(self, name: str) -> str:
           pass

       @abstractmethod
       def set_secret(self, name: str, value: str) -> None:
           pass

   class AWSSecretsAdapter(SecretsAdapter):
       def __init__(self):
           import boto3
           self.client = boto3.client("secretsmanager")

       def get_secret(self, name: str) -> str:
           resp = self.client.get_secret_value(SecretId=name)
           return resp["SecretString"]

       def set_secret(self, name: str, value: str) -> None:
           self.client.put_secret_value(SecretId=name, SecretString=value)

   class VaultAdapter(SecretsAdapter):
       """HashiCorp Vault — cloud-agnostic."""
       def __init__(self, vault_url: str = None):
           import hvac
           self.client = hvac.Client(
               url=vault_url or os.environ["VAULT_ADDR"],
               token=os.environ["VAULT_TOKEN"],
           )

       def get_secret(self, name: str) -> str:
           resp = self.client.secrets.kv.v2.read_secret_version(path=name)
           return resp["data"]["data"]["value"]

       def set_secret(self, name: str, value: str) -> None:
           self.client.secrets.kv.v2.create_or_update_secret(
               path=name, secret={"value": value}
           )

   def get_secrets_adapter(backend: str = "aws") -> SecretsAdapter:
       adapters = {"aws": AWSSecretsAdapter, "vault": VaultAdapter}
       return adapters[backend]()
   ```

1. **Design the event sourcing model**:
   ```
   Traditional (current): Only latest state
   ┌─────────────────────────────────────┐
   │ orders table                        │
   │ order_id | total | status           │
   │ ORD-123  | $0    | refunded         │  ← History lost!
   └─────────────────────────────────────┘

   Event Sourced: Complete immutable history
   ┌─────────────────────────────────────────────────────────────┐
   │ order_events table                                          │
   │ event_id | order_id | event_type     | payload    | time    │
   │ EVT-001  | ORD-123  | order_created  | total=$150 | 10:00AM │
   │ EVT-002  | ORD-123  | item_added     | +$25       | 10:01AM │
   │ EVT-003  | ORD-123  | payment_auth   | $175       | 10:05AM │
   │ EVT-004  | ORD-123  | shipped        |            | 2:00PM  │
   │ EVT-005  | ORD-123  | refund_partial | -$50       | 3:15PM  │
   │ EVT-006  | ORD-123  | refund_full    | -$125      | 5:00PM  │
   └─────────────────────────────────────────────────────────────┘

   Now: "What was order ORD-123 worth at 2:30 PM?" → Replay events → $175
   ```

2. **Build the event store**:
   ```python
   # event_sourcing/event_store.py
   from dataclasses import dataclass, field
   from datetime import datetime, timezone
   from typing import Any
   import json
   import uuid

   @dataclass
   class DomainEvent:
       event_id: str
       aggregate_type: str   # "order", "payment", "refund"
       aggregate_id: str     # e.g., "ORD-123"
       event_type: str       # e.g., "order_created"
       payload: dict
       metadata: dict
       occurred_at: datetime
       version: int          # Monotonically increasing per aggregate

   class EventStore:
       def __init__(self, snowflake_conn, kafka_producer):
           self.conn = snowflake_conn
           self.kafka = kafka_producer

       def append(self, aggregate_type: str, aggregate_id: str,
                  event_type: str, payload: dict,
                  actor: str, reason: str = None) -> DomainEvent:
           """Append an immutable event. Never update or delete."""
           version = self._next_version(aggregate_type, aggregate_id)
           event = DomainEvent(
               event_id=str(uuid.uuid4()),
               aggregate_type=aggregate_type,
               aggregate_id=aggregate_id,
               event_type=event_type,
               payload=payload,
               metadata={
                   "actor": actor,
                   "reason": reason,
                   "source": "data_platform",
               },
               occurred_at=datetime.now(timezone.utc),
               version=version,
           )

           # Store in Snowflake (append-only table)
           self.conn.cursor().execute("""
               INSERT INTO financial_events.event_store
               (event_id, aggregate_type, aggregate_id, event_type,
                payload, metadata, occurred_at, version)
               VALUES (%s, %s, %s, %s, %s, %s, %s, %s)
           """, (
               event.event_id, event.aggregate_type, event.aggregate_id,
               event.event_type, json.dumps(event.payload),
               json.dumps(event.metadata), event.occurred_at.isoformat(),
               event.version,
           ))

           # Also publish to Kafka for real-time consumers
           self.kafka.produce(
               topic=f"events.{aggregate_type}",
               key=aggregate_id,
               value=json.dumps({
                   "event_id": event.event_id,
                   "event_type": event.event_type,
                   "aggregate_id": event.aggregate_id,
                   "payload": event.payload,
                   "occurred_at": event.occurred_at.isoformat(),
                   "version": event.version,
               }),
           )
           return event

       def get_events(self, aggregate_type: str, aggregate_id: str,
                      up_to: datetime = None) -> list[DomainEvent]:
           """Get all events for an aggregate, optionally up to a point in time."""
           query = """
               SELECT event_id, aggregate_type, aggregate_id, event_type,
                      payload, metadata, occurred_at, version
               FROM financial_events.event_store
               WHERE aggregate_type = %s AND aggregate_id = %s
           """
           params = [aggregate_type, aggregate_id]

           if up_to:
               query += " AND occurred_at <= %s"
               params.append(up_to.isoformat())

           query += " ORDER BY version ASC"
           rows = self.conn.cursor().execute(query, params).fetchall()

           return [
               DomainEvent(
                   event_id=r[0], aggregate_type=r[1], aggregate_id=r[2],
                   event_type=r[3], payload=json.loads(r[4]),
                   metadata=json.loads(r[5]),
                   occurred_at=datetime.fromisoformat(r[6]), version=r[7],
               )
               for r in rows
           ]

       def _next_version(self, aggregate_type: str,
                         aggregate_id: str) -> int:
           result = self.conn.cursor().execute("""
               SELECT COALESCE(MAX(version), 0) + 1
               FROM financial_events.event_store
               WHERE aggregate_type = %s AND aggregate_id = %s
           """, (aggregate_type, aggregate_id)).fetchone()
           return result[0]
   ```

3. **Build aggregate reconstruction (CQRS read model)**:

   ```python
   # event_sourcing/projections.py
   from datetime import datetime, timezone

   class OrderProjection:
       """Reconstruct order state from events."""

       def project(self, events: list) -> dict:
           """Replay events to get current (or point-in-time) state."""
           state = {
               "order_id": None,
               "customer_id": None,
               "items": [],
               "total": 0.0,
               "currency": None,
               "status": "unknown",
               "payments": [],
               "refunds": [],
               "last_updated": None,
               "event_count": 0,
           }

           for event in events:
               handler = getattr(self, f"_apply_{event.event_type}", None)
               if handler:
                   handler(state, event)
               state["last_updated"] = event.occurred_at.isoformat()
               state["event_count"] += 1

           return state

       def _apply_order_created(self, state: dict, event):
           state["order_id"] = event.aggregate_id
           state["customer_id"] = event.payload.get("customer_id")
           state["currency"] = event.payload.get("currency")
           state["status"] = "created"
           state["total"] = event.payload.get("initial_total", 0.0)

       def _apply_item_added(self, state: dict, event):
           item = event.payload.get("item", {})
           state["items"].append(item)
           state["total"] += item.get("price", 0.0)

       def _apply_item_removed(self, state: dict, event):
           item_id = event.payload.get("item_id")
           state["items"] = [i for i in state["items"] if i.get("id") != item_id]
           state["total"] -= event.payload.get("amount", 0.0)

       def _apply_payment_authorized(self, state: dict, event):
           state["payments"].append(event.payload)
           state["status"] = "paid"

       def _apply_shipped(self, state: dict, event):
           state["status"] = "shipped"

       def _apply_delivered(self, state: dict, event):
           state["status"] = "delivered"

       def _apply_refund_partial(self, state: dict, event):
           state["refunds"].append(event.payload)
           state["total"] -= event.payload.get("amount", 0.0)

       def _apply_refund_full(self, state: dict, event):
           state["refunds"].append(event.payload)
           state["total"] = 0.0
           state["status"] = "refunded"

   class AuditService:
       def __init__(self, event_store, projection: OrderProjection):
           self.store = event_store
           self.projection = projection

       def state_at(self, order_id: str, at_time: datetime) -> dict:
           """Answer: 'What was order X at time Y?'"""
           events = self.store.get_events("order", order_id, up_to=at_time)
           return self.projection.project(events)

       def full_audit_trail(self, order_id: str) -> list[dict]:
           """Complete history for auditors."""
           events = self.store.get_events("order", order_id)
           trail = []
           for event in events:
               state = self.projection.project(events[:events.index(event) + 1])
               trail.append({
                   "event": event.event_type,
                   "actor": event.metadata.get("actor"),
                   "reason": event.metadata.get("reason"),
                   "time": event.occurred_at.isoformat(),
                   "state_after": {"total": state["total"], "status": state["status"]},
               })
           return trail
   ```

   ```python
   # observability/trace_context.py
   import uuid
   from dataclasses import dataclass, field
   from datetime import datetime, timezone
   from typing import Optional

   @dataclass
   class TraceContext:
       trace_id: str
       span_id: str
       parent_span_id: Optional[str] = None
       baggage: dict = field(default_factory=dict)

       @classmethod
       def new_trace(cls) -> "TraceContext":
           return cls(
               trace_id=str(uuid.uuid4()),
               span_id=str(uuid.uuid4())[:8],
           )

       def child_span(self) -> "TraceContext":
           return TraceContext(
               trace_id=self.trace_id,
               span_id=str(uuid.uuid4())[:8],
               parent_span_id=self.span_id,
               baggage=self.baggage.copy(),
           )

       def to_kafka_headers(self) -> list[tuple[str, bytes]]:
           return [
               ("trace_id", self.trace_id.encode()),
               ("span_id", self.span_id.encode()),
               ("parent_span_id", (self.parent_span_id or "").encode()),
           ]

       @classmethod
       def from_kafka_headers(cls, headers: dict) -> "TraceContext":
           return cls(
               trace_id=headers.get("trace_id", str(uuid.uuid4())),
               span_id=str(uuid.uuid4())[:8],
               parent_span_id=headers.get("span_id"),
           )

   @dataclass
   class Span:
       trace_id: str
       span_id: str
       parent_span_id: Optional[str]
       service: str          # "cdc", "kafka", "flink", "dbt", "snowflake"
       operation: str        # "extract", "transform", "load"
       started_at: datetime
       ended_at: Optional[datetime] = None
       status: str = "ok"    # "ok", "error"
       attributes: dict = field(default_factory=dict)
       events: list = field(default_factory=list)

       @property
       def duration_ms(self) -> float:
           if self.ended_at and self.started_at:
               return (self.ended_at - self.started_at).total_seconds() * 1000
           return 0
   ```

   ```python
   # observability/pipeline_tracer.py
   from datetime import datetime, timezone

   class PipelineTracer:
       """Trace data records across the entire pipeline."""

       def __init__(self, span_store):
           self.store = span_store

       def start_span(self, ctx: TraceContext, service: str,
                      operation: str, **attrs) -> Span:
           span = Span(
               trace_id=ctx.trace_id,
               span_id=ctx.span_id,
               parent_span_id=ctx.parent_span_id,
               service=service,
               operation=operation,
               started_at=datetime.now(timezone.utc),
               attributes=attrs,
           )
           return span

       def end_span(self, span: Span, status: str = "ok", **attrs):
           span.ended_at = datetime.now(timezone.utc)
           span.status = status
           span.attributes.update(attrs)
           self.store.save_span(span)

       def trace_kafka_produce(self, ctx: TraceContext, topic: str,
                               record_key: str) -> Span:
           span = self.start_span(
               ctx, service="kafka", operation="produce",
               topic=topic, record_key=record_key,
           )
           return span

       def trace_flink_process(self, ctx: TraceContext,
                               job_name: str, records_in: int) -> Span:
           span = self.start_span(
               ctx, service="flink", operation="process",
               job_name=job_name, records_in=records_in,
           )
           return span

       def trace_dbt_model(self, ctx: TraceContext,
                           model_name: str, rows_affected: int) -> Span:
           span = self.start_span(
               ctx, service="dbt", operation="transform",
               model_name=model_name, rows_affected=rows_affected,
           )
           return span

   class TraceAnalyzer:
       """Analyze traces to find bottlenecks and failures."""

       def __init__(self, span_store):
           self.store = span_store

       def get_full_trace(self, trace_id: str) -> dict:
           spans = self.store.get_spans_by_trace(trace_id)
           return {
               "trace_id": trace_id,
               "span_count": len(spans),
               "total_duration_ms": self._total_duration(spans),
               "services": list(set(s.service for s in spans)),
               "errors": [s for s in spans if s.status == "error"],
               "slowest_span": max(spans, key=lambda s: s.duration_ms)
               if spans else None,
               "spans": sorted(spans, key=lambda s: s.started_at),
           }

       def find_slow_traces(self, threshold_ms: float,
                            hours: int = 24) -> list[dict]:
           """Find traces where end-to-end latency exceeds threshold."""
           traces = self.store.get_recent_traces(hours=hours)
           slow = []
           for trace_id in traces:
               trace = self.get_full_trace(trace_id)
               if trace["total_duration_ms"] > threshold_ms:
                   slow.append(trace)
           return sorted(slow, key=lambda t: t["total_duration_ms"],
                        reverse=True)

       def _total_duration(self, spans: list[Span]) -> float:
           if not spans:
               return 0
           start = min(s.started_at for s in spans)
           end = max(s.ended_at or s.started_at for s in spans)
           return (end - start).total_seconds() * 1000
   ```

   ```python
   # sox/reconciliation.py
   from dataclasses import dataclass
   from datetime import datetime, date
   from decimal import Decimal

   @dataclass
   class ReconciliationResult:
       control_name: str
       run_date: date
       source_value: Decimal
       target_value: Decimal
       difference: Decimal
       difference_pct: float
       status: str  # "pass", "fail", "warning"
       threshold: Decimal
       details: str

   class FinancialReconciler:
       def __init__(self, snowflake_conn):
           self.conn = snowflake_conn

       def reconcile_revenue(self, for_date: date) -> ReconciliationResult:
           """Compare data platform revenue with General Ledger."""
           # Source: Data platform (gold layer)
           platform_result = self.conn.cursor().execute("""
               SELECT SUM(revenue_usd) AS total_revenue
               FROM gold_finance.fact_revenue
               WHERE revenue_date = %s
               AND status = 'recognized'
           """, (for_date.isoformat(),)).fetchone()
           platform_revenue = Decimal(str(platform_result[0] or 0))

           # Target: General Ledger (Oracle, loaded via Fivetran)
           gl_result = self.conn.cursor().execute("""
               SELECT SUM(amount) AS total_revenue
               FROM raw_finance.general_ledger
               WHERE posting_date = %s
               AND account_code IN ('4000', '4010', '4020')  -- Revenue accounts
           """, (for_date.isoformat(),)).fetchone()
           gl_revenue = Decimal(str(gl_result[0] or 0))

           difference = abs(platform_revenue - gl_revenue)
           threshold = Decimal("100.00")  # $100 tolerance

           if gl_revenue > 0:
               diff_pct = float(difference / gl_revenue * 100)
           else:
               diff_pct = 0.0 if platform_revenue == 0 else 100.0

           status = "pass" if difference <= threshold else "fail"

           result = ReconciliationResult(
               control_name="daily_revenue_reconciliation",
               run_date=for_date,
               source_value=platform_revenue,
               target_value=gl_revenue,
               difference=difference,
               difference_pct=diff_pct,
               status=status,
               threshold=threshold,
               details=(
                   f"Platform: ${platform_revenue:,.2f} | "
                   f"GL: ${gl_revenue:,.2f} | "
                   f"Diff: ${difference:,.2f} ({diff_pct:.3f}%)"
               ),
           )

           # Log result for auditors
           self._log_result(result)
           return result

       def reconcile_record_counts(self, for_date: date) -> ReconciliationResult:
           """Ensure no records were lost in the pipeline."""
           source_count = self.conn.cursor().execute("""
               SELECT COUNT(*)
               FROM raw_commerce.orders_cdc
               WHERE DATE(_extracted_at) = %s
           """, (for_date.isoformat(),)).fetchone()[0]

           target_count = self.conn.cursor().execute("""
               SELECT COUNT(*)
               FROM gold_commerce.fact_orders
               WHERE DATE(order_date) = %s
           """, (for_date.isoformat(),)).fetchone()[0]

           difference = Decimal(str(abs(source_count - target_count)))
           threshold = Decimal(str(max(10, source_count * 0.001)))

           status = "pass" if difference <= threshold else "fail"

           result = ReconciliationResult(
               control_name="daily_record_count_reconciliation",
               run_date=for_date,
               source_value=Decimal(str(source_count)),
               target_value=Decimal(str(target_count)),
               difference=difference,
               difference_pct=float(
                   difference / max(Decimal(str(source_count)), Decimal("1")) * 100
               ),
               status=status,
               threshold=threshold,
               details=(
                   f"Source: {source_count} | Target: {target_count} | "
                   f"Diff: {int(difference)}"
               ),
           )
           self._log_result(result)
           return result

       def _log_result(self, result: ReconciliationResult):
           self.conn.cursor().execute("""
               INSERT INTO audit.reconciliation_log
               (control_name, run_date, source_value, target_value,
                difference, difference_pct, status, threshold, details,
                logged_at)
               VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, 
                       CURRENT_TIMESTAMP())
           """, (
               result.control_name, result.run_date.isoformat(),
               str(result.source_value), str(result.target_value),
               str(result.difference), result.difference_pct,
               result.status, str(result.threshold), result.details,
           ))
   ```

   ```python
   # sox/change_control.py
   class ChangeControl:
       """Enforce SOX change management for financial pipelines."""

       FINANCIAL_PATHS = [
           "models/gold/finance/",
           "models/gold/commerce/fact_revenue",
           "models/gold/commerce/fact_orders",
       ]

       def validate_pr(self, changed_files: list[str],
                       pr_metadata: dict) -> dict:
           """Validate that changes to financial models follow controls."""
           financial_changes = [
               f for f in changed_files
               if any(f.startswith(p) for p in self.FINANCIAL_PATHS)
           ]

           if not financial_changes:
               return {"status": "pass", "reason": "No financial model changes"}

           checks = []

           # Check 1: PR must have Finance team approval
           approvers = pr_metadata.get("approvers", [])
           finance_approved = any(
               a in approvers
               for a in ["finance-lead", "cfo", "finance-data"]
           )
           checks.append({
               "check": "finance_approval",
               "status": "pass" if finance_approved else "fail",
               "detail": f"Finance approval: {finance_approved}",
           })

           # Check 2: Must have dbt test changes alongside model changes
           test_files = [f for f in changed_files if "/tests/" in f]
           has_tests = len(test_files) > 0
           checks.append({
               "check": "test_coverage",
               "status": "pass" if has_tests else "fail",
               "detail": f"Test files changed: {len(test_files)}",
           })

           # Check 3: Must have JIRA ticket in PR description
           has_ticket = bool(pr_metadata.get("jira_ticket"))
           checks.append({
               "check": "change_ticket",
               "status": "pass" if has_ticket else "fail",
               "detail": f"JIRA ticket: {pr_metadata.get('jira_ticket', 'MISSING')}",
           })

           all_passed = all(c["status"] == "pass" for c in checks)
           return {
               "status": "pass" if all_passed else "fail",
               "financial_files_changed": financial_changes,
               "checks": checks,
           }
   ```

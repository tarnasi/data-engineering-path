## Task 216: SRE for Data — Error Budgets and Service Level Objectives

### Business Context
Last month, the executive dashboard was stale 3 times (total: 7 hours). The revenue reconciliation pipeline broke an SLA twice (Finance was furious). And a Kafka consumer lag spike caused real-time features to serve stale data for 2 hours. Each incident was a different root cause, but the pattern is clear: the platform has no formal reliability targets. When everything is "best effort," nothing is reliably available.

### Objective
Implement SRE practices for the data platform: define Service Level Objectives (SLOs) for critical data products, implement error budgets, and build automated SLO tracking.

### Technical Requirements

2. **Build the SLO tracking system**:
   ```python
   # sre/slo_tracker.py
   from dataclasses import dataclass
   from datetime import datetime, timedelta, timezone
   import yaml

   @dataclass
   class SLOStatus:
       name: str
       slo_target: float
       current_sli: float
       error_budget_total: float
       error_budget_remaining: float
       error_budget_consumed_pct: float
       status: str  # "healthy" | "warning" | "critical" | "exhausted"
       window_start: datetime
       window_end: datetime
       violations: list[dict]

   class SLOTracker:
       def __init__(self, slo_config_path: str, conn):
           with open(slo_config_path) as f:
               self.config = yaml.safe_load(f)
           self.conn = conn

       def check_all(self) -> list[SLOStatus]:
           """Check all SLOs and return current status."""
           results = []
           for slo_def in self.config["service_level_objectives"]:
               status = self._check_slo(slo_def)
               results.append(status)
           return results

       def _check_slo(self, slo_def: dict) -> SLOStatus:
           sli_type = slo_def["sli"]["type"]
           window_days = int(slo_def["slo"]["window"].replace("_days", ""))
           now = datetime.now(timezone.utc)
           window_start = now - timedelta(days=window_days)

           # Get events in window
           if sli_type == "freshness":
               events = self._get_freshness_events(
                   slo_def["data_product"], window_start, now
               )
           elif sli_type == "accuracy":
               events = self._get_accuracy_events(
                   slo_def["data_product"], window_start, now
               )
           elif sli_type == "latency":
               events = self._get_latency_events(
                   slo_def["data_product"], window_start, now
               )
           elif sli_type == "completion":
               events = self._get_completion_events(window_start, now)
           else:
               raise ValueError(f"Unknown SLI type: {sli_type}")

           total_events = len(events)
           good_events = sum(1 for e in events if e["is_good"])
           current_sli = good_events / max(total_events, 1) * 100

           target = slo_def["slo"]["target"]
           bad_events = total_events - good_events
           allowed_bad = total_events * (1 - target / 100)
           budget_remaining = max(0, allowed_bad - bad_events)
           budget_consumed = (
               (bad_events / max(allowed_bad, 1)) * 100
               if allowed_bad > 0 else (100 if bad_events > 0 else 0)
           )

           if budget_consumed >= 100:
               status = "exhausted"
           elif budget_consumed >= 80:
               status = "critical"
           elif budget_consumed >= 50:
               status = "warning"
           else:
               status = "healthy"

           violations = [e for e in events if not e["is_good"]]

           return SLOStatus(
               name=slo_def["name"],
               slo_target=target,
               current_sli=round(current_sli, 2),
               error_budget_total=round(allowed_bad, 1),
               error_budget_remaining=round(budget_remaining, 1),
               error_budget_consumed_pct=round(budget_consumed, 1),
               status=status,
               window_start=window_start,
               window_end=now,
               violations=violations[-10:],  # Last 10 violations
           )

       def _get_freshness_events(self, product, start, end) -> list[dict]:
           """Query freshness events from monitoring table."""
           rows = self.conn.execute("""
               SELECT measured_at, value_minutes, threshold_minutes,
                      CASE WHEN value_minutes <= threshold_minutes 
                           THEN TRUE ELSE FALSE END as is_good
               FROM platform_metrics.freshness_checks
               WHERE data_product = %s 
                 AND measured_at BETWEEN %s AND %s
               ORDER BY measured_at
           """, (product, start, end)).fetchall()
           return [
               {"timestamp": r[0], "value": r[1],
                "threshold": r[2], "is_good": r[3]}
               for r in rows
           ]
   ```

3. **Build the error budget policy**:

   ```markdown
   ## Error Budget Policy

   ### What Happens When Budget is Consumed

   | Budget Consumed | Actions |
   |:-:|---|
   | **0-50%** | Normal operations. Deploy freely. |
   | **50-80%** | ⚠️ Caution. Review recent incidents. Slow down risky deploys. |
   | **80-100%** | 🔴 Critical. Only deploy reliability improvements. Conduct incident review. |
   | **100%+** | 🛑 Budget exhausted. Freeze all non-critical changes. War room for reliability. |

   ### Budget Recovery
   - Budget replenishes as the window rolls forward (old violations drop off)
   - A 30-day window means a violation on Day 1 drops off on Day 31
   - Teams can "earn back" budget by improving reliability

   ### Escalation Path
   | Level | Trigger | Who's Involved |
   |-------|---------|---------------|
   | L1 | Budget > 50% | On-call engineer + pod lead |
   | L2 | Budget > 80% | Senior DE (you) + Eng Manager (Ravi) |
   | L3 | Budget exhausted | VP Engineering (Marcus) + all leads |
   ```

   ```python
   # sre/dashboard.py
   class SLODashboard:
       def __init__(self, tracker: SLOTracker):
           self.tracker = tracker

       def generate_weekly_report(self) -> str:
           """Generate weekly SLO report for engineering leadership."""
           statuses = self.tracker.check_all()

           report = "## Weekly SLO Report\n\n"
           report += "| SLO | Target | Current | Budget | Status |\n"
           report += "|-----|:------:|:-------:|:------:|:------:|\n"

           for s in sorted(statuses, key=lambda x: x.error_budget_consumed_pct,
                          reverse=True):
               emoji = {
                   "healthy": "🟢",
                   "warning": "🟡",
                   "critical": "🔴",
                   "exhausted": "⛔",
               }[s.status]
               report += (
                   f"| {s.name} | {s.slo_target}% | {s.current_sli}% | "
                   f"{s.error_budget_consumed_pct:.0f}% used | {emoji} |\n"
               )

           # Summary
           healthy = sum(1 for s in statuses if s.status == "healthy")
           report += f"\n**{healthy}/{len(statuses)} SLOs healthy.**\n"

           exhausted = [s for s in statuses if s.status == "exhausted"]
           if exhausted:
               report += "\n### Action Required\n"
               for s in exhausted:
                   report += (f"- **{s.name}**: Budget exhausted. "
                             f"Freeze deploys and investigate.\n")

           return report
   ```

2. **Quantify tech debt impact**:
   ```python
   # tech_debt/quantifier.py
   from dataclasses import dataclass

   @dataclass
   class TechDebtItem:
       id: str
       title: str
       category: str
       age_months: int
       velocity_impact_pct: float  # % of sprint capacity lost
       incident_risk: str  # "none" | "low" | "medium" | "high"
       fix_effort_days: float
       monthly_cost: float  # Direct cost (compute waste, etc.)
       compound_rate: float  # How much worse it gets per month

   class TechDebtQuantifier:
       def __init__(self, items: list[TechDebtItem]):
           self.items = items

       def total_velocity_impact(self) -> float:
           """Total % of team velocity lost to tech debt."""
           return sum(i.velocity_impact_pct for i in self.items)

       def total_monthly_cost(self) -> float:
           """Direct monthly cost of tech debt."""
           return sum(i.monthly_cost for i in self.items)

       def roi_ranking(self) -> list[dict]:
           """Rank items by ROI: impact saved per day of effort."""
           ranked = []
           for item in self.items:
               # Annualized impact: velocity cost + direct cost
               annual_velocity_cost = item.velocity_impact_pct * 800  # $800/sprint-pct
               annual_direct_cost = item.monthly_cost * 12
               total_annual_impact = annual_velocity_cost + annual_direct_cost

               roi = total_annual_impact / max(item.fix_effort_days, 0.5)
               ranked.append({
                   "id": item.id,
                   "title": item.title,
                   "annual_impact": round(total_annual_impact, 0),
                   "fix_days": item.fix_effort_days,
                   "roi_per_day": round(roi, 0),
                   "priority": (
                       "P0" if roi > 5000
                       else "P1" if roi > 2000
                       else "P2" if roi > 500
                       else "P3"
                   ),
               })

           return sorted(ranked, key=lambda x: x["roi_per_day"], reverse=True)

       def make_business_case(self) -> str:
           """Generate the business case for Marcus."""
           velocity_loss = self.total_velocity_impact()
           monthly_cost = self.total_monthly_cost()
           ranking = self.roi_ranking()
           p0_items = [r for r in ranking if r["priority"] == "P0"]
           p0_days = sum(r["fix_days"] for r in p0_items)
           p0_annual_impact = sum(r["annual_impact"] for r in p0_items)

           return f"""## Business Case: Tech Debt Reduction

### Current Impact
- **{velocity_loss:.0f}% of team velocity** lost to workarounds
- **${monthly_cost:,.0f}/month** in wasted compute and manual effort
- **{len(self.items)} known debt items** across 4 categories

### Proposal: Fix P0 Items (Highest ROI)
- **{len(p0_items)} items** to fix
- **{p0_days:.0f} engineering days** (~{p0_days/5:.0f} weeks)
- **${p0_annual_impact:,.0f}/year** in recovered velocity + savings
- **ROI: {p0_annual_impact / (p0_days * 800):.1f}x** return on engineering investment

### Approach
Not a "tech debt quarter." Instead: **20% of each sprint** allocated to 
debt reduction. At 2 days/sprint, P0 items are complete in {p0_days/2:.0f} sprints.

This is sustainable and doesn't stop feature delivery.
"""
   ```

3. **Build the post-mortem template and tracker**:
   ```python
   # incidents/tracker.py
   from dataclasses import dataclass, field
   from datetime import datetime, timedelta, timezone

   @dataclass
   class Incident:
       id: str
       severity: str  # SEV1-SEV4
       title: str
       description: str
       detected_at: datetime
       resolved_at: datetime | None
       root_cause: str
       mitigation: str
       action_items: list[dict] = field(default_factory=list)
       affected_products: list[str] = field(default_factory=list)
       detection_method: str = ""  # "automated" | "human" | "proactive"

       @property
       def ttd_minutes(self) -> float:
           """Time to detect (from issue start to detection)."""
           return 0  # Would need issue_started_at field

       @property
       def ttr_minutes(self) -> float | None:
           """Time to resolve."""
           if not self.resolved_at:
               return None
           return (self.resolved_at - self.detected_at).total_seconds() / 60

   class IncidentTracker:
       def __init__(self):
           self.incidents: list[Incident] = []

       def add(self, incident: Incident):
           self.incidents.append(incident)

       def quarterly_report(self, quarter_start: datetime,
                           quarter_end: datetime) -> str:
           """Generate incident metrics for Marcus."""
           incidents = [
               i for i in self.incidents
               if quarter_start <= i.detected_at <= quarter_end
           ]

           by_severity = {}
           for i in incidents:
               by_severity.setdefault(i.severity, []).append(i)

           report = "## Quarterly Incident Report\n\n"
           report += f"**Period**: {quarter_start.date()} to {quarter_end.date()}\n\n"
           report += "| Metric | Value |\n|--------|-------|\n"
           report += f"| Total incidents | {len(incidents)} |\n"

           for sev in ["SEV1", "SEV2", "SEV3", "SEV4"]:
               count = len(by_severity.get(sev, []))
               report += f"| {sev} incidents | {count} |\n"

           resolved = [i for i in incidents if i.ttr_minutes is not None]
           if resolved:
               avg_ttr = sum(i.ttr_minutes for i in resolved) / len(resolved)
               report += f"| Avg time to resolve | {avg_ttr:.0f} min |\n"

           automated = sum(
               1 for i in incidents if i.detection_method == "automated"
           )
           report += (
               f"| Automated detection rate | "
               f"{automated/max(len(incidents),1)*100:.0f}% |\n"
           )

           # Root cause categories
           report += "\n### Root Cause Breakdown\n"
           causes = {}
           for i in incidents:
               bucket = self._categorize_cause(i.root_cause)
               causes[bucket] = causes.get(bucket, 0) + 1

           for cause, count in sorted(causes.items(), key=lambda x: -x[1]):
               report += f"- **{cause}**: {count} incidents\n"

           return report

       def _categorize_cause(self, root_cause: str) -> str:
           root_cause_lower = root_cause.lower()
           if "deploy" in root_cause_lower or "release" in root_cause_lower:
               return "Deployment"
           elif "schema" in root_cause_lower:
               return "Schema Change"
           elif "infra" in root_cause_lower or "aws" in root_cause_lower:
               return "Infrastructure"
           elif "data" in root_cause_lower or "quality" in root_cause_lower:
               return "Data Quality"
           elif "capacity" in root_cause_lower or "scale" in root_cause_lower:
               return "Capacity"
           return "Other"
   ```

## Task 219: Zero-Downtime Schema Evolution — Managing Breaking Changes

### Business Context
The Commerce team wants to add a `discount_type` column to `fact_orders` and rename `total_amount` to `order_total`. The ML team's feature pipeline, the Finance reconciliation, and 4 dashboards all depend on the current schema. Last time someone renamed a column, 3 dashboards broke and nobody knew for 6 hours. Priya says: "We need a schema evolution process that guarantees zero downtime for consumers, even for breaking changes."

### Objective
Build a schema evolution framework that enables breaking changes (renames, deletes, type changes) without downstream breakage, using a versioned migration approach.

### Technical Requirements
- **Tools**: dbt, SQL (Snowflake), Python, YAML
- **Skills**: Schema evolution, backward compatibility, migration patterns, consumer management

### Step-by-Step Guidance

1. **Classify schema changes**:
   ```markdown
   ## Schema Change Classification

   ### Non-Breaking (Safe to Deploy Immediately)
   | Change | Why Safe | Example |
   |--------|----------|---------|
   | Add nullable column | Existing queries unaffected | `ALTER TABLE ADD discount_type VARCHAR` |
   | Add new table | No consumers yet | `CREATE TABLE dim_discounts` |
   | Widen column type | Values still valid | `INT → BIGINT` |
   | Add index/clustering | Performance only | `ALTER TABLE CLUSTER BY (order_date)` |

   ### Breaking (Requires Migration Process)
   | Change | Why Breaking | Example |
   |--------|-------------|---------|
   | Rename column | Consumers reference old name | `total_amount → order_total` |
   | Remove column | Consumers may use it | `DROP COLUMN legacy_flag` |
   | Change column type (narrowing) | Values may not fit | `VARCHAR(100) → VARCHAR(50)` |
   | Change grain | Row meaning changes | `1 row per order → 1 row per order line` |
   | Change primary key | Joins break | `order_id → (order_id, line_item_id)` |
   ```

2. **Build the migration framework**:
   ```python
   # schema/migration_manager.py
   from dataclasses import dataclass
   from enum import Enum
   from datetime import datetime, timezone

   class MigrationPhase(Enum):
       PLANNED = "planned"
       EXPAND = "expand"        # Add new, keep old
       MIGRATE = "migrate"      # Update consumers
       CONTRACT = "contract"    # Remove old
       COMPLETE = "complete"

   @dataclass
   class SchemaMigration:
       id: str
       table: str
       description: str
       change_type: str  # "rename", "remove", "retype", "regrain"
       phase: MigrationPhase
       old_schema: dict
       new_schema: dict
       consumers: list[str]
       consumers_migrated: list[str]
       created_at: datetime
       target_complete: datetime

   class MigrationManager:
       """Manage zero-downtime schema migrations."""

       def __init__(self, conn):
           self.conn = conn

       def plan_column_rename(self, table: str, old_name: str,
                               new_name: str) -> SchemaMigration:
           """Plan a column rename migration."""
           consumers = self._find_consumers(table, old_name)

           migration = SchemaMigration(
               id=f"MIG-{table}-{old_name}-to-{new_name}",
               table=table,
               description=f"Rename {table}.{old_name} → {new_name}",
               change_type="rename",
               phase=MigrationPhase.PLANNED,
               old_schema={"column": old_name},
               new_schema={"column": new_name},
               consumers=[c["name"] for c in consumers],
               consumers_migrated=[],
               created_at=datetime.now(timezone.utc),
               target_complete=datetime.now(timezone.utc),
           )

           print(f"Migration planned: {migration.description}")
           print(f"Affected consumers: {len(consumers)}")
           for c in consumers:
               print(f"  - {c['name']} ({c['type']}): {c['reference']}")

           return migration

       def execute_expand(self, migration: SchemaMigration):
           """Phase 1: Add new column, keep old (dual-write)."""
           if migration.change_type == "rename":
               old_col = migration.old_schema["column"]
               new_col = migration.new_schema["column"]

               # Add new column as alias/computed
               self.conn.execute(f"""
                   ALTER TABLE {migration.table} 
                   ADD COLUMN {new_col} AS {old_col}
               """)

               migration.phase = MigrationPhase.EXPAND
               print(f"EXPAND complete: {migration.table}.{new_col} "
                     f"added (alias of {old_col})")

       def check_consumer_migration(self, migration: SchemaMigration) -> dict:
           """Check which consumers have migrated to new schema."""
           old_col = migration.old_schema["column"]
           new_col = migration.new_schema["column"]

           report = {"total": len(migration.consumers), "migrated": 0,
                     "remaining": []}

           for consumer in migration.consumers:
               if self._consumer_uses_new_column(consumer, new_col):
                   report["migrated"] += 1
                   if consumer not in migration.consumers_migrated:
                       migration.consumers_migrated.append(consumer)
               else:
                   report["remaining"].append(consumer)

           return report

       def execute_contract(self, migration: SchemaMigration) -> bool:
           """Phase 3: Remove old column (only when all consumers migrated)."""
           check = self.check_consumer_migration(migration)

           if check["remaining"]:
               print(f"Cannot contract: {len(check['remaining'])} consumers "
                     f"still use old schema:")
               for c in check["remaining"]:
                   print(f"  - {c}")
               return False

           old_col = migration.old_schema["column"]
           self.conn.execute(
               f"ALTER TABLE {migration.table} DROP COLUMN {old_col}"
           )
           migration.phase = MigrationPhase.COMPLETE
           print(f"CONTRACT complete: {migration.table}.{old_col} removed")
           return True

       def _find_consumers(self, table: str, column: str) -> list[dict]:
           """Find all downstream consumers of a table.column."""
           consumers = []

           # dbt model references
           dbt_refs = self.conn.execute("""
               SELECT model_name, raw_sql 
               FROM dbt_metadata.model_definitions
               WHERE raw_sql ILIKE %s AND raw_sql ILIKE %s
           """, (f"%{table}%", f"%{column}%")).fetchall()

           for ref in dbt_refs:
               consumers.append({
                   "name": ref[0], "type": "dbt_model",
                   "reference": f"References {table}.{column}"
               })

           # Dashboard references
           dash_refs = self.conn.execute("""
               SELECT dashboard_name, query_text
               FROM metadata.dashboard_queries
               WHERE query_text ILIKE %s AND query_text ILIKE %s
           """, (f"%{table}%", f"%{column}%")).fetchall()

           for ref in dash_refs:
               consumers.append({
                   "name": ref[0], "type": "dashboard",
                   "reference": f"Query references {column}"
               })

           return consumers
   ```

3. **Build the consumer notification system**:
   ```python
   # schema/consumer_notifier.py
   class ConsumerNotifier:
       """Notify consumers about upcoming schema changes."""

       def notify_migration(self, migration, consumers: list[dict]):
           """Send migration notifications to affected consumers."""
           message = f"""
   ## Schema Migration Notice: {migration.id}

   **Table**: `{migration.table}`
   **Change**: {migration.description}

   **Deadline**: {migration.target_complete.strftime('%Y-%m-%d')}

   ### What You Need to Do
   Update your code to use the new column name:
   - Old: `{migration.old_schema.get('column', 'N/A')}`
   - New: `{migration.new_schema.get('column', 'N/A')}`

   Both columns will work until the deadline. After that, the old 
   column will be removed.

   ### Timeline
   | Phase | Date | Action |
   |-------|------|--------|
   | EXPAND | Now | New column available (dual-write) |
   | MIGRATE | 2 weeks | Update your references |
   | CONTRACT | {migration.target_complete.strftime('%Y-%m-%d')} | Old column removed |
   """

           for consumer in consumers:
               self._send_notification(consumer, message)

           print(f"Notifying {consumer['name']}: {message[:100]}...")
   ```

### Common Mistakes
- Direct renames — `ALTER TABLE RENAME COLUMN` breaks everything instantly; use expand-migrate-contract
- No consumer tracking — you can't contract if you don't know who uses the old schema
- No deadline — "migrate when you can" means nobody migrates; set a hard deadline

### Objective

3. **Build the review evaluation rubric**:
   ```python
   # reviews/evaluator.py
   from dataclasses import dataclass

   @dataclass
   class ReviewCriterion:
       name: str
       weight: float
       score: int  # 1-5
       notes: str

   class ArchitectureEvaluator:
       """Structured evaluation of architecture proposals."""

       CRITERIA = [
           ("problem_clarity", 0.15, "Is the problem well-defined and validated?"),
           ("solution_fit", 0.20, "Does the proposal solve the stated problem?"),
           ("alternatives", 0.10, "Were alternatives fairly evaluated?"),
           ("operational_cost", 0.15, "Is the operational burden acceptable?"),
           ("migration_risk", 0.15, "Is the migration plan safe and reversible?"),
           ("team_capability", 0.10, "Can the team execute this?"),
           ("strategic_alignment", 0.10, "Does this align with our tech strategy?"),
           ("security", 0.05, "Are security implications addressed?"),
       ]

       def evaluate(self, scores: dict[str, tuple[int, str]]) -> dict:
           """
           Evaluate a proposal.
           
           Args:
               scores: {criterion_name: (score_1_to_5, notes)}
           """
           results = []
           weighted_total = 0

           for name, weight, description in self.CRITERIA:
               if name not in scores:
                   raise ValueError(f"Missing score for: {name}")
               score, notes = scores[name]
               if not 1 <= score <= 5:
                   raise ValueError(f"Score must be 1-5, got {score}")
               results.append(ReviewCriterion(
                   name=name, weight=weight, score=score, notes=notes
               ))
               weighted_total += score * weight

           recommendation = (
               "Approve" if weighted_total >= 4.0
               else "Approve with changes" if weighted_total >= 3.0
               else "Defer" if weighted_total >= 2.0
               else "Reject"
           )

           return {
               "weighted_score": round(weighted_total, 2),
               "recommendation": recommendation,
               "criteria": results,
               "strengths": [r for r in results if r.score >= 4],
               "concerns": [r for r in results if r.score <= 2],
           }

       def format_review(self, proposal_title: str, evaluation: dict) -> str:
           """Format review as a readable document."""
           lines = [f"## Architecture Review: {proposal_title}\n"]
           lines.append(f"**Score**: {evaluation['weighted_score']}/5.0")
           lines.append(f"**Recommendation**: {evaluation['recommendation']}\n")

           lines.append("### Scores")
           lines.append("| Criterion | Score | Weight | Notes |")
           lines.append("|-----------|:-----:|:------:|-------|")
           for c in evaluation["criteria"]:
               lines.append(
                   f"| {c.name} | {'⭐' * c.score} | {c.weight:.0%} | {c.notes} |"
               )

           if evaluation["concerns"]:
               lines.append("\n### Concerns to Address")
               for c in evaluation["concerns"]:
                   lines.append(f"- **{c.name}** (score: {c.score}): {c.notes}")

           return "\n".join(lines)
   ```
